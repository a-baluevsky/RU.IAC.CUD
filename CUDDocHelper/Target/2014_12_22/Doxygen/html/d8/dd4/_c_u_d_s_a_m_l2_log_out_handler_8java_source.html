153 
154  if (server == null) {

156  }
157 
158  String sessionID = httpSession.getId();
159 
160  String statusIssuer = statusResponseType.getIssuer().getValue();
161  server.stack().deRegisterTransitParticipant(sessionID, statusIssuer);
162 
163  String nextParticipant = this.getParticipant(server, sessionID, relayState);
164  if (nextParticipant == null || nextParticipant.equals(relayState)) {

171  sts.cancelToken(samlProtocolContext);
172  httpSession.removeAttribute(GeneralConstants.ASSERTION);
173  }
174 
175  // TODO: check the in transit map for partial logouts
176 
177  try {
178  generateSuccessStatusResponseType(statusResponseType.getInResponseTo(), request, response, relayState);
179 
180  boolean isPost = isPostBindingForResponse(server, relayState, request);
181  response.setPostBindingForResponse(isPost);
182  } catch (Exception e) {
183  throw logger.processingError(e);
184  }
185  Map<String, Object> requestOptions = request.getOptions();

190  auditEvent.setType(PicketLinkAuditEventType.INVALIDATE_HTTP_SESSION);
191  auditEvent.setHttpSessionID(httpSession.getId());
192  auditHelper.audit(auditEvent);
193  }
194  httpSession.invalidate(); // We are done with the logout interaction
195  } else {
196  // Put the participant in transit mode
197  server.stack().registerTransitParticipant(sessionID, nextParticipant);
198 
199  boolean isPost = isPostBindingForResponse(server, nextParticipant, request);
200  response.setPostBindingForResponse(isPost);
201 
202  // send logout request to participant with relaystate to orig

204 
205  response.setDestination(nextParticipant);
206

211  response.setSendRequest(true);

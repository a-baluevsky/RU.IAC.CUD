001package ru.spb.iac.cud.services.application;
002
003import java.util.List;
004
005import javax.jws.WebMethod;
006import javax.jws.WebParam;
007import javax.jws.WebResult;
008import javax.jws.WebService;
009import ru.spb.iac.cud.exceptions.GeneralFailure;
010import ru.spb.iac.cud.items.app.AppAttribute;
011import ru.spb.iac.cud.items.app.AppAccept;
012
013@WebService(targetNamespace = ApplicationServiceImpl.NS)
014public interface ApplicationService {
015
016        public static final String NS = ApplicationServiceImpl.NS;
017
018        @WebMethod
019        @WebResult(targetNamespace = NS)
020        public AppAccept system_registration(
021                        @WebParam(name = "attributes", targetNamespace = NS) List<AppAttribute> attributes)
022                        throws GeneralFailure;
023
024        @WebMethod
025        @WebResult(targetNamespace = NS)
026        public AppAccept user_registration(
027                        @WebParam(name = "attributes", targetNamespace = NS) List<AppAttribute> attributes)
028                        throws GeneralFailure;
029
030        @WebMethod
031        @WebResult(targetNamespace = NS)
032        public AppAccept access_roles(
033                        @WebParam(name = "modeExec", targetNamespace = NS) String modeExec,
034                        @WebParam(name = "uidUser", targetNamespace = NS) String uidUser,
035                        @WebParam(name = "codesRoles", targetNamespace = NS) List<String> codesRoles)
036                        throws GeneralFailure;
037
038        @WebMethod
039        @WebResult(targetNamespace = NS)
040        public AppAccept access_groups(
041                        @WebParam(name = "modeExec", targetNamespace = NS) String modeExec,
042                        @WebParam(name = "uidUser", targetNamespace = NS) String uidUser,
043                        @WebParam(name = "codesGroups", targetNamespace = NS) List<String> codesGroups)
044                        throws GeneralFailure;
045
046        @WebMethod
047        @WebResult(targetNamespace = NS)
048        public AppAccept block(
049                        @WebParam(name = "modeExec", targetNamespace = NS) String modeExec,
050                        @WebParam(name = "uidUser", targetNamespace = NS) String uidUser,
051                        @WebParam(name = "blockUser", targetNamespace = NS) String blockReason)
052                        throws GeneralFailure;
053
054        @WebMethod
055        @WebResult(targetNamespace = NS)
056        public AppAccept system_modification(
057                        @WebParam(name = "attributes", targetNamespace = NS) List<AppAttribute> attributes)
058                        throws GeneralFailure;
059
060        @WebMethod
061        @WebResult(targetNamespace = NS)
062        public AppAccept user_modification(
063                        @WebParam(name = "uidUser", targetNamespace = NS) String uidUser,
064                        @WebParam(name = "attributes", targetNamespace = NS) List<AppAttribute> attributes)
065                        throws GeneralFailure;
066
067        @WebMethod
068        @WebResult(targetNamespace = NS)
069        public AppAccept user_identity_modification(
070                        @WebParam(name = "uidUser", targetNamespace = NS) String uidUser,
071                        @WebParam(name = "login", targetNamespace = NS) String login,
072                        @WebParam(name = "password", targetNamespace = NS) String password)
073                        throws GeneralFailure;
074
075        @WebMethod
076        @WebResult(targetNamespace = NS)
077        public AppAccept user_cert_modification(
078                        @WebParam(name = "modeExec", targetNamespace = NS) String modeExec,
079                        @WebParam(name = "uidUser", targetNamespace = NS) String uidUser,
080                        @WebParam(name = "certBase64", targetNamespace = NS) String certBase64)
081                        throws GeneralFailure;
082
083}

001package ru.spb.iac.cud.services.audit;
002
003import javax.xml.bind.JAXBElement;
004import javax.xml.bind.annotation.XmlElementDecl;
005import javax.xml.namespace.QName;
006
007import ru.spb.iac.cud.items.wrapper.Audit;
008import ru.spb.iac.cud.items.wrapper.SyncFunctions;
009
010 public class ObjectFactory {
011
012        private static final  QName _audit_QNAME = new QName(AuditServiceImpl.NS,
013                        "audit");
014        private static final  QName _sync_functions_QNAME = new QName(
015                        AuditServiceImpl.NS, "sync_functions");
016
017        public ObjectFactory() {
018        }
019
020        @XmlElementDecl(namespace = AuditServiceImpl.NS, name = "audit")
021        public JAXBElement<Audit> createSayHello(Audit value) {
022                return new JAXBElement<Audit>(_audit_QNAME, Audit.class, null, value);
023        }
024
025        @XmlElementDecl(namespace = AuditServiceImpl.NS, name = "sync_functions")
026        public JAXBElement<SyncFunctions> createSyncFunctions(SyncFunctions value) {
027                return new JAXBElement<SyncFunctions>(_sync_functions_QNAME,
028                                SyncFunctions.class, null, value);
029        }
030}

051 public class SignValidateSOAPHandler implements SOAPHandler<SOAPMessageContext> {
052
053        private static final Logger LOGGER = LoggerFactory.getLogger(SignValidateSOAPHandler.class);
054
055        private static PublicKey publicKey = null;
056        
057        private static PrivateKey privateKey = null;
058        
059        public Set<QName> getHeaders() {
060                return null;
061        }
062
063        public void close(MessageContext mc) {
064                LOGGER.debug("close");
065        }
066
067        public boolean handleFault(SOAPMessageContext mc) {
068                return true;
069        }
070
071        public boolean handleMessage(SOAPMessageContext mc) {
072
073                LOGGER.debug("handleMessage:01+:"
074                                + mc.get(MessageContext.MESSAGE_OUTBOUND_PROPERTY));
075
076                try {
077
078                        HttpServletRequest req = (HttpServletRequest) mc
079                                        .get(MessageContext.SERVLET_REQUEST);
080                        HttpSession http_session = req.getSession();
081
082                        String cud_sts_principal = null;
083
084                        PublicKey publicKey2 = null;
085
086                        X509Certificate cert_user = null;
087
088                        SOAPMessage sm = mc.getMessage();
089                        
090
091                        SOAPHeader soapHeader = sm.getSOAPHeader();
092                        SOAPBody soapBody = sm.getSOAPBody();
093
094                        // запрос
095                        if (Boolean.FALSE.equals(mc
096                                        .get(MessageContext.MESSAGE_OUTBOUND_PROPERTY))) {
097
098                                 
099
100                                // взятие ключа из хранилища доверенных
101
102                                NodeList assertionList = soapHeader.getElementsByTagNameNS("*",
103                                                "Assertion");
104
105                                if (assertionList == null || assertionList.getLength() == 0) {
106                                         
107                                        throw new GeneralFailure(
108                                                        "This service requires <saml:Assertion>, which is missing!!!");
109                                }
110
111                                NodeList subjectList = ((Element) assertionList.item(0))
112                                                .getElementsByTagNameNS("*", "Subject");
113
114                                if (subjectList == null || subjectList.getLength() == 0) {
115                                         
116                                        throw new GeneralFailure(
117                                                        "This service requires <Subject>, which is missing!!!");
118                                }
119
120                                NodeList nameIDList = ((Element) subjectList.item(0))
121                                                .getElementsByTagNameNS("*", "NameID");
122
123                                if (nameIDList == null || nameIDList.getLength() == 0) {
124                                         
125                                        throw new GeneralFailure(
126                                                        "This service requires <NameID>, which is missing!!!");
127                                }
128
129                        
130                                String codeSystem = nameIDList.item(0).getTextContent();
131
132                                 
133
134                                cud_sts_principal = codeSystem;
135                                http_session.setAttribute("cud_sts_principal",
136                                                cud_sts_principal);
137
138                         if(Configuration.isSignRequired()){
139                                /*
140                                 * //1-й вариант: взять код системы из Assertion/subject/NameID
141                                 * //по этому коду найти в базе сертификат
142                                 * 
143                                 * cert_/user = (n/ew
144                                 * Contex/tIDPUtilManager()).syst/em_cert(codeSystem);
145                                 */
146
147                                // 2-й вариант: взять прямо сертификат из
148                                // Assertion/subject/SubjectConfirmationData/X509Certificate
149
150                                NodeList x509CertificateList = ((Element) subjectList.item(0))
151                                                .getElementsByTagNameNS("*", "X509Certificate");
152
153                                if (x509CertificateList == null
154                                                || x509CertificateList.getLength() == 0) {
155                                         
156                                        throw new GeneralFailure(
157                                                        "This service requires SubjectConfirmationData/X509Certificate, which is missing!!!");
158                                }
159
160                                String x509CertSystem = x509CertificateList.item(0)
161                                                .getTextContent();
162
163                                 
164
165                                byte[] byteX509Certificate = Base64.decode(x509CertSystem);
166
167                                CertificateFactory cf = CertificateFactory.getInstance("X.509");
168                                ByteArrayInputStream bais = new ByteArrayInputStream(
169                                                byteX509Certificate);
170
171                                while (bais.available() > 0) {
172                                        cert_user = (X509Certificate) cf.generateCertificate(bais);
173                                }
174
175                                if (cert_user != null) {
176
177                                        publicKey2 = cert_user.getPublicKey();
178
179                                        /*
180                                         * cert_verify_sign =
181                                         * Han/dlerUtils.getStore().getCertificate(domainAlias);
182                                         * 
183                                         * 
184                                         * if(cer/t_verify_sign!=null){
185                                         * public/Key2=cert_veri/fy_sign.getPublicKey(); }
186                                         */
187                                } else {
188                                        throw new GeneralFailure(
189                                                        "Cannot to define a client public key!!!");
190                                }
191
192                                 
193
194                                Provider xmlDSigProvider = new ru.CryptoPro.JCPxml.dsig.internal.dom.XMLDSigRI();
195
196                                XMLSignatureFactory fac = XMLSignatureFactory.getInstance(
197                                                "DOM", xmlDSigProvider);
198
199                                Node securityNode = soapHeader.getFirstChild();
200
201                                if (securityNode == null) {
202                                         
203                                        throw new GeneralFailure(
204                                                        "This service requires <wsse:Security>, which is missing!!!");
205                                }
206
207                                NodeList securityNodeChilds = securityNode.getChildNodes();
208
209                                Node signatureNode1 = null;
210
211                                for (int i = 0; i < securityNodeChilds.getLength(); i++) {
212
213                                         
214
215                                        if (securityNodeChilds.item(i).getLocalName() != null
216                                                        && securityNodeChilds.item(i).getLocalName()
217                                                                        .equals("Signature")) {
218                                                signatureNode1 = securityNodeChilds.item(i);
219                                        }
220                                }
221
222                                 
223
224                                // signature
225
226                                if (signatureNode1 == null) {
227                                         
228                                        throw new GeneralFailure(
229                                                        "This service requires <dsig:Signature>, which is missing!!!");
230                                }
231
232                                 
233
234                                DOMValidateContext valContext1 = new DOMValidateContext(
235                                                publicKey2, signatureNode1);
236
237                                valContext1.putNamespacePrefix(XMLSignature.XMLNS, "dsig");
238
239                                valContext1
240                                                .setIdAttributeNS(
241                                                                soapHeader,
242                                                                "http://docs.oasis-open.org/wss/2004/01/oasis-200401-wss-wssecurity-utility-1.0.xsd",
243                                                                "Id");
244                                valContext1
245                                                .setIdAttributeNS(
246                                                                soapBody,
247                                                                "http://docs.oasis-open.org/wss/2004/01/oasis-200401-wss-wssecurity-utility-1.0.xsd",
248                                                                "Id");
249
250                                javax.xml.crypto.dsig.XMLSignature signature1 = fac
251                                                .unmarshalXMLSignature(valContext1);
252
253                                boolean result1 = signature1.validate(valContext1);
254
255                                LOGGER.debug("010:" + result1);
256
257                                if (result1 == false) {
258                                        throw new GeneralFailure("Signature is not valid!!!");
259                                }
260
261                          }
262                        } else {
263                                // ответ
264
265                                if(Configuration.isSignRequired()){
266                                        
267                                
268                                char[] signingKeyPass = "Access_Control".toCharArray();
269                                String signingAlias = "cudvm_export";
270
271                                
272                                if(publicKey==null){
273                                
274                                KeyStore ks = KeyStore.getInstance("HDImageStore", "JCP");
275                                ks.load(null, null);
276
277                                privateKey = (PrivateKey) ks.getKey(signingAlias,
278                                                signingKeyPass);
279
280                                Certificate cert = ks.getCertificate(signingAlias);
281                                publicKey = cert.getPublicKey();
282
283                                }
284                                if(!org.apache.xml.security.Init.isInitialized()){
285                                  org.apache.xml.security.Init.init();
286                                }
287                                Provider xmlDSigProvider = new ru.CryptoPro.JCPxml.dsig.internal.dom.XMLDSigRI();
288
289                                XMLSignatureFactory fac = XMLSignatureFactory.getInstance(
290                                                "DOM", xmlDSigProvider);
291
292                                Node SecuritySOAP = soapHeader.getFirstChild();
293
294                        
295                                soapBody.addNamespaceDeclaration(
296                                                "wsu",
297                                                "http://docs.oasis-open.org/wss/2004/01/oasis-200401-wss-wssecurity-utility-1.0.xsd");
298                                soapBody.setAttributeNS(
299                                                "http://docs.oasis-open.org/wss/2004/01/oasis-200401-wss-wssecurity-utility-1.0.xsd",
300                                                "wsu:Id", "Body");
301
302                                soapHeader
303                                                .addNamespaceDeclaration(
304                                                                "wsu",
305                                                                "http://docs.oasis-open.org/wss/2004/01/oasis-200401-wss-wssecurity-utility-1.0.xsd");
306                                soapHeader
307                                                .setAttributeNS(
308                                                                "http://docs.oasis-open.org/wss/2004/01/oasis-200401-wss-wssecurity-utility-1.0.xsd",
309                                                                "wsu:Id", "Header");
310
311                                List<Transform> transformList = new ArrayList<Transform>();
312                                Transform transform = fac.newTransform(Transform.ENVELOPED,
313                                                (XMLStructure) null);
314                                Transform transformC14N = fac.newTransform(
315                                                Transforms.TRANSFORM_C14N_EXCL_OMIT_COMMENTS,
316                                                (XMLStructure) null);
317                                transformList.add(transform);
318                                transformList.add(transformC14N);
319
320                                Reference ref1 = fac
321                                                .newReference(
322                                                                "#Header",
323                                                                fac.newDigestMethod(
324                                                                                "http://www.w3.org/2001/04/xmldsig-more#gostr3411",
325                                                                                null), transformList, null, null);
326                                Reference ref2 = fac.newReference("#Body", fac.newDigestMethod(
327                                                "http://www.w3.org/2001/04/xmldsig-more#gostr3411",
328                                                null), transformList, null, null);
329                                List<Reference> referenceList = new ArrayList<Reference>();
330
331                                referenceList.add(ref1);
332                                referenceList.add(ref2);
333
334                                SignedInfo si = fac
335                                                .newSignedInfo(
336                                                                fac.newCanonicalizationMethod(
337                                                                                CanonicalizationMethod.EXCLUSIVE,
338                                                                                (C14NMethodParameterSpec) null),
339                                                                fac.newSignatureMethod(
340                                                                                "http://www.w3.org/2001/04/xmldsig-more#gostr34102001-gostr3411",
341                                                                                null), referenceList);
342
343                                KeyInfoFactory kif = fac.getKeyInfoFactory();
344
345                                KeyValue kv = kif.newKeyValue(publicKey);
346                                KeyInfo ki = kif.newKeyInfo(Collections.singletonList(kv));
347
348                        
349                                javax.xml.crypto.dsig.XMLSignature sig = fac.newXMLSignature(
350                                                si, ki);
351
352                                // куда вставлять подпись
353                                DOMSignContext signContext = new DOMSignContext(privateKey,
354                                                SecuritySOAP);
355
356                                signContext.putNamespacePrefix(XMLSignature.XMLNS, "dsig");
357
358                                // фиксация аттрибута id в подписываемом элементе
359                                // место ответственное за факт появления Pre-digested input в
360                                // логе
361
362                                signContext
363                                                .setIdAttributeNS(
364                                                                soapBody,
365                                                                "http://docs.oasis-open.org/wss/2004/01/oasis-200401-wss-wssecurity-utility-1.0.xsd",
366                                                                "Id");
367                                signContext
368                                                .setIdAttributeNS(
369                                                                soapHeader,
370                                                                "http://docs.oasis-open.org/wss/2004/01/oasis-200401-wss-wssecurity-utility-1.0.xsd",
371                                                                "Id");
372
373                        
374                                sig.sign(signContext);
375
376                                // !!!
377                                // проверка
378                                // включать для проверки
379
380                                
381                                }
382                        }
383
384                } catch (Exception e) {
385                        LOGGER.error("handleMessage:error:", e);
386                        throw new ProtocolException(e);
387                }
388                return true;
389        }

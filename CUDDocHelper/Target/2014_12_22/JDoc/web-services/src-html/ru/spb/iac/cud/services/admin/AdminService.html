001package ru.spb.iac.cud.services.admin;
002
003import java.util.List;
004
005import javax.jws.WebMethod;
006import javax.jws.WebParam;
007import javax.jws.WebService;
008import ru.spb.iac.cud.exceptions.GeneralFailure;
009import ru.spb.iac.cud.items.Function;
010import ru.spb.iac.cud.items.Group;
011import ru.spb.iac.cud.items.Resource;
012import ru.spb.iac.cud.items.Role;
013
014@WebService(targetNamespace = AdminServiceImpl.NS)
015public interface AdminService {
016
017        public static final String NS = AdminServiceImpl.NS;
018
019        @WebMethod
020        public void sync_roles(
021                        @WebParam(name = "roles", targetNamespace = NS) List<Role> roles,
022                        @WebParam(name = "modeExec", targetNamespace = NS) String modeExec)
023                        throws GeneralFailure;
024
025        @WebMethod
026        public void access_roles(
027                        @WebParam(name = "uidsUsers", targetNamespace = NS) List<String> uidsUsers,
028                        @WebParam(name = "codesRoles", targetNamespace = NS) List<String> codesRoles,
029                        @WebParam(name = "modeExec", targetNamespace = NS) String modeExec)
030                        throws GeneralFailure;
031
032        @WebMethod
033        public void cert_change_sys(
034                        @WebParam(name = "newCert", targetNamespace = NS) String newCert)
035                        throws GeneralFailure;
036
037        @WebMethod
038        public void sync_functions(
039                        @WebParam(name = "functions", targetNamespace = NS) List<Function> functions,
040                        @WebParam(name = "modeExec", targetNamespace = NS) String modeExec)
041                        throws GeneralFailure;
042
043        @WebMethod
044        public void sync_groups(
045                        @WebParam(name = "groups", targetNamespace = NS) List<Group> groups,
046                        @WebParam(name = "modeExec", targetNamespace = NS) String modeExec)
047                        throws GeneralFailure;
048
049        @WebMethod
050        public void sync_groups_roles(
051                        @WebParam(name = "codesGroups", targetNamespace = NS) List<String> codesGroups,
052                        @WebParam(name = "codesRoles", targetNamespace = NS) List<String> codesRoles,
053                        @WebParam(name = "modeExec", targetNamespace = NS) String modeExec)
054                        throws GeneralFailure;
055
056        @WebMethod
057        public void access_groups(
058                        @WebParam(name = "uidsUsers", targetNamespace = NS) List<String> uidsUsers,
059                        @WebParam(name = "codesGroups", targetNamespace = NS) List<String> codesGroups,
060                        @WebParam(name = "modeExec", targetNamespace = NS) String modeExec)
061                        throws GeneralFailure;
062
063        @WebMethod
064        public void sync_resources(
065                        @WebParam(name = "resources", targetNamespace = NS) List<Resource> resources,
066                        @WebParam(name = "modeExec", targetNamespace = NS) String modeExec)
067                        throws GeneralFailure;
068
069        @WebMethod
070        public void sync_resources_roles(
071                        @WebParam(name = "codesResources", targetNamespace = NS) List<String> codesResources,
072                        @WebParam(name = "codesRoles", targetNamespace = NS) List<String> codesRoles,
073                        @WebParam(name = "modeExec", targetNamespace = NS) String modeExec)
074                        throws GeneralFailure;
075
076}

001package ru.spb.iac.cud.exs.shedule;
002
003import iac.cud.infosweb.dataitems.BaseParamItem;
004import iac.cud.infosweb.local.service.ServiceReestrAction;
005import iac.cud.infosweb.local.service.ServiceReestrPro;
006import iac.cud.infosweb.remote.frontage.IRemoteFrontageLocal;
007
008import java.util.Calendar;
009import java.util.concurrent.Executors;
010import java.util.concurrent.ScheduledExecutorService;
011import java.util.concurrent.TimeUnit;
012
013import javax.naming.Context;
014import javax.naming.InitialContext;
015
016import org.slf4j.Logger;
017import org.slf4j.LoggerFactory;
018
019 public class LaunchUCCertTask {
020
021        private static final Logger LOGGER = LoggerFactory
022                                .getLogger(LaunchUCCertTask.class);
023
024         
025        private static final ScheduledExecutorService scheduler = Executors
026                        .newScheduledThreadPool(1);
027
028        private static String jndiBinding = "java:global/InfoS-ear/InfoS-ejb/IRemoteFrontage!iac.cud.infosweb.remote.frontage.IRemoteFrontageLocal";
029
030        
031        
032        public static void initTask(int delaySeconds) {
033
034                LOGGER.debug("initTask");
035
036                scheduler.schedule(new Runnable() {
037
038                        public void run() {
039
040                                String  period = null;
041                                Long startSpan = null;
042
043                                try {
044
045                                        LOGGER.debug("initTask:run:01+");
046
047                                
048                                        // запускаем на автомате
049                                        // даже если админ не запустил процесс из консоли
050                                        // а соответственно нет файла
051
052                                        LOGGER.debug("initTask:run:02+");
053
054                                        // временное решение
055                                        // запуск на автомате без админа
056
057                                        LOGGER.debug("initTask:run:03+!+");
058
059                                
060                                        BaseParamItem bpi = new BaseParamItem(
061                                                        ServiceReestrPro.UCCert.name());
062
063                                        LOGGER.debug("initTask:run:05+");
064
065                                        bpi.put("gactiontype",
066                                                        ServiceReestrAction.PROCESS_START.name());
067
068                                        LOGGER.debug("initTask:run:06+");
069
070                                        period = "1";
071                                        startSpan = calendar();
072
073                                        LOGGER.debug("initTask:run:07+");
074
075                                        bpi.put("startSpan", startSpan);
076                                        bpi.put("period", new Long(period));
077
078                                        Context ctx = new InitialContext();
079                                        LOGGER.debug("initTask:run:02");
080                                        ((IRemoteFrontageLocal) ctx.lookup(jndiBinding)).run(bpi);
081                                        LOGGER.debug("initTask:run:03");
082
083                                } catch (Exception e) {
084                                        LOGGER.error("initTask:run:error:", e);
085                                } finally {
086                                        try {
087                                                scheduler.shutdown();
088                                        } catch (Exception e) {
089                                                LOGGER.error("initTask:run:finally:is:error:", e);
090                                        }
091                                }
092                        }
093                }, delaySeconds, TimeUnit.SECONDS);
094        }
095
096        public static Long calendar() {
097
098                Long currentTime = System.currentTimeMillis();
099
100                Calendar cln = Calendar.getInstance();
101                //cln.set(Calendar.HOUR_OF_DAY, 3);
102                cln.set(Calendar.HOUR_OF_DAY, 15);
103                cln.set(Calendar.MINUTE, 5);
104                cln.set(Calendar.SECOND, 0);
105                cln.set(Calendar.MILLISECOND, 0);
106
107                Long trans = cln.getTimeInMillis();
108
109                Long start = trans - currentTime;
110
111                if (start <= 0) {
112                        start = start + 24 * 60 * 60 * 1000;
113                }
114
115                LOGGER.debug("calendar:03:start:" + start);
116
117                return start;
118                                
119        }
120}

001package iac.grn.infosweb.servlet.support;
002
003import iac.grn.infosweb.session.support.SupportManager;
004
005import java.io.IOException;
006import java.io.PrintWriter;
007
008import javax.servlet.ServletConfig;
009import javax.servlet.ServletException;
010import javax.servlet.http.HttpServlet;
011import javax.servlet.http.HttpServletRequest;
012import javax.servlet.http.HttpServletResponse;
013
014import org.jboss.seam.Component;
015import org.jboss.seam.ScopeType;
016import org.jboss.seam.servlet.ContextualHttpServletRequest;
017import org.slf4j.Logger;
018import org.slf4j.LoggerFactory;
019
020 public class SupportServlet extends HttpServlet {
021
022        private static final Logger LOGGER = LoggerFactory.getLogger(SupportServlet.class);
023
024        private static final long serialVersionUID = 1L;
025
026        private String helpFio = null, helpPost = null, helpMail = null,
027                        helpText = null, helpTel = null;
028
029        public SupportServlet() {
030                super();
031        }
032
033        public void init(ServletConfig config) throws ServletException {
034                LOGGER.debug("SupportServlet:init:01");
035        }
036
037        protected void service(HttpServletRequest request,
038                        HttpServletResponse response) throws ServletException, IOException {
039                LOGGER.debug("SupportServlet:service:01");
040
041                final HttpServletRequest f_request = request;
042                final HttpServletResponse f_response = response;
043
044                try {
045                        response.setContentType("text/html;charset=UTF-8");
046                        PrintWriter out = response.getWriter();
047                        out.println("<html>");
048                        out.println("<head>");
049                        out.println("<title>SupportServlet</title>");
050                        out.println("</head>");
051                        out.println("<body>");
052                        out.println("<h1>SupportServlet</h1>");
053                        out.println("</body>");
054                        out.println("<script>");
055                        out.println("window.close();");
056                        out.println("</script>");
057                        out.println("</html>");
058
059                        helpFio = request.getParameter("help_fio");
060                        helpPost = request.getParameter("help_post");
061                        helpMail = request.getParameter("help_mail");
062                        helpText = request.getParameter("help_text");
063                        helpTel = request.getParameter("help_tel");
064
065                        LOGGER.debug("SupportServlet:service:help_fio:" + helpFio);
066                        LOGGER.debug("SupportServlet:service:help_post:" + helpPost);
067                        LOGGER.debug("SupportServlet:service:help_mail:" + helpMail);
068                        LOGGER.debug("SupportServlet:service:help_text:" + helpText);
069                        LOGGER.debug("SupportServlet:service:help_texl:" + helpTel);
070
071                        new ContextualHttpServletRequest(request) {
072                                @Override
073                                public void process() throws Exception {
074                                        doWork(f_request, f_response);
075                                }
076                        }.run();
077
078                } catch (Exception e) {
079                        LOGGER.error("SupportServlet:service:ERROR:", e);
080                }
081                LOGGER.debug("SupportServlet:service:02");
082        }
083
084        private void doWork(HttpServletRequest request, HttpServletResponse response) {
085                SupportManager sm = (SupportManager) Component.getInstance(
086                                "supportManager", ScopeType.EVENT);
087                sm.sendMail(helpFio, helpPost, helpMail, helpText, helpTel);
088
089        }
090        
091}

146        protected void registerResources() throws FacesException {
147                try {
148                        ClassLoader loader = Thread.currentThread().getContextClassLoader();
149                        Enumeration<URL> e = loader
150                                        .getResources("META-INF/resources-config.xml");
151                        while (e.hasMoreElements()) {
152                                URL resource = e.nextElement();
153                                registerConfig(resource);
154                        }
155                } catch (IOException e) {
156                        throw new FacesException(e);
157                }
158        }

160        private void registerConfig(URL resourceConfig) {
161                try {
162                        if (log.isDebugEnabled()) {
163                                log.debug("Process resources configuration file "
164                                                + resourceConfig.toExternalForm());
165                        }
166
167                        InputStream in = URLToStreamHelper.urlToStream(resourceConfig);
168                        try {
169                                Digester digester = new Digester();
170                                digester.setValidating(false);
171                                digester.setEntityResolver(new EntityResolver() {
172                                        // Dummi resolver - alvays do nothing
173                                        public InputSource resolveEntity(String publicId,
174                                                        String systemId) throws SAXException, IOException {
175                                                return new InputSource(new StringReader(""));
176                                        }
177                                });
178                                digester.setNamespaceAware(false);
179                                digester.setUseContextClassLoader(true);
180                                digester.push(this);
181                                digester.addObjectCreate("resource-config/resource", "class",
182                                                JarResource.class);
183                                digester.addObjectCreate("resource-config/resource/renderer",
184                                                "class", HTMLRenderer.class);
185                                digester.addCallMethod(
186                                                "resource-config/resource/renderer/content-type",
187                                                "setContentType", 0);
188                                digester.addSetNext("resource-config/resource/renderer",
189                                                "setRenderer", ResourceRenderer.class.getName());
190                                digester.addCallMethod("resource-config/resource/name",
191                                                "setKey", 0);
192                                digester.addCallMethod("resource-config/resource/path",
193                                                "setPath", 0);
194                                digester.addCallMethod("resource-config/resource/cacheable",
195                                                "setCacheable", 0);
196                                digester.addCallMethod(
197                                                "resource-config/resource/session-aware",
198                                                "setSessionAware", 0);
199                                digester.addCallMethod("resource-config/resource/property",
200                                                "setProperty", 2);
201                                digester.addCallParam("resource-config/resource/property/name",
202                                                0);
203                                digester.addCallParam(
204                                                "resource-config/resource/property/value", 1);
205                                digester.addCallMethod("resource-config/resource/content-type",
206                                                "setContentType", 0);
207                                digester.addSetNext("resource-config/resource", "addResource",
208                                                InternetResource.class.getName());
209                                digester.parse(in);
210                        } finally {
211                                in.close();
212                        }
213                } catch (IOException e) {
214                        throw new FacesException(e);
215                } catch (SAXException e) {
216                        throw new FacesException(e);
217                }
218        }

227        /**
228         * Base point for creating resource. Must detect type and build appropriate
229         * instance. Currently - make static resource for ordinary request, or
230         * instance of class.
231         * 
232         * @param base
233         *            base object for resource ( resourcess in classpath will be get
234         *            relative to it package )
235         * @param path
236         *            key - path to resource, resource class name etc.
237         * @return
238         * @throws FacesException
239         */
240        public InternetResource createResource(Object base, String path)
241                        throws FacesException {
242                // TODO - detect type of resource ( for example, resources location path
243                // in Skin
244                try {
245                        return getResource(path);
246                } catch (ResourceNotFoundException e) {
247                        try {
248                                return getResource(buildKey(base, path));
249                        } catch (ResourceNotFoundException e1) {
250                                if (log.isDebugEnabled()) {
251                                        log.debug(Messages.getMessage(Messages.BUILD_RESOURCE_INFO,
252                                                        path));
253                                }
254                        }
255                }
256                // path - is class name ?
257                InternetResource res;
258                try {
259                        Class<?> resourceClass = Class.forName(path);
260                        res = createDynamicResource(path, resourceClass);
261                } catch (Exception e) {
262                        try {
263                                res = createJarResource(base, path);
264                        } catch (ResourceNotFoundException ex) {
265                                res = createStaticResource(path);
266                        }
267                        // TODO - if resource not found, create static ?
268                }
269                return res;
270        }

272        private String buildKey(Object base, String path) {
273                if (path.startsWith("/")) {
274                        return path.substring(1);
275                }
276                if (null == base) {
277                        return path;
278                }
279                StringBuffer packageName = new StringBuffer(base.getClass()
280                                .getPackage().getName().replace('.', '/'));
281                return packageName.append("/").append(path).toString();
282        }

284        public String getUri(InternetResource resource, FacesContext context,
285                        Object storeData) {
286                
287                StringBuffer uri = new StringBuffer();
288                uri.append(resource.getKey());
289                // append serialized data as Base-64 encoded request string.
290                if (storeData != null) {
291                        try {
292                                byte[] objectData;
293                                if (storeData instanceof byte[]) {
294                                        objectData = (byte[]) storeData;
295                                        uri.append(DATA_BYTES_SEPARATOR);
296                                } else {
297                                        ByteArrayOutputStream dataSteram = new ByteArrayOutputStream(
298                                                        1024);
299                                        ObjectOutputStream objStream = new ObjectOutputStream(
300                                                        dataSteram);
301                                        objStream.writeObject(storeData);
302                                        objStream.flush();
303                                        objStream.close();
304                                        dataSteram.close();
305                                        objectData = dataSteram.toByteArray();
306                                        uri.append(DATA_SEPARATOR);
307                                }
308                                byte[] dataArray = encrypt(objectData);
309                                uri.append(new String(dataArray, "ISO-8859-1"));
310
311                        
312                                
313                        } catch (Exception e) {
314                                // Ignore errors, log it
315                                log.error(Messages
316                                                .getMessage(Messages.QUERY_STRING_BUILDING_ERROR), e);
317                        }
318                }
319                
320                boolean isGlobal = !resource.isSessionAware();
321                
322                String resourceURL = getWebXml(context).getFacesResourceURL(context,
323                                uri.toString(), isGlobal);
324                if (!isGlobal) {
325                        resourceURL = context.getExternalContext().encodeResourceURL(
326                                        resourceURL);
327                }
328                if (log.isDebugEnabled()) {
329                        log.debug(Messages.getMessage(Messages.BUILD_RESOURCE_URI_INFO,
330                                        resource.getKey(), resourceURL));
331                }
332                return resourceURL;
333
334        }

340        public InternetResource getResourceForKey(String key)
341                        throws ResourceNotFoundException {
342
343                Matcher matcher = DATA_SEPARATOR_PATTERN.matcher(key);
344                if (matcher.find()) {
345                        int data = matcher.start();
346                        key = key.substring(0, data);
347                }
348
349                return getResource(key);
350        }

352        public Object getResourceDataForKey(String key) {
353                
354                Object data = null;
355                String dataString = null;
356                Matcher matcher = DATA_SEPARATOR_PATTERN.matcher(key);
357                if (matcher.find()) {
358                        if (log.isDebugEnabled()) {
359                                log.debug(Messages.getMessage(
360                                                Messages.RESTORE_DATA_FROM_RESOURCE_URI_INFO, key,
361                                                dataString));
362                        }
363                        int dataStart = matcher.end();
364                        dataString = key.substring(dataStart);
365                        byte[] objectArray = null;
366                        byte[] dataArray;
367                        try {
368                                
369                                dataString=URLDecoder.decode(dataString,"ISO-8859-1");
370                                
371                                dataArray = dataString.getBytes("ISO-8859-1");
372                                objectArray = decrypt(dataArray);
373                        } catch (UnsupportedEncodingException e1) {
374                                // default encoding always presented.
375                        }
376                        if ("B".equals(matcher.group(1))) {
377                                data = objectArray;
378                        } else {
379                                try {
380                                        ObjectInputStream in = new ObjectInputStream(
381                                                        new ByteArrayInputStream(objectArray));
382                                        data = in.readObject();
383                                } catch (StreamCorruptedException e) {
384                                        log.error(Messages
385                                                        .getMessage(Messages.STREAM_CORRUPTED_ERROR), e);
386                                } catch (IOException e) {
387                                        log.error(Messages
388                                                        .getMessage(Messages.DESERIALIZE_DATA_INPUT_ERROR),
389                                                        e);
390                                } catch (ClassNotFoundException e) {
391                                        log
392                                                        .error(
393                                                                        Messages
394                                                                                        .getMessage(Messages.DATA_CLASS_NOT_FOUND_ERROR),
395                                                                        e);
396                                }
397                        }
398                }
399
400                return data;
401        }

436        /**
437         * Build resource for link to static context in webapp.
438         * 
439         * @param path
440         * @return
441         * @throws FacesException
442         */
443        protected InternetResource createStaticResource(String path)
444                        throws ResourceNotFoundException, FacesException {
445                FacesContext context = FacesContext.getCurrentInstance();
446                if (null != context) {
447                        try {
448                                URL in = context.getExternalContext().getResource(path);
449                                if (null != in) {
450                                        InternetResourceBase res = new StaticResource(path);
451                                        setRenderer(res, path);
452                                        // Detect last modification time.
453                                        res.setLastModified(new Date(getStartTime()));
454                                        addResource(path, res);
455                                        return res;
456                                }
457                        } catch (MalformedURLException e) {
458                                throw new ResourceNotFoundException(Messages.getMessage(
459                                                Messages.STATIC_RESOURCE_NOT_FOUND_ERROR, path),e);
460                        }
461                }
462                throw new ResourceNotFoundException(Messages.getMessage(
463                                Messages.STATIC_RESOURCE_NOT_FOUND_ERROR, path));
464        }

466        private void setRenderer(InternetResource res, String path)
467                        throws FacesException {
468                int lastPoint = path.lastIndexOf('.');
469                if (lastPoint > 0) {
470                        String ext = path.substring(lastPoint);
471                        ResourceRenderer resourceRenderer =  getRendererByExtension(ext);
472                        if (null != resourceRenderer) {
473                                res.setRenderer(resourceRenderer);
474                        } else {
475                                if (log.isDebugEnabled()) {
476                                        log.debug(Messages.getMessage(
477                                                        Messages.NO_RESOURCE_REGISTERED_ERROR_2, path,
478                                                        renderers.keySet()));
479                                }
480
481                                
482                                res.setRenderer(defaultRenderer);
483                        }
484                }
485        }

497         * Create resurce to send from classpath relative to base class.
498         * 
499         * @param base
500         * @param path
501         * @return
502         * @throws FacesException
503         */
504        protected InternetResource createJarResource(Object base, String path)
505                        throws ResourceNotFoundException, FacesException {
506                String key = buildKey(base, path);
507                ClassLoader loader = Thread.currentThread().getContextClassLoader();
508                if (null != loader.getResource(key)) {
509                        JarResource res = new JarResource(key);
510                        setRenderer(res, path);
511                        res.setLastModified(new Date(getStartTime()));
512                        addResource(key, res);
513                        return res;
514                } else {
515                        throw new ResourceNotFoundException(Messages.getMessage(
516                                        Messages.NO_RESOURCE_EXISTS_ERROR, key));
517                }
518
519        }

521        /**
522         * Create resource by instatiate given class.
523         * 
524         * @param path
525         * @param instatiate
526         * @return
527         */
528        protected InternetResource createDynamicResource(String path,
529                        Class<?> instatiate) throws ResourceNotFoundException {
530                if (InternetResource.class.isAssignableFrom(instatiate)) {
531                        InternetResource resource;
532                        try {
533                                resource = (InternetResource) instatiate.newInstance();
534                                addResource(path, resource);
535                        } catch (Exception e) {
536                                String message = Messages.getMessage(
537                                                Messages.INSTANTIATE_RESOURCE_ERROR, instatiate
538                                                                .toString());
539                                log.error(message, e);
540                                throw new ResourceNotFoundException(message, e);
541                        }
542                        return resource;
543                }
544                throw new FacesException(Messages
545                                .getMessage(Messages.INSTANTIATE_CLASS_ERROR));
546        }

579        private String getUserResourceKey(boolean cacheable, boolean session,
580                        String mime) {
581                StringBuffer pathBuffer = new StringBuffer(UserResource.class.getName());
582                pathBuffer.append(cacheable ? "/c" : "/n");
583                pathBuffer.append(session ? "/s" : "/n");
584                if (null != mime) {
585                        pathBuffer.append('/').append(mime.hashCode());
586                }
587                String path = pathBuffer.toString();
588                return path;
589        }

598        protected byte[] encrypt(byte[] src) {
599                try {
600                        Deflater compressor = new Deflater(Deflater.BEST_SPEED);
601                        byte[] compressed = new byte[src.length + 100];
602                        compressor.setInput(src);
603                        compressor.finish();
604                        int totalOut = compressor.deflate(compressed);
605                        byte[] zipsrc = new byte[totalOut];
606                        System.arraycopy(compressed, 0, zipsrc, 0, totalOut);
607                        compressor.end();
608                        return codec.encode(zipsrc);
609                } catch (Exception e) {
610                        throw new FacesException("Error encode resource data", e);
611                }
612        }
613
614        protected byte[] decrypt(byte[] src) {
615                try {
616                        
617                        byte[] zipsrc = codec.decode(src);
618                        Inflater decompressor = new Inflater();
619                        byte[] uncompressed = new byte[zipsrc.length * 5];
620                        decompressor.setInput(zipsrc);
621                        int totalOut = decompressor.inflate(uncompressed);
622                        byte[] out = new byte[totalOut];
623                        System.arraycopy(uncompressed, 0, out, 0, totalOut);
624                        decompressor.end();
625                        return out;
626                } catch (Exception e) {
627                        throw new FacesException("Error decode resource data", e);
628                }
629        }

001package iac.cud.infosweb.dataitems;
002
003 public class AppSystemItem extends AppItem {
004
005        private String fullName;
006        private String shortName;
007        private String description;
008
009        private Long idArm;
010        private String codeArm;
011        private String fullNameArm;
012        private String descriptionArm;
013
014        public AppSystemItem() {
015        }
016
017        public AppSystemItem(Long idApp, String created, int status,
018                        String fullName, String shortName, String description,
019                        String orgName, String usrFio, String rejectReason, Long idArm,
020                        String codeArm, String fullNameArm, String descriptionArm,
021                        String comment) {
022                super(idApp, created, status, orgName, usrFio, rejectReason, comment);
023                this.fullName = fullName;
024                this.shortName = shortName;
025                this.description = description;
026
027                this.idArm = idArm;
028                this.codeArm = codeArm;
029                this.fullNameArm = fullNameArm;
030                this.descriptionArm = descriptionArm;
031        }
032
033        public String getFullName() {
034                return fullName;
035        }
036
037        public void setFullName(String fullName) {
038                this.fullName = fullName;
039        }
040
041        public String getShortName() {
042                return shortName;
043        }
044
045        public void setShortName(String shortName) {
046                this.shortName = shortName;
047        }
048
049        public String getDescription() {
050                return description;
051        }
052
053        public void setDescription(String description) {
054                this.description = description;
055        }
056
057        public Long getIdArm() {
058                return idArm;
059        }
060
061        public void setIdArm(Long idArm) {
062                this.idArm = idArm;
063        }
064
065        public String getFullNameArm() {
066                return fullNameArm;
067        }
068
069        public void setFullNameArm(String fullNameArm) {
070                this.fullNameArm = fullNameArm;
071        }
072
073        public String getCodeArm() {
074                return codeArm;
075        }
076
077        public void setCodeArm(String codeArm) {
078                this.codeArm = codeArm;
079        }
080
081        public String getDescriptionArm() {
082                return descriptionArm;
083        }
084
085        public void setDescriptionArm(String descriptionArm) {
086                this.descriptionArm = descriptionArm;
087        }
088}

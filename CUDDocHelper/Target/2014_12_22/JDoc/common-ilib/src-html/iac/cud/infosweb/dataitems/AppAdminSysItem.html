001package iac.cud.infosweb.dataitems;
002
003 public class AppAdminSysItem extends AppItem {
004
005        private Long idArm;
006        private String codeArm;
007        private String fullNameArm;
008        private String descriptionArm;
009
010        private String nameOrg;
011
012        private Long idUser;
013        private String loginUser;
014        private String fioUser;
015        private String positionUser;
016
017        private String nameDep;
018
019        private int modeExec;
020
021        private String modeExecValue;
022
023        public AppAdminSysItem() {
024        }
025
026        public AppAdminSysItem(Long idApp, String created, int status,
027                        String orgName, String usrFio, String rejectReason, String comment,
028                        Long idArm, String codeArm, String fullNameArm,
029                        String descriptionArm, String nameOrg, Long idUser,
030                        String loginUser, String fioUser, String positionUser,
031                        String nameDep, int modeExec) {
032                super(idApp, created, status, orgName, usrFio, rejectReason, comment);
033
034                this.idArm = idArm;
035                this.codeArm = codeArm;
036                this.fullNameArm = fullNameArm;
037                this.descriptionArm = descriptionArm;
038
039                this.nameOrg = nameOrg;
040                this.idUser = idUser;
041                this.loginUser = loginUser;
042                this.fioUser = fioUser;
043                this.positionUser = positionUser;
044                this.nameDep = nameDep;
045
046                this.modeExec = modeExec;
047        }
048
049        public Long getIdArm() {
050                return idArm;
051        }
052
053        public void setIdArm(Long idArm) {
054                this.idArm = idArm;
055        }
056
057        public String getCodeArm() {
058                return codeArm;
059        }
060
061        public void setCodeArm(String codeArm) {
062                this.codeArm = codeArm;
063        }
064
065        public String getFullNameArm() {
066                return fullNameArm;
067        }
068
069        public void setFullNameArm(String fullNameArm) {
070                this.fullNameArm = fullNameArm;
071        }
072
073        public String getDescriptionArm() {
074                return descriptionArm;
075        }
076
077        public void setDescriptionArm(String descriptionArm) {
078                this.descriptionArm = descriptionArm;
079        }
080
081        public String getNameOrg() {
082                return nameOrg;
083        }
084
085        public void setNameOrg(String nameOrg) {
086                this.nameOrg = nameOrg;
087        }
088
089        public Long getIdUser() {
090                return idUser;
091        }
092
093        public void setIdUser(Long idUser) {
094                this.idUser = idUser;
095        }
096
097        public String getLoginUser() {
098                return loginUser;
099        }
100
101        public void setLoginUser(String loginUser) {
102                this.loginUser = loginUser;
103        }
104
105        public String getFioUser() {
106                return fioUser;
107        }
108
109        public void setFioUser(String fioUser) {
110                this.fioUser = fioUser;
111        }
112
113        public String getPositionUser() {
114                return positionUser;
115        }
116
117        public void setPositionUser(String positionUser) {
118                this.positionUser = positionUser;
119        }
120
121        public String getNameDep() {
122                return nameDep;
123        }
124
125        public void setNameDep(String nameDep) {
126                this.nameDep = nameDep;
127        }
128
129        public int getModeExec() {
130                return modeExec;
131        }
132
133        public void setModeExec(int modeExec) {
134                this.modeExec = modeExec;
135        }
136
137        public String getModeExecValue() {
138                if (this.modeExec == 0) {
139                        this.modeExecValue = "Замена";
140                } else if (this.modeExec == 1) {
141                        this.modeExecValue = "Добавление";
142                } else if (this.modeExec == 2) {
143                        this.modeExecValue = "Удаление";
144                }
145                return modeExecValue;
146        }
147
148        public void setModeExecValue(String modeExecValue) {
149                this.modeExecValue = modeExecValue;
150        }
151
152}

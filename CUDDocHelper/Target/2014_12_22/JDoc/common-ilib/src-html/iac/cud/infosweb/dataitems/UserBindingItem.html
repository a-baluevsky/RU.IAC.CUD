001package iac.cud.infosweb.dataitems;

003import java.util.List;
004
005 public class UserBindingItem extends UserItem {
006
007        private static final long serialVersionUID = 1L;
008
009        private List<BaseItem> bindingList;
010
011        // есть ли привязки полученные на автомате у пользователя
012        private int binFlag;
013
014        public UserBindingItem() {
015        }
016
017        public UserBindingItem(Long idUser, String login, String cert,
018                        String usrCode, String fio, String phone, String email,
019                        String position, String department, String orgCode, String orgName,
020                        String orgAdr, String orgTel, String start, String finish,
021                        Long status, String crtDate, String crtUserLogin, String updDate,
022                        String updUserLogin, String depCode, String orgIogvStatus,
023                        String usrIogvStatus, String depIogvStatus, Long iogvBindType,

025
026                super(idUser, login, cert, usrCode, fio, phone, email, position,
027                                department, orgCode, orgName, orgAdr, orgTel, start, finish,
028                                status, crtDate, crtUserLogin, updDate, updUserLogin, depCode,
029                                orgIogvStatus, usrIogvStatus, depIogvStatus, iogvBindType);
030
031                this.binFlag = binFlag;
032        }
033
034        public List<BaseItem> getBindingList() {
035                return bindingList;
036        }
037
038        public void setBindingList(List<BaseItem> bindingList) {
039                this.bindingList = bindingList;
040        }
041
042        public int getBinFlag() {

044        }
045
046        public void setBinFlag(int binFlag) {
047                this.binFlag = binFlag;
048        }
049
050}

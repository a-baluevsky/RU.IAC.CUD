001package iac.cud.infosweb.dataitems;
002
003import java.io.Serializable;
004
005 public class AuditFuncItem implements Serializable {
006
007        private static final long serialVersionUID = 1L;
008
009        private String user;
010
011        private String action;
012
013        private Long count;
014
015        public AuditFuncItem() {
016        }
017
018        public AuditFuncItem(String user, String action, Long count) {
019                this.user = user;
020                this.action = action;
021                this.count = count;
022        }
023
024        public String getUser() {
025                return this.user;
026        }
027
028        public void setUser(String user) {
029                this.user = user;
030        }
031
032        public String getAction() {
033                return this.action;
034        }
035
036        public void setAction(String action) {
037                this.action = action;
038        }
039
040        public Long getCount() {
041                return this.count;
042        }
043
044        public void setCount(Long count) {
045                this.count = count;
046        }
047
048}

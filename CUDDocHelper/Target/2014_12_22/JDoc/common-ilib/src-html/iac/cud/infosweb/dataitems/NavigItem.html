001package iac.cud.infosweb.dataitems;
002
003import java.util.List;
004
005 public class NavigItem {
006
007        private String idRes;
008        private String pageCode;
009        private String pageName;
010        private NavigItem parent;
011        private List<NavigItem> children;
012
013        public NavigItem() {
014        }
015
016        public NavigItem(String idRes, String pageCode, String pageName) {
017                this.idRes = idRes;
018                this.pageCode = pageCode;
019                this.pageName = pageName;
020        }
021
022        public String getIdRes() {
023                return idRes;
024        }
025
026        public void setIdRes(String idRes) {
027                this.idRes = idRes;
028        }
029
030        public String getPageCode() {
031                return pageCode;
032        }
033
034        public void setPageCode(String pageCode) {
035                this.pageCode = pageCode;
036        }
037
038        public String getPageName() {
039                return pageName;
040        }
041
042        public void setPageName(String pageName) {
043                this.pageName = pageName;
044        }
045
046        public NavigItem getParent() {
047                return parent;
048        }
049
050        public void setParent(NavigItem parent) {
051                this.parent = parent;
052        }
053
054        public List<NavigItem> getChildren() {
055                return children;
056        }
057
058        public void setChildren(List<NavigItem> children) {
059                this.children = children;
060        }
061}

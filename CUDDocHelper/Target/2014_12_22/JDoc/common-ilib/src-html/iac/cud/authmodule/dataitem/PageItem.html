001package iac.cud.authmodule.dataitem;
002
003import java.io.Serializable;
004import java.util.ArrayList;
005import java.util.List;
006
007/**
008 * Данные по наличию прав доступа к ресурсу приложения
009 * 
010 * @author bubnov
011 */
012 public class PageItem implements Serializable {
013
014        private static final long serialVersionUID = 1L;
015
016        private List<String> permList = new ArrayList<String>();
017
018        public PageItem() {
019        }
020
021        public PageItem(ArrayList<String> permList) {
022                this.permList = permList;
023        }
024
025        /**
026         * привилегии на ресурс
027         * 
028         * @return список доступных привилегий, если есть, иначе - empty
029         */
030        public List<String> getPermList() {
031                return this.permList;
032        }
033
034        public void setPermList(ArrayList<String> permList) {
035                this.permList = permList;
036        }
037
038}

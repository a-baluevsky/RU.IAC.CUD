001package ru.spb.iac.cud.items.wrapper;
002
003import java.util.List;
004
005import javax.xml.bind.annotation.XmlAccessType;
006import javax.xml.bind.annotation.XmlAccessorType;
007import javax.xml.bind.annotation.XmlElement;
008import javax.xml.bind.annotation.XmlType;
009
010import ru.spb.iac.cud.items.AuditFunction;
011
012@XmlAccessorType(XmlAccessType.FIELD)
013@XmlType(name = "audit", propOrder = { "login", "userFunctions" })
014// используется совместно с @RequestWrapper
015 public class Audit {
016
017        @XmlElement(name = "login", required = true, namespace = "http://audit.services.cud.iac.spb.ru/")
018        protected String login;
019
020        @XmlElement(name = "userFunctions", required = true, namespace = "http://audit.services.cud.iac.spb.ru/")
021        protected List<AuditFunction> userFunctions;
022
023        public String getLogin() {
024                return login;
025        }
026
027        public void setLogin(String login) {
028                this.login = login;
029        }
030
031        public List<AuditFunction> getUserFunctions() {
032                return userFunctions;
033        }
034
035        public void setUserFunctions(List<AuditFunction> userFunctions) {
036                this.userFunctions = userFunctions;
037        }
038
039}

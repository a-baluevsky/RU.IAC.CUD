001package ru.spb.iac.cud.items.app;
002
003 public class AppSystemResult extends AppResult {
004
005        private String codeArm;
006        private String fullNameArm;
007        private String descriptionArm;
008
009        public AppSystemResult() {
010        }
011
012        public AppSystemResult(String status, String codeArm, String fullNameArm,
013                        String descriptionArm, String rejectReason) {
014
015                super(status);
016                this.codeArm = codeArm;
017                this.fullNameArm = fullNameArm;
018                this.descriptionArm = descriptionArm;
019        }
020
021        public String getFullNameArm() {
022                return fullNameArm;
023        }
024
025        public void setFullNameArm(String fullNameArm) {
026                this.fullNameArm = fullNameArm;
027        }
028
029        public String getCodeArm() {
030                return codeArm;
031        }
032
033        public void setCodeArm(String codeArm) {
034                this.codeArm = codeArm;
035        }
036
037        public String getDescriptionArm() {
038                return descriptionArm;
039        }
040
041        public void setDescriptionArm(String descriptionArm) {
042                this.descriptionArm = descriptionArm;
043        }
044}

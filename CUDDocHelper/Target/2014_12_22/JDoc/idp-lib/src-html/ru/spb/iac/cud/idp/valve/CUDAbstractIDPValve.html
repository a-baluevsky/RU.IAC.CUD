333        @Override
334        public void invoke(Request request, Response response) throws IOException,
335                        ServletException {
336                // Look for unauthorized status
337                if (isUnauthorized(response)) {
338                        handleUnauthorizedResponse(request, response);
339                        return;
340                }
341
342                LOGGERSLF4J.debug("invoke:01");
343
344                if (isNotNull(request.getParameter(GeneralConstants.SAML_REQUEST_KEY))) {
345                        // считаем, что это новый запрос от »—
346
347                        if (request.getSessionInternal().getNote("overauth") != null) {
348                                request.getSessionInternal().removeNote("overauth");
349                                // сберегаем прозапас
350                                request.getSessionInternal().setNote(
351                                                GeneralConstants.SAML_REQUEST_KEY + "_overauth",
352                                                request.getSessionInternal().getNote(
353                                                                GeneralConstants.SAML_REQUEST_KEY));
354
355                                LOGGERSLF4J.debug("invoke:02:"
356                                                + request.getSession().getAttribute(
357                                                                "incoming_http_method"));
358
359                                request.getSession().setAttribute(
360                                                "incoming_http_method" + "_overauth",
361                                                request.getSession().getAttribute(
362                                                                "incoming_http_method"));
363                        }
364                }
365
366                // first, we populate all required parameters sent into session for
367                // later retrieval. If they exists.
368                populateSessionWithSAMLParameters(request);
369
370                // get an authenticated user or tries to authenticate if this is a
371                // authentication request
372                Principal userPrincipal = getUserPrincipal(request, response);
373
374                LOGGERSLF4J.debug("invoke:03:" + userPrincipal);
375
376                // we only handle SAML messages for authenticated users.
377                
378                if (userPrincipal != null
379                                && request.getSessionInternal().getNote("overauth") == null) {
380                        handleSAMLMessage(request, response);
381                } else if (isNotNull(request
382                                .getParameter(GeneralConstants.SAML_REQUEST_KEY))
383                                && isLogout(request)) {
384                        // logout
385                        LOGGERSLF4J.debug("invoke:04");
386                        try {
387                                // !!!
388                                // нужен reset, т.к. в фильтре перед этим может быть установлен
389                                // sendRedirect
390                                // хот€ € в фильтре и сделал ветку на logout, но на вс€кий
391                                // случай пусть reset() будет.
392                                response.reset();
393                                response.recycle();
394                                handleLogout(request, response);
395
396                        } finally {
397                                // cleanUpSessionNote(request);
398                                request.getSession().invalidate();
399                        }
400                } else if (isIsPassiveFailed(request)) {
401                        // нет активной сессии при isPassive запросе
402                        LOGGERSLF4J.debug("invoke:05");
403                        try {
404                                // response.reset();
405                                response.recycle();
406                                handleIsPassiveFailedResponse(request, response);
407
408                        } finally {
409                                resetIsPassiveFailed(request);
410                                cleanUpSessionNote(request);
411                        }
412                }
413                LOGGERSLF4J.debug("invoke:06");
414        }

416        private void handleSAMLMessage(Request request, Response response)
417                        throws IOException, ServletException {
418                if (hasSAML11Target(request)) {
419                        // We have SAML 1.1 IDP first scenario. Now we need to create a
420                        // SAMLResponse and send back
421                        // to SP as per target
422                        handleSAML11(request, response);
423                } else {
424                        Session session = request.getSessionInternal();
425
426                        String samlRequestMessage = (String) session
427                                        .getNote(GeneralConstants.SAML_REQUEST_KEY);
428                        String samlResponseMessage = (String) session
429                                        .getNote(GeneralConstants.SAML_RESPONSE_KEY);
430
431                        /**
432                         * Since the container has finished the authentication, we can
433                         * retrieve the original saml message as well as any relay state
434                         * from the SP
435                         */
436                        String relayState = (String) session
437                                        .getNote(GeneralConstants.RELAY_STATE);
438                        String signature = (String) session
439                                        .getNote(GeneralConstants.SAML_SIGNATURE_REQUEST_KEY);
440                        String sigAlg = (String) session
441                                        .getNote(GeneralConstants.SAML_SIG_ALG_REQUEST_KEY);
442
443                        if (LOGGER.isTraceEnabled()) {
444                                StringBuilder builder = new StringBuilder();
445                                builder.append("Retrieved saml messages and relay state from session");
446                                builder.append("saml Request message=").append(
447                                                samlRequestMessage);
448                                builder.append("::").append("SAMLResponseMessage=");
449                                builder.append(samlResponseMessage).append(":")
450                                                .append("relay state=").append(relayState);
451
452                                builder.append("Signature=").append(signature)
453                                                .append("::sigAlg=").append(sigAlg);
454                                LOGGER.trace(builder.toString());
455                        }
456
457                        if (isNotNull(samlRequestMessage)) {
458                                processSAMLRequestMessage(request, response);
459                        } else if (isNotNull(samlResponseMessage)) {
460                                processSAMLResponseMessage(request, response);
461                        } else if (request.getRequestURI().equals(
462                                        request.getContextPath() + "/")) {
463                                // no SAML processing and the request is asking for /.
464                                forwardHosted(request, response);
465                        }
466                }
467        }

482        private void forwardHosted(Request request, Response response)
483                        throws ServletException, IOException {
484                LOGGER.trace("SAML 1.1::Proceeding to IDP index page");
485                RequestDispatcher dispatch = getContext().getServletContext()
486                                .getRequestDispatcher(this.idpConfiguration.getHostedURI());
487
488                recycle(response);
489
490                try {
491                        includeResource(request, response, dispatch);

493                        // JBAS5.1 and 6 quirkiness
494                        includeResource(request.getRequest(), response, dispatch);
495                }

523        /**
524         * <p>
525         * SAML parameters are also populated into session if they are present in
526         * the request. This allows the IDP to retrieve them later when handling a
527         * specific SAML request or response.
528         * </p>
529         * 
530         * @param request
531         * @return
532         * @throws IOException
533         */
534        private void populateSessionWithSAMLParameters(Request request)
535                        throws IOException {
536                String samlRequestMessage = request
537                                .getParameter(GeneralConstants.SAML_REQUEST_KEY);
538                String samlResponseMessage = request
539                                .getParameter(GeneralConstants.SAML_RESPONSE_KEY);
540
541                boolean containsSAMLRequestMessage = isNotNull(samlRequestMessage);
542                boolean containsSAMLResponseMessage = isNotNull(samlResponseMessage);
543
544                String signature = request
545                                .getParameter(GeneralConstants.SAML_SIGNATURE_REQUEST_KEY);
546                String sigAlg = request
547                                .getParameter(GeneralConstants.SAML_SIG_ALG_REQUEST_KEY);
548                String relayState = request.getParameter(GeneralConstants.RELAY_STATE);
549
550                if (isNotNull(relayState)) {
551                        relayState = RedirectBindingUtil.urlDecode(relayState);
552                }
553
554                Session session = request.getSessionInternal();
555
556                if (containsSAMLRequestMessage || containsSAMLResponseMessage) {
557                        LOGGER.trace("Storing the SAMLRequest/SAMLResponse and RelayState in session");
558                        if (isNotNull(samlRequestMessage)) {
559                                session.setNote(GeneralConstants.SAML_REQUEST_KEY,
560                                                samlRequestMessage);
561                        }
562                        if (isNotNull(samlResponseMessage)) {
563                                session.setNote(GeneralConstants.SAML_RESPONSE_KEY,
564                                                samlResponseMessage);
565                        }
566                        if (isNotNull(relayState)) {
567                                session.setNote(GeneralConstants.RELAY_STATE, relayState.trim());
568                        }
569                        if (isNotNull(signature)) {
570                                session.setNote(GeneralConstants.SAML_SIGNATURE_REQUEST_KEY,
571                                                signature.trim());
572                        }
573                        if (isNotNull(sigAlg)) {
574                                session.setNote(GeneralConstants.SAML_SIG_ALG_REQUEST_KEY,
575                                                sigAlg.trim());
576                        }
577                }
578        }

580        /**
581         * <p>
582         * Handles an unauthorized response returned by a service provider.
583         * </p>
584         * 
585         * @param request
586         * @param response
587         * @throws IOException
588         * @throws ServletException
589         */
590        private void handleUnauthorizedResponse(Request request, Response response)
591                        throws IOException, ServletException {
592                IDPWebRequestUtil webRequestUtil = new IDPWebRequestUtil(request,
593                                idpConfiguration, keyManager);
594                Document samlErrorResponse = null;
595                String referer = request.getHeader("Referer");
596                String relayState = request.getParameter(GeneralConstants.RELAY_STATE);
597
598                if (isNotNull(relayState)) {
599                        relayState = RedirectBindingUtil.urlDecode(relayState);
600                }
601
602                try {
603                        samlErrorResponse = webRequestUtil.getErrorResponse(referer,
604                                        JBossSAMLURIConstants.STATUS_AUTHNFAILED.get(),
605                                        getIdentityURL(),
606                                        this.idpConfiguration.isSupportsSignature());
607
608                        WebRequestUtilHolder holder = webRequestUtil.getHolder();
609                        holder.setResponseDoc(samlErrorResponse).setDestination(referer)
610                                        .setRelayState(relayState).setAreWeSendingRequest(false)
611                                        .setPrivateKey(null).setSupportSignature(false)

2066         * provided for current SP and if metadata specifies that SAMLRequest is not

2068         * 
2069         * @param spIssuer
2070         * @return true if signature is not expected in SAMLRequest and so signature

2074                SPSSODescriptorType currentSPMetadata = spSSOMetadataMap.get(spIssuer);

2076                if (currentSPMetadata == null) {
2077                        return false;
2078                }

2081
2082                LOGGER.trace("Issuer: " + spIssuer + ", isRequestSigned: "
2083                                + isRequestSigned);
2084

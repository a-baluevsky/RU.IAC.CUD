045/**
046 * <p>
047 * This {@code LoginModule} authenticates clients by validating their SAML
048 * assertions with an external security token service (such as PicketLinkSTS).
049 * If the supplied assertion contains roles, these roles are extracted and
050 * included in the {@code Group} returned by the {@code getRoleSets} method.
051 * </p>
052 * <p>
053 * This module defines the following module options:
054 * <ul>
055 * <li>
056 * configFile - this property identifies the properties file that will be used
057 * to establish communication with the external security token service.</li>
058 * <li>
059 * cache.invalidation: set it to true if you require invalidation of JBoss Auth
060 * Cache at SAML Principal expiration.</li>
061 * <li>
062 * jboss.security.security_domain: name of the security domain where this login
063 * module is configured. This is only required if the cache.invalidation option
064 * is configured.</li>
065 * <li>
066 * roleKey: a comma separated list of strings that define the attributes in SAML
067 * assertion for user roles</li>
068 * <li>
069 * localValidation: if you want to validate the assertion locally for signature
070 * and expiry</li>
071 * <li>
072 * localValidationSecurityDomain: the security domain for the trust store
073 * information (via the JaasSecurityDomain)</li>
074 * <li>
075 * tokenEncodingType: encoding type of SAML token delivered via http request's
076 * header. Possible values are: base64 - content encoded as base64. In case of
077 * encoding will vary between base64 and gzip use base64 and LoginModule will
078 * detect gzipped data. gzip - gzipped content encoded as base64 none - content
079 * not encoded in any way</li>
080 * <li>
081 * samlTokenHttpHeader - name of http request header to fetch SAML token from.
082 * For example: "Authorize"</li>
083 * <li>
084 * samlTokenHttpHeaderRegEx - Java regular expression to be used to get SAML
085 * token from "samlTokenHttpHeader". Example: use: ."(.)".* to parse SAML token
086 * from header content like this: SAML_assertion="HHDHS=", at the same time set
087 * samlTokenHttpHeaderRegExGroup to 1.</li>
088 * <li>
089 * samlTokenHttpHeaderRegExGroup - Group value to be used when parsing out value
090 * of http request header specified by "samlTokenHttpHeader" using
091 * "samlTokenHttpHeaderRegEx".</li>
092 * </ul>
093 * </p>
094 * <p>
095 * Any properties specified besides the above properties are assumed to be used
096 * to configure how the {@code STSClient} will connect to the STS. For example,
097 * the JBossWS {@code StubExt.PROPERTY_SOCKET_FACTORY} can be specified in order
098 * to inform the socket factory that must be used to connect to the STS. All
099 * properties will be set in the request context of the {@code Dispatch}
100 * instance used by the {@code STSClient} to send requests to the STS.
101 * </p>
102 * <p>
103 * An example of a {@code configFile} can be seen bellow:
104 * 
105 * <pre>
106 * serviceName=PicketLinkSTS
107 * portName=PicketLinkSTSPort
108 * endpointAddress=http://localhost:8080/picketlink-sts/PicketLinkSTS
109 * username=JBoss
110 * password=JBoss
111 * </pre>
112 * 
113 * The first three properties specify the STS endpoint URL, service name, and
114 * port name. The last two properties specify the username and password that are
115 * to be used by the application server to authenticate to the STS and have the
116 * SAML assertions validated.
117 * </p>
118 * <p>
119 * <b>NOTE:</b> Sub-classes can use {@link #getSTSClient()} method to customize
120 * the {@link STSClient} class to make calls to STS/
121 * </p>
122 * 
123 * @author <a href="mailto:sguilhen@redhat.com">Stefan Guilhen</a>
124 * @author Anil.Saldhana@redhat.com
125 */

200         * org.jboss.security.auth.spi.AbstractServerLoginModule#initialize(javax

204        @Override
205        public void initialize(Subject subject, CallbackHandler callbackHandler,
206                        Map<String, ?> sharedState, Map<String, ?> options) {
207                super.initialize(subject, callbackHandler, sharedState, options);
208
209                LOGGERSLF4J.debug("initialize:01");
210
211                this.options.putAll(options);
212                this.rawOptions.putAll(options);
213
214                if (logger.isTraceEnabled()) {
215                        logger.trace(options.toString());
216                }
217                // save the config file and cache validation options, removing them from
218                // the map - all remaining properties will
219                // be set in the request context of the Dispatch instance used to send
220                // requests to the STS.
221                this.stsConfigurationFile = (String) this.options
222                                .remove(STS_CONFIG_FILE);
223                String cacheInvalidation = (String) this.options
224                                .remove("cache.invalidation");
225                if (cacheInvalidation != null && !cacheInvalidation.isEmpty()) {
226                        this.enableCacheInvalidation = Boolean
227                                        .parseBoolean(cacheInvalidation);
228
229                        this.securityDomain = (String) this.options
230                                        .remove(SecurityConstants.SECURITY_DOMAIN_OPTION);
231                        if (this.securityDomain == null || this.securityDomain.isEmpty()) {
232                                throw logger
233                                                .optionNotSet(SecurityConstants.SECURITY_DOMAIN_OPTION);
234                        }
235                }
236
237                String roleKeyStr = (String) options.get("roleKey");
238                if (StringUtil.isNotNull(roleKeyStr)) {
239                        roleKey = roleKeyStr.trim();
240                }
241
242                String localValidationStr = (String) options.get("localValidation");
243                if (StringUtil.isNotNull(localValidationStr)) {
244                        localValidation = Boolean.parseBoolean(localValidationStr);
245                        localValidationSecurityDomain = (String) options
246                                        .get("localValidationSecurityDomain");
247
248                        if (localValidationSecurityDomain == null) {
249                                logger.error(ErrorCodes.LOCAL_VALIDATION_SEC_DOMAIN_MUST_BE_SPECIFIED);
250                                throw logger.optionNotSet("localValidationSecurityDomain");
251                        }
252
253                        if (localValidationSecurityDomain.startsWith("java:") == false) {
254                                localValidationSecurityDomain = SecurityConstants.JAAS_CONTEXT_ROOT
255                                                + "/" + localValidationSecurityDomain;
256                        }
257
258                        String localTestingOnlyStr = (String) options
259                                        .get("localTestingOnly");
260                        if (StringUtil.isNotNull(localTestingOnlyStr)) {
261                                localTestingOnly = Boolean.valueOf(localTestingOnlyStr);
262                        }
263                }
264
265                LOGGERSLF4J.debug("initialize:02");
266        }

268        /*
269         * (non-Javadoc)
270         * 
271         * @see org.jboss.security.auth.spi.AbstractServerLoginModule#login()
272         */
273        @Override
274        public boolean login() throws LoginException {
275                // if shared data exists, set our principal and assertion variables.
276
277                LOGGERSLF4J.debug("login:01");
278
279                if (super.login()) {
280                        Object sharedPrincipal = super.sharedState
281                                        .get("javax.security.auth.login.name");
282                        if (sharedPrincipal instanceof Principal) {
283                                this.principal = (Principal) sharedPrincipal;
284                        } else {
285                                try {
286                                        this.principal = createIdentity(sharedPrincipal.toString());
287                                } catch (Exception e) {
288                                        throw logger.authFailedToCreatePrincipal(e);
289                                }
290                        }
291
292                        Object credential = super.sharedState
293                                        .get("javax.security.auth.login.password");
294                        if (credential instanceof SamlCredential) {
295                                this.credential = (SamlCredential) credential;
296                        } else {
297                                throw logger.authSharedCredentialIsNotSAMLCredential(credential
298                                                .getClass().getName());
299                        }
300                        return true;
301                }
302
303                LOGGERSLF4J.debug("login:02");
304
305                // obtain the assertion from the callback handler.
306                ObjectCallback callback = new ObjectCallback(null);
307                Element assertionElement = null;
308                try {
309                        if (getSamlTokenHttpHeader() != null) {
310                                this.credential = getCredentialFromHttpRequest();
311                        } else {
312                                LOGGERSLF4J.debug("login:02_2");
313
314                                super.callbackHandler.handle(new Callback[] { callback });
315
316                                if (callback.getCredential() instanceof String) {
317                                        callback.setCredential(new SamlCredential(DocumentUtil
318                                                        .getDocument(callback.getCredential().toString())
319                                                        .getDocumentElement()));
320                                }
321
322                                if (callback.getCredential() instanceof SamlCredential == false) {
323
324                                        LOGGERSLF4J.debug("login:02_3");
325
326                                        throw logger
327                                                        .authSharedCredentialIsNotSAMLCredential(callback
328                                                                        .getCredential().getClass().getName());
329                                }
330
331                                LOGGERSLF4J.debug("login:02_4");
332
333                                this.credential = (SamlCredential) callback.getCredential();
334                        }
335                        assertionElement = this.credential.getAssertionAsElement();
336                } catch (Exception e) {
337                        throw logger.authErrorHandlingCallback(e);
338                }
339
340                LOGGERSLF4J.debug("login:03");
341
342                // if there is no shared data, validate the assertion using the STS.
343                if (localValidation) {
344
345                        LOGGERSLF4J.debug("login:04");
346
347                        logger.trace("Local Validation is being Performed");
348                        try {
349                                boolean isValid = localValidation(assertionElement);
350
351                                LOGGERSLF4J.debug("login:05:" + isValid);
352                                if (isValid) {
353                                        logger.trace("Local Validation passed.");
354                                }
355                        } catch (Exception e) {
356                                LoginException le = new LoginException();
357                                le.initCause(e);
358                                throw le;
359                        }
360                } else {
361                        logger.trace("Local Validation is disabled. Verifying with STS");
362
363                        // sts config file has to be present to call STS (using sts client)
364                        if (this.stsConfigurationFile == null) {
365                                throw logger.authSTSConfigFileNotFound();
366                        }
367
368                        // send the assertion to the STS for validation.
369                        STSClient client = this.getSTSClient();
370                        try {
371                                boolean isValid = client.validateToken(assertionElement);
372                                // if the STS says the assertion is invalid, throw an exception
373                                // to signal that authentication has failed.
374                                if (isValid == false) {
375                                        throw logger.authInvalidSAMLAssertionBySTS();
376                                }
377                        } catch (WSTrustException we) {
378                                throw logger.authAssertionValidationError(we);
379                        }
380                }
381
382                LOGGERSLF4J.debug("login:06");
383
384                // if the assertion is valid, create a principal containing the
385                // assertion subject.
386                try {
387                        this.assertion = SAMLUtil.fromElement(assertionElement);
388                        SubjectType subject = assertion.getSubject();
389                        if (subject != null) {
390                                BaseIDAbstractType baseID = subject.getSubType().getBaseID();
391                                if (baseID instanceof NameIDType) {
392                                        NameIDType nameID = (NameIDType) baseID;
393                                        this.principal = new PicketLinkPrincipal(nameID.getValue());
394
395                                        // If the user has configured cache invalidation of subject
396                                        // based on saml token expiry
397                                        if (enableCacheInvalidation) {
398                                                TimeCacheExpiry cacheExpiry = this.getCacheExpiry();
399                                                XMLGregorianCalendar expiry = AssertionUtil
400                                                                .getExpiration(assertion);
401                                                if (expiry != null) {
402                                                        Date expiryDate = expiry.toGregorianCalendar()
403                                                                        .getTime();
404
405                                                        logger.trace("Creating Cache Entry for JBoss at ["
406                                                                        + new Date()
407                                                                        + "] , with expiration set to SAML expiry = "
408                                                                        + expiryDate);
409
410                                                        cacheExpiry.register(securityDomain, expiryDate,
411                                                                        principal);
412                                                } else {
413                                                        logger.samlAssertionWithoutExpiration(assertion
414                                                                        .getID());
415                                                }
416                                        }
417                                }
418                        }
419                } catch (Exception e) {
420                        throw logger.authFailedToParseSAMLAssertion(e);
421                }
422
423                LOGGERSLF4J.debug("login:07");
424
425                // if password-stacking has been configured, set the principal and the
426                // assertion in the shared map.
427                if (getUseFirstPass()) {
428                        super.sharedState.put("javax.security.auth.login.name",
429                                        this.principal);
430                        super.sharedState.put("javax.security.auth.login.password",
431                                        this.credential);
432                }
433                LOGGERSLF4J.debug("login:08");
434
435                return (super.loginOk = true);
436        }

443        @Override
444        public boolean commit() throws LoginException {
445                if (super.commit()) {
446                        final boolean added = subject.getPublicCredentials().add(
447                                        this.credential);
448                        if (added && logger.isTraceEnabled()) {
449                                logger.trace("Added Credential " + this.credential);
450                        }
451                        return true;
452                } else {
453                        return false;
454                }
455        }

489        /*
490         * (non-Javadoc)
491         * 
492         * @see org.jboss.security.auth.spi.AbstractServerLoginModule#getRoleSets()
493         */
494        @Override
495        protected Group[] getRoleSets() throws LoginException {
496                if (this.assertion == null) {
497                        try {
498                                this.assertion = SAMLUtil.fromElement(this.credential
499                                                .getAssertionAsElement());
500                        } catch (Exception e) {
501                                throw logger.authFailedToParseSAMLAssertion(e);
502                        }
503                }
504                if (logger.isTraceEnabled()) {
505                        try {
506                                logger.trace("Assertion from where roles will be sought = "
507                                                + AssertionUtil.asString(assertion));
508                        } catch (ProcessingException ignore) {
509                        }
510                }
511
512                List<String> roleKeys = new ArrayList<String>();
513                if (StringUtil.isNotNull(roleKey)) {
514                        roleKeys.addAll(StringUtil.tokenize(roleKey));
515                }
516
517                String groupName = SecurityConstants.ROLES_IDENTIFIER;
518                Group rolesGroup = new PicketLinkGroup(groupName);
519                List<String> roles = AssertionUtil.getRoles(assertion, roleKeys);
520                for (String role : roles) {
521                        rolesGroup.addMember(new SimplePrincipal(role));
522                }
523
524                return new Group[] { rolesGroup };
525        }

527        /**
528         * Get the {@link STSClient} object with which we can make calls to the STS
529         * 
530         * @return
531         */
532        protected STSClient getSTSClient() {
533                /*
534                 * Builder builder = new Builder(this.stsConfigurationFile); STSClient
535                 * client = new STSClient(builder.build());
536                 */
537
538                Builder builder = null;
539                STSClient client = null;
540                if (rawOptions.containsKey(STS_CONFIG_FILE)) {
541                        builder = new Builder(this.stsConfigurationFile);
542                        client = new STSClient(builder.build());
543                } else {
544                        builder = new Builder();
545                        builder.endpointAddress((String) rawOptions.get(ENDPOINT_ADDRESS));
546                        builder.portName((String) rawOptions.get(PORT_NAME)).serviceName(
547                                        (String) rawOptions.get(SERVICE_NAME));
548                        builder.username((String) rawOptions.get(USERNAME_KEY)).password(
549                                        (String) rawOptions.get(PASSWORD_KEY));
550
551                        String passwordString = (String) rawOptions.get(PASSWORD_KEY);
552                        if (passwordString != null
553                                        && passwordString
554                                                        .startsWith(PicketLinkFederationConstants.PASS_MASK_PREFIX)) {
555                                // password is masked
556                                String salt = (String) rawOptions
557                                                .get(PicketLinkFederationConstants.SALT);
558                                if (StringUtil.isNullOrEmpty(salt)) {
559                                        throw logger.optionNotSet("Salt");
560                                }
561
562                                String iCount = (String) rawOptions
563                                                .get(PicketLinkFederationConstants.ITERATION_COUNT);
564                                if (StringUtil.isNullOrEmpty(iCount)) {
565                                        throw logger.optionNotSet("Iteration Count");
566                                }
567
568                                int iterationCount = Integer.parseInt(iCount);
569                                try {
570                                        builder.password(StringUtil.decode(passwordString, salt,
571                                                        iterationCount));
572                                } catch (Exception e) {
573                                        throw logger.unableToDecodePasswordError(passwordString);
574                                }
575                        }
576                        client = new STSClient(builder.build());
577                }
578
579                // if the login module options map still contains any properties, assume
580                // they are for configuring the connection
581                // to the STS and set them in the Dispatch request context.
582                if (!this.options.isEmpty()) {
583                        Dispatch<Source> dispatch = client.getDispatch();
584                        for (Map.Entry<String, ?> entry : this.options.entrySet()) {
585                                dispatch.getRequestContext().put(entry.getKey(),
586                                                entry.getValue());
587                        }
588                }
589                return client;
590        }

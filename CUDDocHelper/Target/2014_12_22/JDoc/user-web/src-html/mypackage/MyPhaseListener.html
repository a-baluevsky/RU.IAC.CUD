001package mypackage;
002
003import javax.faces.event.PhaseEvent;
004import javax.faces.event.PhaseId;
005import javax.faces.event.PhaseListener;
006
007import org.slf4j.Logger;
008import org.slf4j.LoggerFactory;
009
010
011
012/**
013 * MyPhaseListener. The phase listener which prints the current phase to the system console.
014 *
015 * @author BalusC
016 * @link http://balusc.blogspot.com/2006/06/using-datatables.html
017 */
018
019 public class MyPhaseListener implements PhaseListener {
020
021        
022        private static final Logger LOGGER = LoggerFactory.getLogger(MyPhaseListener.class);
023        
024    private Long phaseTime;
025    private Long requestTime;
026    
027    public PhaseId getPhaseId() {
028        return PhaseId.ANY_PHASE;
029    }
030
031    /**
032     * This is invoked at the beginning of any JSF lifecycle phase.
033     * @param event The Phase Event.
034     */
035    public void beforePhase(PhaseEvent event) {
036        if(event.getPhaseId().equals(PhaseId.RESTORE_VIEW)){
037                requestTime=System.currentTimeMillis();
038        }
039        phaseTime=System.currentTimeMillis();
040        
041        LOGGER.debug("Phase Start: " + event.getPhaseId());
042       
043     }
044
045    /**
046     * This is invoked at the end of any JSF lifecycle phase.
047     * @param event The Phase Event.
048     */
049    public void afterPhase(PhaseEvent event) {
050        LOGGER.debug("Phase Finish: " + event.getPhaseId());
051        LOGGER.debug("Phase Time: " + (System.currentTimeMillis()-phaseTime));
052      if(event.getPhaseId().equals(PhaseId.RENDER_RESPONSE)){
053          LOGGER.debug("Request Time: " + (System.currentTimeMillis()-requestTime));
054         }
055   }
056}

001package ru.spb.iac.cud.uarm.web.context.reg;

003
004import java.io.Serializable;
005import javax.ejb.EJB;
006import javax.faces.bean.ManagedBean;
007import javax.faces.bean.ManagedProperty;
008import javax.faces.bean.RequestScoped;
009import javax.faces.context.FacesContext;
010import javax.servlet.http.HttpServletRequest;
011import javax.servlet.http.HttpSession;
012
013import org.slf4j.Logger;
014import org.slf4j.LoggerFactory;
015
016import ru.spb.iac.cud.uarm.ejb.context.reg.UserOrgManEJB;
017import ru.spb.iac.cud.uarm.ejb.entity.JournAppOrgManagerBssT;
018import ru.spb.iac.cud.uarm.util.CUDUserConsoleConstants;
019import ru.spb.iac.cud.uarm.web.context.user.UserSessionBean;
020
021@ManagedBean(name = "userOrgManBean")
022@RequestScoped
023 public class UserOrgManBean implements Serializable {
024
025        private static final Logger LOGGER = LoggerFactory.getLogger(UserOrgManBean.class);
026        
027        @EJB(beanName = "CUDUserConsole-ejb.jar#UserOrgManEJB")
028        private UserOrgManEJB userOrgManEJB;
029
030        @ManagedProperty("#{userSessionBean}")
031        private UserSessionBean userSessionBean;
032        
033        private static final long serialVersionUID = 1L;
034
035        private String reason;
036
037        private Long roleCUD;
038        
039        public void action() {
040
041                try {
042
043                        LOGGER.debug("userOrgManBean:action:01");
044
045                        
046                        
047                        HttpSession hs = (HttpSession)FacesContext.getCurrentInstance().getExternalContext().getSession(false); 
048                        Long authUserID = (Long) hs.getAttribute(CUDUserConsoleConstants.authUserID);
049                
050                        
051                        JournAppOrgManagerBssT t1 = new JournAppOrgManagerBssT();
052                        
053                        t1.setModeExec(1L); //ADD
054                        t1.setStatus(0L);
055                        //заявитель
056                        t1.setAcUsersKnlT2Long(authUserID);
057                        //кому назначаются роли
058                        t1.setAcUsersKnlT1Long(authUserID);
059                        
060                        userOrgManEJB.save(t1);
061
062                        FacesContext
063                                        .getCurrentInstance()
064                                        .getExternalContext()
065                                        .redirect(
066                                                        ((HttpServletRequest) FacesContext
067                                                                        .getCurrentInstance().getExternalContext()
068                                                                        .getRequest()).getContextPath()
069                                                                        + "/context/app/org_man/list.xhtml");
070
071                } catch (Exception e) {
072                        LOGGER.error("userOrgManBean:action:error:", e);
073                }
074        }
075
076        public String getReason() {
077                return reason;
078        }
079
080        public void setReason(String reason) {
081                this.reason = reason;
082        }
083
084        public UserSessionBean getUserSessionBean() {
085                return userSessionBean;
086        }
087
088        public void setUserSessionBean(UserSessionBean userSessionBean) {
089                this.userSessionBean = userSessionBean;
090        }
091
092        public Long getRoleCUD() {
093                return roleCUD;
094        }
095
096        public void setRoleCUD(Long roleCUD) {
097                this.roleCUD = roleCUD;
098        }
099
100}

023@ManagedBean(name="userForgotBean")
024@RequestScoped
025 public class UserForgotBean implements Serializable {
026 
027        private static final Logger LOGGER = LoggerFactory.getLogger(UserForgotBean.class);
028        
029        private static final long serialVersionUID = 1L;
030           
031    @EJB(beanName = "CUDUserConsole-ejb.jar#UserForgotEJB")
032        private UserForgotEJB userForgotEJB;
033        
034    private String userLogin;
035 
036    private String userEmail;
037    
038    private static final String userEmailsList = "userEmailsList";
039    
040    private String newPass;
041    private String reNewPass;
042    
043    public void action() {
044        
045        try{
046        
047        LOGGER.debug("UserForgotBean:action:01");
048        
049         String userNewPassword = newPass;
050         
051         String userReNewPassword = reNewPass;
052                 
053         LOGGER.debug("UserForgotBean:changePassword:03:"+userNewPassword);
054         LOGGER.debug("UserForgotBean:changePassword:04:"+userReNewPassword);
055         
056         
057         if(userNewPassword==null||userNewPassword.isEmpty()||
058          userReNewPassword==null||userReNewPassword.isEmpty()){
059          
060           LOGGER.debug("UserForgotBean:changePassword:05");
061          
062          FacesContext.getCurrentInstance().addMessage(null, 
063                                        new FacesMessage("Обязятельны все поля!"));
064           return;
065         }
066         
067         
068         Boolean latin = Pattern.matches("[^а-яА-Я]*", userNewPassword);
069       
070         if(!latin){
071          
072          LOGGER.debug("UserForgotBean:changePassword:06");
073          
074          FacesContext.getCurrentInstance().addMessage(null, 
075                                        new FacesMessage("В пароле не допустима кириллица!"));
076           return;
077         }
078         
079         if(!userNewPassword.equals(userReNewPassword)){
080          
081          LOGGER.debug("UserForgotBean:changePassword:07");
082          
083                        FacesContext.getCurrentInstance().addMessage(null, 
084                                        new FacesMessage("Пароли не совпадают!"));
085                 }else{
086         
087                        HttpSession hs = (HttpSession)FacesContext.getCurrentInstance().getExternalContext().getSession(false); 
088                        String userLogin = (String) hs.getAttribute(CUDUserConsoleConstants.userLoginForgot);
089                                
090                        LOGGER.debug("UserForgotBean:changePassword:08:"+userLogin);
091                        
092                        userForgotEJB.changePassword(userLogin, userNewPassword);
093                        
094                        FacesContext
095                        .getCurrentInstance()
096                        .getExternalContext()
097                        .redirect(
098                                        ((HttpServletRequest) FacesContext
099                                                        .getCurrentInstance().getExternalContext()
100                                                        .getRequest()).getContextPath()
101                                                        + "/welcome.xhtml");
102         }
103                 
104         LOGGER.debug("UserForgotBean:changePassword:09");
105      
106        }catch(Exception e){
107                LOGGER.error("UserForgotBean:action:error:"+e);
108        }
109   }
110
111   public void step1() {
112        
113           //здесь в 2-х случаях:
114           //1) на первом шаге (this.userEmail = null)
115           //2) пользователь уточнил свой email (this.userEmail = уточнённый email)
116     try{
117        
118        LOGGER.debug("UserForgotBean:step1:01");
119        
120        HttpSession hs = (HttpSession)FacesContext.getCurrentInstance().getExternalContext().getSession(false); 
121                
122        //th/is.use/rLogin!=null при шаге1
123        //th/is.us/erLogin==null при уточнении пользователем email
124        String userLoginFact=(this.userLogin!=null?this.userLogin:(String)hs.getAttribute(CUDUserConsoleConstants.userLoginForgot));
125        
126        HttpServletRequest request = (HttpServletRequest)FacesContext.getCurrentInstance().getExternalContext().getRequest();
127        String context_url=request.getScheme()+"://"+request.getServerName()+":"+request.getServerPort()+request.getContextPath();
128                        
129        List<String> emails = userForgotEJB.step1(userLoginFact, this.userEmail, context_url);
130        //при уточнении email, emails будет = этому уточненнному email  
131        
132        if(emails==null||emails.isEmpty()){
133                //у пользователя нет прикреплённых email
134                
135                LOGGER.debug("UserForgotBean:step1:02");
136                
137                  FacesContext.getCurrentInstance().getExternalContext().redirect(((HttpServletRequest)FacesContext.getCurrentInstance().getExternalContext().getRequest())
138                                .getContextPath()+"/context/forgot/pass_step1_message_not_email.xhtml");
139           
140        }else if(emails.size()>1){
141                //у пользователя не один email
142                //даём пользователю выбрать email
143                
144                LOGGER.debug("UserForgotBean:step1:03:"+emails.size());
145                
146                
147                 hs.setAttribute(userEmailsList, emails);
148                 
149                 hs.setAttribute(CUDUserConsoleConstants.userLoginForgot, this.userLogin);
150                        
151                  FacesContext.getCurrentInstance().getExternalContext().redirect(((HttpServletRequest)FacesContext.getCurrentInstance().getExternalContext().getRequest())
152                                .getContextPath()+"/context/forgot/pass_step1_message_more_email.xhtml");
153        
154                  
155        }else {
156                //нормальный вариант - один email
157                
158                LOGGER.debug("UserForgotBean:step1:04");
159                
160            FacesContext.getCurrentInstance().getExternalContext().redirect(((HttpServletRequest)FacesContext.getCurrentInstance().getExternalContext().getRequest())
161                        .getContextPath()+"/context/forgot/pass_step1_message.xhtml");
162   
163        }
164        
165      }catch(BaseError be){
166                LOGGER.error("UserForgotBean:step1:berror:"+be);
167                
168                if(CodesErrors.NOT_FOUND.equals(be.getCodeError())){
169                        //пользователь не определён
170                         try{
171                                 
172                           FacesContext.getCurrentInstance().addMessage(null, 
173                                        new FacesMessage("Пользователь не определён!"));
174                
175                         }catch(Exception e){
176                        LOGGER.error("UserForgotBean:step1:error2:"+e);
177                  }
178                        
179                }
180                
181          }catch(Exception e){
182                LOGGER.error("UserForgotBean:step1:error:"+e);
183      }
184     }

060        /**
061         * аутентификация пользователя по логин/паролю
062         */
063        public String authenticate_login(String login, String password,
064                        AuthMode authMode, String IPAddress, String codeSys)
065                        throws GeneralFailure, InvalidCredentials {
066
067                LOGGER.debug("authenticate_login:login:" + login);
068
069                 
070                
071                Object[] dataUser = null;
072
073                Long idUser = null;
074                String loginUser = null;
075
076                Long authModeValue = null;
077
078                if (authMode.equals(AuthMode.WEB_SERVICES)) {
079                        authModeValue = 2L;
080                } else if (authMode.equals(AuthMode.HTTP_REDIRECT)) {
081                        authModeValue = 97L;
082                } else if (authMode.equals(AuthMode.HTTP_REDIRECT_EXT_AUTH_OPEN)) {
083                        authModeValue = 99L;
084                } else if (authMode.equals(AuthMode.HTTP_REDIRECT_EXT_AUTH_ENCRYPT)) {
085                        authModeValue = 79L;
086                }
087
088                try {
089
090                        if (authModeValue.equals(78L)) {
091                                LOGGER.debug("authenticate:01");
092
093                                dataUser = (Object[]) em
094                                                .createNativeQuery(
095                                                                "select AU.ID_SRV, AU.login, AU.PASSWORD_ "
096                                                                                + "from "
097                                                                                + "AC_USERS_KNL_T au "
098                                                                                + "where AU.LOGIN=? "
099                                                                                + "and (AU.START_ACCOUNT is null or au.START_ACCOUNT <= sysdate) "
100                                                                                + "and (AU.START_ACCOUNT is null or au.START_ACCOUNT > sysdate) "
101                                                                                + "and AU.STATUS = 1 ")
102                                                .setParameter(1, login).getSingleResult();
103
104                                boolean matched = HashPassword.validatePassword(
105                                                (dataUser[2] != null ? dataUser[2].toString() : ""),
106                                                password);
107
108                                if (!matched) {
109                                        throw new NoResultException();
110                                }
111                                idUser = ((java.math.BigDecimal) dataUser[0]).longValue();
112                                loginUser = dataUser[1].toString();
113                                LOGGER.debug("authenticate:02");
114
115                        } else {
116
117                                LOGGER.debug("authenticate:03");
118
119                                dataUser = (Object[]) em
120                                                .createNativeQuery(
121                                                                "select AU.ID_SRV, AU.login "
122                                                                                + "from "
123                                                                                + "AC_USERS_KNL_T au "
124                                                                                + "where AU.LOGIN=? "
125                                                                                + "and AU.PASSWORD_=? "
126                                                                                + "and (AU.START_ACCOUNT is null or au.START_ACCOUNT <= sysdate) "
127                                                                                + "and (AU.START_ACCOUNT is null or au.START_ACCOUNT > sysdate) "
128                                                                                + "and AU.STATUS = 1 ")
129                                                .setParameter(1, login).setParameter(2, password)
130                                                .getSingleResult();
131
132                                idUser = ((java.math.BigDecimal) dataUser[0]).longValue();
133                                loginUser = dataUser[1].toString();
134
135                                LOGGER.debug("authenticate:04");
136                        }
137
138                        sys_audit(authModeValue, "login:" + login + "; passw:***", "true",
139                                        IPAddress, idUser, codeSys);
140
141                        return loginUser;
142
143                } catch (NoResultException ex) {
144
145                        sys_audit(authModeValue, "login:" + login + "; passw:" + password,
146                                        "false", IPAddress, null, codeSys);
147
148                        LOGGER.debug("authenticate:NoResultException");
149                        throw new InvalidCredentials("Учетной записи нет в системе!");
150                } catch (Exception e) {
151
152                        sys_audit(authModeValue, "login:" + login + "; passw:***", "error",
153                                        IPAddress, idUser, codeSys);
154
155                        LOGGER.error("authenticate:Error:", e);
156                        throw new GeneralFailure(e.getMessage());
157                }
158        }

284        public String authenticate_cert_sn(String sn, AuthMode authMode,
285                        String IPAddress, String codeSys) throws GeneralFailure,
286                        InvalidCredentials, RevokedCertificate {
287
288                LOGGER.debug("authenticate_cert_sn:01");
289
290                Object[] dataUser = null;
291
292                Long idUser = null;
293                String loginUser = null;
294                Long authModeValue = null;
295
296                if (authMode.equals(AuthMode.WEB_SERVICES)) {
297                        authModeValue = 62L;
298                } else if (authMode.equals(AuthMode.HTTP_REDIRECT)) {
299                        authModeValue = 98L;
300                }
301
302                if (sn == null) {
303                        sys_audit(2L, "sn:" + sn, "error:Serial Number is Empty!",
304                                        IPAddress, null, codeSys);
305                        throw new GeneralFailure("Serial Number is Empty!");
306                }
307
308                sn = sn.replaceAll(" ", "").toUpperCase();
309
310                try {
311
312                        LOGGER.debug("authenticate_cert_sn:02+:" + sn);
313
314                        // с учётом множественности сертификатов
315                        dataUser = ((Object[]) em
316                                        .createNativeQuery(
317                                                        "select AU.ID_SRV, AU.login "

320                                                                        + "AC_USERS_CERT_BSS_T auc "
321                                                                        + "where (UPPER(AU.CERTIFICATE)= UPPER(:certNum) or UPPER(AUC.CERT_NUM) = UPPER(:certNum)  ) "
322                                                                        + "and (AU.START_ACCOUNT is null or au.START_ACCOUNT <= sysdate) "
323                                                                        + "and (AU.START_ACCOUNT is null or au.START_ACCOUNT > sysdate) "
324                                                                        + "and AU.STATUS = 1 "
325                                                                        + "and AU.ID_SRV = AUC.UP_USER(+) ")
326                                        .setParameter("certNum", sn).getSingleResult());
327
328                        idUser = ((java.math.BigDecimal) dataUser[0]).longValue();
329                        loginUser = dataUser[1].toString();
330
331                        LOGGER.debug("authenticate_cert_sn:03+:" + loginUser);
332
333                        if (isRevoked(get_reestr("curr_crl"), sn)) {
334                                throw new RevokedCertificate("Сертификат отозван!");
335                        }
336
337                        LOGGER.debug("authenticate_cert_sn:04");
338
339                        LOGGER.debug("authenticate_cert_sn:06");
340
341                        sys_audit(authModeValue, "sn:" + sn, "true", IPAddress, idUser,
342                                        codeSys);
343
344                        LOGGER.debug("authenticate_cert_sn:07");
345
346                        return loginUser;
347
348                } catch (NoResultException ex) {
349
350                        sys_audit(authModeValue, "sn:" + sn, "false", IPAddress, null,
351                                        codeSys);
352
353                        LOGGER.debug("authenticate_cert_sn:NoResultException");
354                        throw new InvalidCredentials("Учетной записи нет в системе!");
355                } catch (RevokedCertificate er) {
356
357                        sys_audit(authModeValue, "sn:" + sn, "false, RevokedCertificate",
358                                        IPAddress, null, codeSys);
359
360                        LOGGER.debug("authenticate_cert_sn:RevokedCertificate");
361                        throw er;
362
363                } catch (Exception e) {
364
365                        sys_audit(authModeValue, "sn:" + sn, "error", IPAddress, idUser,
366                                        codeSys);
367
368                        LOGGER.error("authenticate_cert_sn:Error+:", e);
369
370                        throw new GeneralFailure(e.getMessage());
371                }

374        /**
375         * аудит действий пользователя
376         */
377        public void audit_pro(String codeSys, String login,
378                        List<AuditFunction> userFunctions, Long idUserAuth, String IPAddress)
379                        throws GeneralFailure {
380
381                LOGGER.debug("audit:01");
382
383                Map<String, Long> actcl = new HashMap<String, Long>();
384                Long idFunc = null;
385                Long idIS = null;
386                Long idUserSubject = null;
387
388                if (userFunctions == null || userFunctions.isEmpty()) {
389                        LOGGER.debug("audit:return");
390                        return;
391                }
392
393                try {
394
395                        if (codeSys == null) {
396                                LOGGER.debug("audit:System Code is null");
397                                throw new GeneralFailure("System Code is null");
398                        }
399                        if (!codeSys.startsWith(CUDConstants.armPrefix)&&!codeSys.startsWith(CUDConstants.subArmPrefix)) {
400                                LOGGER.debug("audit:System Code is incorrect");
401                                throw new GeneralFailure("System Code is incorrect");
402                        }
403                        if (login == null) {
404                                LOGGER.debug("audit:User login is nul");
405                                throw new GeneralFailure("User login is null");
406                        }
407
408                        idIS = get_id_is(codeSys);
409                        // решили определять пользователей извне ЦУД по их ИД, а не логинам
410                        
411                        idUserSubject = new Long(login);
412
413                        LOGGER.debug("audit:02:idIS:" + idIS);
414                        LOGGER.debug("audit:03:idUserSubject:" + idUserSubject);
415
416                        List<Object[]> lo = em
417                                        .createNativeQuery(
418                                                        "select ACT.SIGN_OBJECT, ACT.ID_SRV "
419                                                                        + "from ACTIONS_BSS_T act, "
420                                                                        + "AC_IS_BSS_T app "
421                                                                        + "where APP.ID_SRV = ACT.UP_IS "
422                                                                        + "and APP.ID_SRV=?").setParameter(1, idIS)
423                                        .getResultList();
424                        LOGGER.debug("audit:04");
425
426                        for (Object[] objectArray : lo) {
427                                actcl.put(
428                                                objectArray[0] != null ? objectArray[0].toString() : "",
429                                                objectArray[1] != null ? new Long(objectArray[1]
430                                                                .toString()) : -1L);
431                        }
432
433                        
434                        //для отчёта аудита вместе с ролями
435                        //сохраняем имеющиеся у пользователя роли
436                        
437                        String idRec =(String) em.createNativeQuery("select to_char(ACTIONS_LOG_ROLES_KNL_SEQ.nextval) sgnc from dual").getSingleResult();
438                      
439                        em.createNativeQuery(
440                                        "insert into ACTIONS_LOG_ROLES_KNL_T(ID_SRV, ROLES_NAMES, ROLES_CODES, created)  " + 
441                                        "select  :idRec ,  " + 
442                                        "t1.names, t1.codes, sysdate " + 
443                                        "from( " + 
444                                        "select " + 
445                                        "     rtrim( xmlserialize(content  extract( xmlagg(xmlelement(\"e\", t1_roles. role_name ||chr(13)) order by role_name)  , '//text()' ) )  , ','  ) names, " + 
446                                        "        rtrim( xmlserialize(content  extract( xmlagg(xmlelement(\"e\", t1_roles. role_code ||chr(13)) order by  role_name)  , '//text()' ) )  , ','  ) codes, " + 
447                                        "          sysdate " + 
448                                        "from( " + 
449                                        "select  ROL.FULL_ role_name , ROL.SIGN_OBJECT role_code " + 
450                                        "                                        from  AC_IS_BSS_T sys,  " + 
451                                        "                                                 AC_ROLES_BSS_T rol,  " + 
452                                        "                                                 AC_USERS_LINK_KNL_T url,  " + 
453                                        "                                                 AC_USERS_KNL_T AU,  " + 
454                                        "                                                 AC_SUBSYSTEM_CERT_BSS_T subsys,  " + 
455                                        "                                                 LINK_GROUP_USERS_ROLES_KNL_T lugr,  " + 
456                                        "                                                 LINK_GROUP_USERS_USERS_KNL_T lugu  " + 
457                                        "                                        where (SYS.SIGN_OBJECT= :codeSys or  SUBSYS.SUBSYSTEM_CODE= :codeSys )  " + 
458                                        "                                              and (ROL.ID_SRV = URL.UP_ROLES or ROL.ID_SRV = LUGR.UP_ROLES )  " + 
459                                        "                                              and LUGU.UP_GROUP_USERS= LUGR.UP_GROUP_USERS(+)  " + 
460                                        "                                              and LUGU.UP_USERS(+)  = AU.ID_SRV  " + 
461                                        "                                              and URL.UP_USERS(+)  = AU.ID_SRV  " + 
462                                        "                                              and ROL.UP_IS=sys.ID_SRV  " + 
463                                        "                                              and AU. ID_SRV = :idUserSubject " + 
464                                        "                                              and  SUBSYS.UP_IS(+) =SYS.ID_SRV  " + 
465                                        "                                        group by  ROL.FULL_, ROL.SIGN_OBJECT  " + 
466                                        "                                        ) t1_roles) t1")
467                                        .setParameter("idRec", new Long(idRec))
468                                        .setParameter("idUserSubject", idUserSubject)
469                                        .setParameter("codeSys", codeSys)
470                                        .executeUpdate();
471                        
472                        for (AuditFunction func : userFunctions) {
473                                idFunc = actcl.get(func.getCodeFunction());
474                                LOGGER.debug("audit:05:func.getIdFunction():"
475                                                + func.getCodeFunction());
476                                LOGGER.debug("audit:06:idFunc:" + idFunc);
477
478                                if (func.getDetailsFunction() != null
479                                                && func.getDetailsFunction().length() > 500) {
480                                        throw new GeneralFailure(
481                                                        "Field length 'DetailsFunction' is more than 500 characters! ");
482                                }
483
484                                if (idFunc != null && func.getDateFunction() != null) {
485
486                                        // насчёт func.getDateFunction подумать:
487                                        // выдавать исключение или просто эту запись не вставлять
488
489                                        /*
490                                         * if(func.getDateFunction()==null){ throw new
491                                         * GeneralFailure("Отсутствует код роли!"); }
492                                         */
493
494                                        em.createNativeQuery(
495                                                        "insert into ACTIONS_LOG_KNL_T(ID_SRV,  UP_ACTIONS, UP_USERS, DATE_ACTION, DETAILS_ACTION, created, UP_LOG_ROLES) "
496                                                                        + "values(ACTIONS_LOG_KNL_SEQ.nextval, ?, ?, ?, ?, sysdate, ?) ")
497                                                        .setParameter(1, idFunc)
498                                                        .setParameter(2, idUserSubject)
499                                                        .setParameter(3, func.getDateFunction(),
500                                                                        TemporalType.TIMESTAMP)
501                                                        .setParameter(4, func.getDetailsFunction())
502                                                        .setParameter(5, new Long(idRec))
503                                                        .executeUpdate();
504                                }
505                        }
506
507                        sys_audit(6L, "idIS:" + idIS + "; size of funct_list: "
508                                        + userFunctions.size(), "true", IPAddress, idUserAuth,
509                                        codeSys);
510
511                } catch (Exception e) {
512                        sys_audit(6L, "idIS:" + idIS + "; size of funct_list: "
513                                        + userFunctions.size(), "error", IPAddress, idUserAuth,
514                                        codeSys);
515
516                        LOGGER.error("audit:Error:", e);
517                        throw new GeneralFailure(e.getMessage());
518                }
519        }

612         * аудит вызовов сервисов (кроме аутентификации систем и logout)
613         */
614        private void sys_audit(Long idServ, String inp_param, String result,
615                        String ip_adr, Long idUser, String codeSys) {
616                LOGGER.debug("sys_audit");
617
618                Long idSys = null;
619                Long idGrSys = null;
620                try {
621
622                        if (codeSys != null) {
623
624                                // ищем код в системах и подсистемах
625                                if (codeSys.startsWith(CUDConstants.armPrefix)
626                                                || codeSys.startsWith(CUDConstants.subArmPrefix)) {
627
628                                        idSys = get_id_is(codeSys);
629
630                                } else if (codeSys.startsWith(CUDConstants.groupArmPrefix)) {
631                                        // группы систем
632
633                                        idGrSys = ((java.math.BigDecimal) em
634                                                        .createNativeQuery(
635                                                                        "select T1.ID_SRV "
636                                                                                        + "from GROUP_SYSTEMS_KNL_T t1 "
637                                                                                        + "where T1.GROUP_CODE=? ")
638                                                        .setParameter(1, codeSys).getSingleResult())
639                                                        .longValue();
640
641                                }
642                        }
643
644                        if (idUser != null && !idUser.equals(-1L)) {
645                                em.createNativeQuery(
646                                                "insert into SERVICES_LOG_KNL_T( "
647                                                                + "ID_SRV,  UP_SERVICES, DATE_ACTION, CREATED, "
648                                                                + "input_param, result_value, ip_address, UP_USERS, "
649                                                                + "UP_SYS, UP_GR_SYS ) "
650                                                                + "values(SERVICES_LOG_KNL_SEQ.nextval , ?, sysdate, sysdate, "
651                                                                + "?, ?, ?, ?, ?, ? ) ")
652                                                .setParameter(1, idServ).setParameter(2, inp_param)
653                                                .setParameter(3, result).setParameter(4, ip_adr)
654                                                .setParameter(5, idUser)
655                                                .setParameter(6, idSys != null ? idSys : "")
656                                                .setParameter(7, idGrSys != null ? idGrSys : "")
657                                                .executeUpdate();
658                        } else {
659                                em.createNativeQuery(
660                                                "insert into SERVICES_LOG_KNL_T( "
661                                                                + "ID_SRV,  UP_SERVICES, DATE_ACTION, CREATED, "
662                                                                + "input_param, result_value, ip_address, "
663                                                                + "UP_SYS, UP_GR_SYS ) "
664                                                                + "values(SERVICES_LOG_KNL_SEQ.nextval , ?, sysdate, sysdate, "
665                                                                + "?, ?, ?, ?, ? ) ").setParameter(1, idServ)
666                                                .setParameter(2, inp_param).setParameter(3, result)
667                                                .setParameter(4, ip_adr)
668                                                .setParameter(5, idSys != null ? idSys : "")
669                                                .setParameter(6, idGrSys != null ? idGrSys : "")
670                                                .executeUpdate();
671                        }
672                } catch (Exception e) {
673                        LOGGER.error("sys_audit:Error:", e);
674                }
675
676        }

681        public void sys_audit_public(Long idServ, String inp_param, String result,

684                // используется при аутентификации системы и logout

687                try {
688                        // при аудите logout

690

692

705
706                } catch (Exception e) {
707                        LOGGER.error("sys_audit_public:Error:", e);

749        /**
750         * проверка сертификата на наличие в реестре отозванных сертификатов
751         */
752        private synchronized boolean isRevoked(String curr_crl, String hex) {
753
754                boolean result = false;
755
756                CertificateFactory cf = null;
757                InputStream is = null;
758                Collection<? extends X509CRL> x509Crls = null;
759                File crlFile = null;
760
761                try {
762
763                        crlFile = new File(curr_crl);
764
765                        LOGGER.debug("isRevoked:01");
766
767                        cf = CertificateFactory.getInstance("X.509");
768
769                        x509Crls = (Collection<? extends X509CRL>) cf
770                                        .generateCRLs(is = new FileInputStream(crlFile));
771
772                        LOGGER.debug("isRevoked:02");
773
774                        for (X509CRL crl : x509Crls) {
775
776                        
777                                X509CRLEntry xce = crl.getRevokedCertificate(hex_to_dec(hex));
778
779                                if (xce != null) {
780                                        return true;
781                                }
782                                LOGGER.debug("isRevoked:revoked:" + (xce == null));
783                        
784                        }
785
786                        LOGGER.debug("isRevoked:03");
787
788                } catch (Exception e) {
789                        LOGGER.error("isRevoked:error:", e);
790                } finally {
791                        try {
792                                if (is != null) {
793                                        is.close();
794                                }
795                        } catch (Exception e) {
796                                LOGGER.error("isRevoked:finally:is:error:", e);
797                        }
798                }
799
800                return result;
801        }

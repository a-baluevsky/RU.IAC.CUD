012        public static String createLine (List<String> uids){
013                String result = null;
014                
015                if (uids != null && !uids.isEmpty()) {
016
017                        for (String uidValue : uids) {
018                                if (result == null) {
019                                        result = "'" + uidValue + "'";
020                                } else {
021                                        result = result + ", '" + uidValue + "'";
022                                }
023                        }
024                }
025                
026                return result;
027        }

080        /**
081         * <p>
082         * Send the request to the IDP. Subclasses should override this method to
083         * implement how requests must be sent to the IDP.
084         * </p>
085         * 
086         * @param destination
087         *            idp url
088         * @param samlDocument
089         *            request or response document
090         * @param relayState
091         * @param response
092         * @param willSendRequest
093         *            are we sending Request or Response to IDP
094         * @param destinationQueryStringWithSignature
095         *            used only with Redirect binding and with signature enabled.
096         * @throws ProcessingException
097         * @throws ConfigurationException
098         * @throws IOException
099         */
100        protected void sendRequestToIDP(String destination, Document samlDocument,
101                        String relayState, Response response, boolean willSendRequest,
102                        String destinationQueryStringWithSignature)
103                        throws ProcessingException, ConfigurationException, IOException {
104                if (isHttpPostBinding()) {
105                        sendHttpPostBindingRequest(destination, samlDocument, relayState,
106                                        response, willSendRequest);
107                } else {
108                        sendHttpRedirectRequest(destination, samlDocument, relayState,
109                                        response, willSendRequest,
110                                        destinationQueryStringWithSignature);
111                }
112        }

114        /**
115         * <p>
116         * Sends a HTTP Redirect request to the IDP.
117         * </p>
118         * 
119         * @param destination
120         * @param relayState
121         * @param response
122         * @param willSendRequest
123         * @param destinationQueryStringWithSignature
124         * @throws IOException
125         * @throws UnsupportedEncodingException
126         * @throws ConfigurationException
127         * @throws ProcessingException
128         */
129        protected void sendHttpRedirectRequest(String destination,
130                        Document samlDocument, String relayState, Response response,
131                        boolean willSendRequest, String destinationQueryStringWithSignature)
132                        throws IOException, ProcessingException, ConfigurationException {
133                String destinationQueryString = null;
134
135                // We already have queryString with signature from
136                // SAML2SignatureGenerationHandler
137                if (destinationQueryStringWithSignature != null) {
138                        destinationQueryString = destinationQueryStringWithSignature;
139                } else {
140                        String samlMessage = DocumentUtil.getDocumentAsString(samlDocument);
141                        String base64Request = RedirectBindingUtil
142                                        .deflateBase64URLEncode(samlMessage.getBytes("UTF-8"));
143                        destinationQueryString = RedirectBindingUtil
144                                        .getDestinationQueryString(base64Request, relayState,
145                                                        willSendRequest);
146                }
147
148                RedirectBindingUtilDestHolder holder = new RedirectBindingUtilDestHolder();
149
150                holder.setDestination(destination).setDestinationQueryString(
151                                destinationQueryString);
152
153                CUDHTTPRedirectUtil.sendRedirectForRequestor(
154                                RedirectBindingUtil.getDestinationURL(holder), response);
155        }

185        /**
186         * <p>
187         * Initialize the KeyProvider configurations. This configurations are to be
188         * used during signing and validation of SAML assertions.
189         * </p>
190         * 
191         * @param context
192         * @throws LifecycleException
193         */
194        protected void initKeyProvider(Context context) throws LifecycleException {
195                if (!doSupportSignature()) {
196                        return;
197                }
198
199                KeyProviderType keyProvider = this.spConfiguration.getKeyProvider();
200
201                if (keyProvider == null && doSupportSignature()) {
202                        throw new LifecycleException(ErrorCodes.NULL_VALUE
203                                        + "KeyProvider is null for context=" + context.getName());
204                }
205
206                try {
207                        String keyManagerClassName = keyProvider.getClassName();
208                        if (keyManagerClassName == null) {
209                                throw new RuntimeException(ErrorCodes.NULL_VALUE
210                                                + "KeyManager class name");
211                        }
212
213                        Class<?> clazz = SecurityActions.loadClass(getClass(),
214                                        keyManagerClassName);
215
216                        if (clazz == null) {
217                                throw new ClassNotFoundException(ErrorCodes.CLASS_NOT_LOADED
218                                                + keyManagerClassName);
219                        }
220                        this.keyManager = (TrustKeyManager) clazz.newInstance();
221
222                        List<AuthPropertyType> authProperties = CoreConfigUtil
223                                        .getKeyProviderProperties(keyProvider);
224
225                        keyManager.setAuthProperties(authProperties);
226                        keyManager.setValidatingAlias(keyProvider.getValidatingAlias());
227
228                        String identityURL = this.spConfiguration.getIdentityURL();
229
230                        // Special case when you need X509Data in SignedInfo
231                        if (authProperties != null) {
232                                for (AuthPropertyType authPropertyType : authProperties) {
233                                        String key = authPropertyType.getKey();
234                                        if (GeneralConstants.X509CERTIFICATE.equals(key)) {
235                                                // we need X509Certificate in SignedInfo. The value is
236                                                // the alias name
237                                                keyManager.addAdditionalOption(
238                                                                GeneralConstants.X509CERTIFICATE,
239                                                                authPropertyType.getValue());
240                                                break;
241                                        }
242                                }
243                        }
244                        keyManager.addAdditionalOption(
245                                        ServiceProviderBaseProcessor.IDP_KEY,
246                                        new URL(identityURL).getHost());
247                } catch (Exception e) {
248                        logger.trustKeyManagerCreationError(e);
249                        throw new LifecycleException(e.getLocalizedMessage());
250                }
251
252                logger.trace("Key Provider=" + keyProvider.getClassName());
253        }

255        /**
256         * Authenticate the request
257         * 
258         * @param request
259         * @param response
260         * @param config
261         * @return
262         * @throws IOException
263         * @throws {@link RuntimeException} when the response is not of type
264         *         catalina response object
265         */
266        public boolean authenticate(Request request, HttpServletResponse response,
267                        LoginConfig config) throws IOException {
268                if (response instanceof Response) {
269                        Response catalinaResponse = (Response) response;
270                        return authenticate(request, catalinaResponse, config);
271                }
272                throw logger.samlSPResponseNotCatalinaResponseError(response);
273        }

275        /*
276         * (non-Javadoc)
277         * 
278         * @see
279         * org.apache.catalina.authenticator.FormAuthenticator#authenticate(org.
280         * apache.catalina.connector.Request,
281         * org.apache.catalina.connector.Response,
282         * org.apache.catalina.deploy.LoginConfig)
283         */
284        // @Override
285        public boolean authenticate(Request request, Response response,
286                        LoginConfig loginConfig) throws IOException {
287                try {
288                        // needs to be done first, *before* accessing any parameters.
289                        // super.authenticate(..) gets called to late
290                        String characterEncoding = getCharacterEncoding();
291                        if (characterEncoding != null) {
292                                request.setCharacterEncoding(characterEncoding);
293                        }
294
295                        Session session = request.getSessionInternal(true);
296
297                        // check if this call is resulting from the redirect after
298                        // successful authentication.
299                        // if so, make the authentication successful and continue the
300                        // original request
301                        if (saveRestoreRequest && matchRequest(request)) {
302                                logger.trace("Restoring request from session '"
303                                                + session.getIdInternal() + "'");
304                                Principal savedPrincipal = (Principal) session
305                                                .getNote(Constants.FORM_PRINCIPAL_NOTE);
306                                register(request, response, savedPrincipal,
307                                                Constants.LOGIN_METHOD /* .FORM_METHOD */,
308                                                (String) session.getNote(Constants.SESS_USERNAME_NOTE),
309                                                (String) session.getNote(Constants.SESS_PASSWORD_NOTE));
310
311                                // try to restore the original request (including post data,
312                                // etc...)
313                                if (restoreRequest(request, session)) {
314                                        // success! user is authenticated; continue processing
315                                        // original request
316                                        logger.trace("Continuing with restored request.");
317                                        return true;
318                                } else {
319                                        // no saved request found...
320                                        logger.trace("Restore of original request failed!");
321                                        response.sendError(HttpServletResponse.SC_BAD_REQUEST);
322                                        return false;
323                                }
324                        }
325
326                        // Eagerly look for Local LogOut
327                        boolean localLogout = isLocalLogout(request);
328
329                        if (localLogout) {
330                                try {
331                                        sendToLogoutPage(request, response, session);
332                                } catch (ServletException e) {
333                                        logger.samlLogoutError(e);
334                                        throw new IOException(e);
335                                }
336                                return false;
337                        }
338
339                        String samlRequest = request
340                                        .getParameter(GeneralConstants.SAML_REQUEST_KEY);
341                        String samlResponse = request
342                                        .getParameter(GeneralConstants.SAML_RESPONSE_KEY);
343
344                        Principal principal = request.getUserPrincipal();
345
346                        // If we have already authenticated the user and there is no request
347                        // from IDP or logout from user
348                        if (principal != null
349                                        && !(isGlobalLogout(request) || isNotNull(samlRequest) || isNotNull(samlResponse))) {
350                                return true;
351                        }
352
353                        // General User Request
354                        if (!isNotNull(samlRequest) && !isNotNull(samlResponse)) {
355                                return generalUserRequest(request, response, loginConfig);
356                        }
357
358                        // Handle a SAML Response from IDP
359                        if (isNotNull(samlResponse)) {
360                                return handleSAMLResponse(request, response, loginConfig);
361                        }
362
363                        // Handle SAML Requests from IDP
364                        if (isNotNull(samlRequest)) {
365                                return handleSAMLRequest(request, response, loginConfig);
366                        }// end if
367
368                        return localAuthentication(request, response, loginConfig);
369                } catch (IOException e) {
370                        if (StringUtil.isNotNull(spConfiguration.getErrorPage())) {
371                                try {
372                                        request.getRequestDispatcher(spConfiguration.getErrorPage())
373                                                        .forward(request.getRequest(), response);
374                                } catch (ServletException e1) {
375                                        logger.samlErrorPageForwardError(
376                                                        spConfiguration.getErrorPage(), e1);
377                                }
378                                return false;
379                        } else {
380                                throw e;
381                        }
382                }
383        }

411        /**
412         * Handle the IDP Request
413         * 
414         * @param request
415         * @param response
416         * @param loginConfig
417         * @return
418         * @throws IOException
419         */
420        private boolean handleSAMLRequest(Request request, Response response,
421                        LoginConfig loginConfig) throws IOException {
422                String samlRequest = request
423                                .getParameter(GeneralConstants.SAML_REQUEST_KEY);
424                HTTPContext httpContext = new HTTPContext(request, response,
425                                context.getServletContext());
426                Set<SAML2Handler> handlers = chain.handlers();
427
428                try {
429                        ServiceProviderSAMLRequestProcessor requestProcessor = new ServiceProviderSAMLRequestProcessor(
430                                        request.getMethod().equals("POST"), this.serviceURL,
431                                        this.picketLinkConfiguration);
432                        requestProcessor.setTrustKeyManager(keyManager);
433                        boolean result = requestProcessor.process(samlRequest, httpContext,
434                                        handlers, chainLock);
435
436                        if (enableAudit) {
437                                PicketLinkAuditEvent auditEvent = new PicketLinkAuditEvent(
438                                                AuditLevel.INFO);
439                                auditEvent.setType(PicketLinkAuditEventType.REQUEST_FROM_IDP);
440                                auditEvent.setWhoIsAuditing(getContextPath());
441                                auditHelper.audit(auditEvent);
442                        }
443
444                        // If response is already commited, we need to stop with processing
445                        // of HTTP request
446                        if (response.isCommitted() || response.isAppCommitted()) {
447                                return false;
448                        }
449
450                        if (result) {
451                                return result;
452                        }
453                } catch (Exception e) {
454                        logger.samlSPHandleRequestError(e);
455                        throw logger.samlSPProcessingExceptionError(e);
456                }
457
458                return localAuthentication(request, response, loginConfig);
459        }

461        /**
462         * Handle IDP Response
463         * 
464         * @param request
465         * @param response
466         * @param loginConfig
467         * @return
468         * @throws IOException
469         */
470        private boolean handleSAMLResponse(Request request, Response response,
471                        LoginConfig loginConfig) throws IOException {
472                Session session = request.getSessionInternal(true);
473                String samlResponse = request
474                                .getParameter(GeneralConstants.SAML_RESPONSE_KEY);
475
476                boolean willSendRequest = false;
477                HTTPContext httpContext = new HTTPContext(request, response,
478                                context.getServletContext());
479                Set<SAML2Handler> handlers = chain.handlers();
480
481                Principal principal = request.getUserPrincipal();
482
483                if (!super.validate(request)) {
484                        throw new IOException(ErrorCodes.VALIDATION_CHECK_FAILED);
485                }
486
487                // deal with SAML response from IDP
488                try {
489                        ServiceProviderSAMLResponseProcessor responseProcessor = new ServiceProviderSAMLResponseProcessor(
490                                        request.getMethod().equals("POST"), serviceURL,
491                                        this.picketLinkConfiguration);
492                        if (auditHelper != null) {
493                                responseProcessor.setAuditHelper(auditHelper);
494                        }
495
496                        responseProcessor.setTrustKeyManager(keyManager);
497
498                        SAML2HandlerResponse saml2HandlerResponse = responseProcessor
499                                        .process(samlResponse, httpContext, handlers, chainLock);
500
501                        Document samlResponseDocument = saml2HandlerResponse
502                                        .getResultingDocument();
503                        String relayState = saml2HandlerResponse.getRelayState();
504
505                        String destination = saml2HandlerResponse.getDestination();
506
507                        willSendRequest = saml2HandlerResponse.getSendRequest();
508
509                        String destinationQueryStringWithSignature = saml2HandlerResponse
510                                        .getDestinationQueryStringWithSignature();
511
512                        if (destination != null && samlResponseDocument != null) {
513                                sendRequestToIDP(destination, samlResponseDocument, relayState,
514                                                response, willSendRequest,
515                                                destinationQueryStringWithSignature);
516                        } else {
517                                // See if the session has been invalidated
518
519                                boolean sessionValidity = session.isValid();
520
521                                if (!sessionValidity) {
522                                        sendToLogoutPage(request, response, session);
523                                        return false;
524                                }
525
526                                // We got a response with the principal
527                                List<String> roles = saml2HandlerResponse.getRoles();
528                                if (principal == null) {
529                                        principal = (Principal) session.getSession().getAttribute(
530                                                        GeneralConstants.PRINCIPAL_ID);
531                                }
532
533                                String username = principal.getName();
534                                String password = ServiceProviderSAMLContext.EMPTY_PASSWORD;
535
536                                if (logger.isTraceEnabled()) {
537                                        logger.trace("Roles determined for username=" + username
538                                                        + "=" + Arrays.toString(roles.toArray()));
539                                }
540
541                                // Map to JBoss specific principal
542                                if ((new ServerDetector()).isJboss() || jbossEnv) {
543                                        // Push a context
544                                        ServiceProviderSAMLContext.push(username, roles);
545                                        principal = context.getRealm().authenticate(username,
546                                                        password);
547                                        ServiceProviderSAMLContext.clear();
548                                } else {
549                                        // tomcat env
550                                        principal = getGenericPrincipal(request, username, roles);
551                                }
552
553                                session.setNote(Constants.SESS_USERNAME_NOTE, username);
554                                session.setNote(Constants.SESS_PASSWORD_NOTE, password);
555                                request.setUserPrincipal(principal);
556
557                                if (enableAudit) {
558                                        PicketLinkAuditEvent auditEvent = new PicketLinkAuditEvent(
559                                                        AuditLevel.INFO);
560                                        auditEvent
561                                                        .setType(PicketLinkAuditEventType.RESPONSE_FROM_IDP);
562                                        auditEvent.setSubjectName(username);
563                                        auditEvent.setWhoIsAuditing(getContextPath());
564                                        auditHelper.audit(auditEvent);
565                                }
566
567                                // Redirect the user to the originally requested URL
568                                if (saveRestoreRequest) {
569                                        // Store the authenticated principal in the session.
570                                        session.setNote(Constants.FORM_PRINCIPAL_NOTE, principal);
571
572                                        // Redirect to the original URL. Note that this will trigger
573                                        // the
574                                        // authenticator again, but on resubmission we will look in
575                                        // the
576                                        // session notes to retrieve the authenticated principal and
577                                        // prevent reauthentication
578                                        String requestURI = savedRequestURL(session);
579
580                                        if (requestURI == null) {
581                                                requestURI = getConfiguration().getServiceURL();
582                                        }
583
584                                        logger.trace("Redirecting back to original Request URI: "
585                                                        + requestURI);
586                                        response.sendRedirect(response
587                                                        .encodeRedirectURL(requestURI));
588                                        return false;
589                                }
590
591                                register(request, response, principal,
592                                                Constants.LOGIN_METHOD /* Constants.FORM_METHOD */,
593                                                username, password);
594                                return true;
595                        }
596                } catch (ProcessingException pe) {
597                        Throwable t = pe.getCause();
598                        if (t != null && t instanceof AssertionExpiredException) {
599                                logger.error("Assertion has expired. Asking IDP for reissue");
600                                if (enableAudit) {
601                                        PicketLinkAuditEvent auditEvent = new PicketLinkAuditEvent(
602                                                        AuditLevel.INFO);
603                                        auditEvent
604                                                        .setType(PicketLinkAuditEventType.EXPIRED_ASSERTION);
605                                        auditEvent.setAssertionID(((AssertionExpiredException) t)
606                                                        .getId());
607                                        auditHelper.audit(auditEvent);
608                                }
609                                // Just issue a fresh request back to IDP
610                                return generalUserRequest(request, response, loginConfig);
611                        }
612                        logger.samlSPHandleRequestError(pe);
613                        throw logger.samlSPProcessingExceptionError(pe);
614                } catch (Exception e) {
615                        logger.samlSPHandleRequestError(e);
616                        throw logger.samlSPProcessingExceptionError(e);
617                }
618
619                return localAuthentication(request, response, loginConfig);
620        }

638        /**
639         * Handle the user invocation for the first time
640         * 
641         * @param request
642         * @param response
643         * @param loginConfig
644         * @return
645         * @throws IOException
646         */
647        private boolean generalUserRequest(Request request, Response response,
648                        LoginConfig loginConfig) throws IOException {
649                Session session = request.getSessionInternal(true);
650                boolean willSendRequest = false;
651                HTTPContext httpContext = new HTTPContext(request, response,
652                                context.getServletContext());
653                Set<SAML2Handler> handlers = chain.handlers();
654
655                boolean postBinding = spConfiguration.getBindingType().equals("POST");
656
657                // Neither saml request nor response from IDP
658                // So this is a user request
659                SAML2HandlerResponse saml2HandlerResponse = null;
660                try {
661                        ServiceProviderBaseProcessor baseProcessor = new ServiceProviderBaseProcessor(
662                                        postBinding, serviceURL, this.picketLinkConfiguration);
663                        if (issuerID != null) {
664                                baseProcessor.setIssuer(issuerID);
665                        }
666
667                        baseProcessor.setIdentityURL(identityURL);
668                        baseProcessor.setAuditHelper(auditHelper);
669
670                        saml2HandlerResponse = baseProcessor.process(httpContext, handlers,
671                                        chainLock);
672                } catch (ProcessingException pe) {
673                        logger.samlSPHandleRequestError(pe);
674                        throw new RuntimeException(pe);
675                } catch (ParsingException pe) {
676                        logger.samlSPHandleRequestError(pe);
677                        throw new RuntimeException(pe);
678                } catch (ConfigurationException pe) {
679                        logger.samlSPHandleRequestError(pe);
680                        throw new RuntimeException(pe);
681                }
682
683                willSendRequest = saml2HandlerResponse.getSendRequest();
684
685                Document samlResponseDocument = saml2HandlerResponse
686                                .getResultingDocument();
687                String relayState = saml2HandlerResponse.getRelayState();
688
689                String destination = saml2HandlerResponse.getDestination();
690                String destinationQueryStringWithSignature = saml2HandlerResponse
691                                .getDestinationQueryStringWithSignature();
692
693                if (destination != null && samlResponseDocument != null) {
694                        try {
695                                if (saveRestoreRequest) {
696                                        this.saveRequest(request, session);
697                                }
698                                if (enableAudit) {
699                                        PicketLinkAuditEvent auditEvent = new PicketLinkAuditEvent(
700                                                        AuditLevel.INFO);
701                                        auditEvent.setType(PicketLinkAuditEventType.REQUEST_TO_IDP);
702                                        auditEvent.setWhoIsAuditing(getContextPath());
703                                        auditHelper.audit(auditEvent);
704                                }
705                                sendRequestToIDP(destination, samlResponseDocument, relayState,
706                                                response, willSendRequest,
707                                                destinationQueryStringWithSignature);
708                                return false;
709                        } catch (Exception e) {
710                                logger.samlSPHandleRequestError(e);
711                                throw logger.samlSPProcessingExceptionError(e);
712                        }
713                }
714
715                return localAuthentication(request, response, loginConfig);
716        }

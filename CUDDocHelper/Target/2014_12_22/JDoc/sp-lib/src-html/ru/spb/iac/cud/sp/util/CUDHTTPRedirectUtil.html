001package ru.spb.iac.cud.sp.util;
002
003import javax.servlet.http.HttpServletResponse;
004import java.io.IOException;
005
006
007 public class CUDHTTPRedirectUtil {
008        /**
009         * Send the response to the redirected destination while adding the
010         * character encoding of "UTF-8" as well as adding headers for cache-control
011         * and Pragma
012         * 
013         * @param destination
014         *            Destination URI where the response needs to redirect
015         * @param response
016         *            HttpServletResponse
017         * @throws IOException
018         */
019        public static void sendRedirectForRequestor(String destination,
020                        HttpServletResponse response) throws IOException {
021                common(destination, response);
022                response.setHeader("Cache-Control", "no-cache, no-store");
023                sendRedirect(response, destination);
024        }
025
026        /**
027         * @see #sendRedirectForRequestor(String, HttpServletResponse)
028         */
029        public static void sendRedirectForResponder(String destination,
030                        HttpServletResponse response) throws IOException {
031                common(destination, response);
032                response.setHeader("Cache-Control",
033                                "no-cache, no-store, must-revalidate,private");
034                sendRedirect(response, destination);
035        }
036
037        private static void common(String destination, HttpServletResponse response) {
038                response.setCharacterEncoding("UTF-8");
039                response.setHeader("Location", destination);
040                response.setHeader("Pragma", "no-cache");
041        }
042
043        private static void sendRedirect(HttpServletResponse response,
044                        String destination) throws IOException {
045
046                // !!!
047                // response.reset();
048                response.setStatus(HttpServletResponse.SC_MOVED_TEMPORARILY);
049                response.sendRedirect(destination);
050        }
051
052}

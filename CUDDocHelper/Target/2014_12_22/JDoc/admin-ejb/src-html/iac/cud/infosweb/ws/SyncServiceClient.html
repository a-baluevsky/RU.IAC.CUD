053                        role = new Role();

055                        role.setName("Роль менеджера");
056                        role.setDescription("Управление системой");
057                        roles.add(role);
058
059                        role = new Role();
060                        role.setCode("urn:role:user");
061                        role.setName("Роль пользователя");
062                        role.setDescription("Просмотр списков");
063                        roles.add(role);
064
065                        getPort(endpointURI).sync_roles(idIS, roles);

067                } catch (GeneralFailure e1) {
068                        log.error("sync_roles:error1:" + e1);
069                        throw e1;
070                }

450        private SyncServices getPort(String endpointURI)
451                        throws MalformedURLException {
452
453                if (this.syncServices == null) {
454                        QName serviceName = new QName("http://sync.services.cud.iac/",
455                                        "SyncServicesImplService");
456                        URL wsdlURL = new URL(endpointURI);
457
458                        Service service = Service.create(wsdlURL, serviceName);
459
460                        QName portName = new QName("http://sync.services.cud.iac/",
461                                        "SyncServicesImplPort");
462
463                        this.syncServices = service.getPort(portName, SyncServices.class);
464                }
465                return this.syncServices;
466        }

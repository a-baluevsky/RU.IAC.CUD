001package iac.cud.infosweb.local.service;
002
003import java.util.EnumSet;
004import java.util.HashMap; import java.util.Map;
005import org.slf4j.Logger;
006import org.slf4j.LoggerFactory;
007
008 public class ReestrItem {
009
010        private static final Map<String, String> lookup = new HashMap<String,String>();
011
012        private static final Logger LOGGER = LoggerFactory.getLogger(ReestrItem.class);
013        
014    static {
015      for(ServiceReestrPro s : EnumSet.allOf(ServiceReestrPro.class)){
016          try{
017           lookup.put(s.name(), (String)ServiceReestr.class.getDeclaredField(s.name()).get(null));
018          }catch(Exception e){
019                  LOGGER.error("ReestrItem:Error:"+e);
020          }
021         }
022    }
023    
024    public static String getUrl(String s){
025        return lookup.get(s);
026    }
027}

331    public List<IspBssT> autocomplete(Object suggest) throws Exception{
332        String pref = (String)suggest;
333        
334            log.info("autocomplete:01:pref:"+pref);
335            try {
336                
337                if(listOrg==null){
338                        
339                        log.info("autocomplete:02");
340                        
341                        AcUser  cau = (AcUser) Component.getInstance("currentUser",ScopeType.SESSION);
342                        
343                        listOrg=entityManager.createQuery(
344                                        "select o from IspBssT o where o.status='A' " +
345                                        "and o.signObject like '%00000' " +
346                                        "and ( 1 = :orgAccFlag  or o.signObject = :orgCode) " +
347                                //      "and o.full like '"+pref+"%' " +
348                                        "and upper(o.full) like upper(:pref) " +
349                                        "order by o.full ")
350                                        .setParameter("pref", pref+"%")
351                                        .setParameter("orgAccFlag", cau.getIsAccOrgManagerValue() ? -1 : 1)
352                                        .setParameter("orgCode", cau.getUpSign()!=null? cau.getUpSign():"")
353                                        .getResultList();
354                        
355                        log.info("autocomplete:03:size:"+listOrg.size());
356                }
357             } catch (Exception e) {
358                 log.error("autocomplete:ERROR:"+e);
359                 throw e;
360             }
361            return listOrg;

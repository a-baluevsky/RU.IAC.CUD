052        public List<BaseItem> getAuditList(int firstRow, int numberOfRows){
053                  
054                  String remoteAuditUgroupUsr = FacesContext.getCurrentInstance().getExternalContext()
055                             .getRequestParameterMap()
056                             .get("remoteAudit");
057                   String sessionId = FacesContext.getCurrentInstance().getExternalContext()
058                                .getRequestParameterMap()
059                                .get("sessionId");
060                   
061                   log.info("ugroupUsrManager:getAuditList:remoteAudit:"+remoteAuditUgroupUsr);
062                   log.info("ugroupUsrManager:getAuditList:sessionId:"+sessionId);
063                   
064                  
065                  log.info("ugroupUsrManager:getAuditList:firstRow:"+firstRow);
066                  log.info("ugroupUsrManager:getAuditList:numberOfRows:"+numberOfRows);
067                  
068                  List<BaseItem> ugroupUsrListCached = (List<BaseItem>)
069                                  Component.getInstance("ugroupUsrListCached",ScopeType.SESSION);
070                  if(auditList==null){
071                          log.info("ugroupUsrManager:getAuditList:01");
072                                if(("rowSelectFact".equals(remoteAuditUgroupUsr)||
073                                    "selRecAllFact".equals(remoteAuditUgroupUsr)||
074                                    "clRecAllFact".equals(remoteAuditUgroupUsr)||
075                                    "clSelOneFact".equals(remoteAuditUgroupUsr)||
076                                    "onSelColSaveFact".equals(remoteAuditUgroupUsr))&&
077                                    ugroupUsrListCached!=null){
078                                        log.info("ugroupUsrManager:getAuditList:02:"+ugroupUsrListCached.size());
079                                        this.auditList=ugroupUsrListCached;
080                                }else{
081                                        log.info("ugroupUsrManager:getAuditList:03");
082                                invokeLocal("list", firstRow, numberOfRows, null);
083                                    Contexts.getSessionContext().set("ugroupUsrListCached", this.auditList);
084                                    log.info("ugroupUsrManager:getAuditList:03:"+this.auditList.size());
085                                }
086                                
087                                
088                                
089                                try{
090                                         List<Long> listUsr=entityManager.createQuery(
091                                                 "select o.idUser from AcUser o,  LinkGroupUsersUsersKnlT o1 " +
092                                                 "where o1.pk.acUser = o.idUser " +
093                                                 "and o1.pk.groupUser = :groupUser ")
094                                                         .setParameter("groupUser", new Long(sessionId))
095                                                 .getResultList();
096                                 
097                           
098                             
099                                     for(BaseItem user :this.auditList){
100                                   if (listUsr.contains(user.getBaseId())){  
101                                          ((UserItem)user).setUsrChecked(true);
102                                           }
103                                 } 
104                                        
105                                }catch(Exception eUgroupUsr){
106                                          log.error("UsrManager:getUsrAlfList:error:"+eUgroupUsr);
107                                }
108                                
109                        }
110                        return this.auditList;
111                }

115                public void invokeLocal(String type, int firstRow, int numberOfRows,
116                           String sessionId) {
117                        try{
118                                 String orderQuery=null;
119                                 log.info("ugroupUsrManager:invokeLocal");
120                                 
121                                 UgroupUsrStateHolder ugroupUsrStateHolder = (UgroupUsrStateHolder)
122                                                  Component.getInstance("ugroupUsrStateHolder",ScopeType.SESSION);
123                                 
124                                 Map<String, String> filterMap = ugroupUsrStateHolder.getColumnFilterValues();
125                                 String st=null;
126                                 
127                                  RolUsrManager rolUsrManager = (RolUsrManager)
128                                  Component.getInstance("rolUsrManager");
129                                  
130                                 if("list".equals(type)){
131                                         log.info("ugroupUsrManager:list:01");
132                                         
133                                         Set<Map.Entry<String, String>> set = ugroupUsrStateHolder.getSortOrders().entrySet();
134                         for (Map.Entry<String, String> me : set) {
135                                
136                               if(orderQuery==null){
137                                 orderQuery="order by "+me.getKey()+" "+me.getValue();
138                               }else{
139                                 orderQuery=orderQuery+", "+me.getKey()+" "+me.getValue();  
140                               }
141                             }
142                         log.info("ugroupUsrManager:invokeLocal:list:orderQuery:"+orderQuery);
143                         
144                         if(filterMap!=null){
145                                 Set<Map.Entry<String, String>> setFilterUgroupUsr = filterMap.entrySet();
146                              for (Map.Entry<String, String> me : setFilterUgroupUsr) {
147                                 
148                                     if("t1_crt_date".equals(me.getKey())){  
149                                   
150                                   //делаем фильтр на начало  
151                                     st=(st!=null?st+" and " :"")+" lower(to_char("+me.getKey()+",'DD.MM.YY HH24:MI:SS')) like lower('"+me.getValue()+"%') ";
152                             }else{
153                                        //делаем фильтр на начало
154                                  st=(st!=null?st+" and " :"")+" lower("+me.getKey()+") like lower('"+me.getValue()+"%') ";
155                                  }
156                              }
157                           }
158                         log.info("ugroupUsrManager:invokeLocal:list:filterQuery:"+st);
159                         
160                        
161                       AcUser cau = (AcUser) Component.getInstance("currentUser",ScopeType.SESSION);
162                                 
163                       if(cau.getIsAccOrgManagerValue()){
164                                   st=(st!=null?st+" and ":" ")+" t1_org_code = '"+cau.getUpSign()+"' ";
165                       }
166                       
167                     
168                       
169                       auditList = rolUsrManager.getSharedUserList( orderQuery, st, firstRow, numberOfRows);
170                       
171                       
172                       
173                     log.info("ugroupUsrManager:invokeLocal:list:02");
174                     
175                                 } else if("count".equals(type)){
176                                         log.info("ugroupUsrManager:count:01");
177                                         
178                         
179                         if(filterMap!=null){
180                                 Set<Map.Entry<String, String>> setFilterUgroupUsr = filterMap.entrySet();
181                              for (Map.Entry<String, String> me : setFilterUgroupUsr) {
182                               
183                                                //делаем фильтр на начало
184                                  st=(st!=null?st+" and " :"")+" lower("+me.getKey()+") like lower('"+me.getValue()+"%') ";
185                                
186                              }
187                           }
188                                         
189                         AcUser cau = (AcUser) Component.getInstance("currentUser",ScopeType.SESSION);
190                                 
191                               if(cau.getIsAccOrgManagerValue()){
192                                           st=(st!=null?st+" and ":" ")+" t1_org_code = '"+cau.getUpSign()+"' ";
193                               }
194                                        
195                               //!!!
196                              auditCount = rolUsrManager.getSharedUserCount (st);
197                                      
198                                      
199                         
200                       log.info("ugroupUsrManager:invokeLocal:count:02:"+auditCount);
201                         }
202                        }catch(Exception eUgroupUsr){
203                                  log.error("invokeLocal:error:"+eUgroupUsr);
204                                  evaluteForList=false;
205                                  FacesMessages.instance().add("Ошибка!");
206                        }
207                }

209                 public void updUgroupUserAlf(){
210                           
211                           log.info("ugroupUsrManager:updUgroupUserAlf:01");
212                           
213                           LinkGroupUsersUsersKnlT lguu=null;
214                           
215                           
216                           
217                           GroupUsersKnlT ugroupBean = (GroupUsersKnlT)
218                                                  Component.getInstance("ugroupBean",ScopeType.CONVERSATION);
219                           
220                                   String  sessionId = FacesContext.getCurrentInstance().getExternalContext()
221                                        .getRequestParameterMap()
222                                        .get("sessionId");
223                           log.info("ugroupUsrManager:updUgroupUserAlf:sessionId:"+sessionId);
224                          
225                           
226                           if(ugroupBean==null || sessionId==null){
227                                   return;
228                           }
229                        
230                                   
231                           try {
232                                   AcUser au = (AcUser) Component.getInstance("currentUser",ScopeType.SESSION);
233                                   
234                                   GroupUsersKnlT aum = entityManager.find(GroupUsersKnlT.class, new Long(sessionId));
235                                   
236                                   List<LinkGroupUsersUsersKnlT> oldLinkList = aum.getLinkGroupUsersUsersKnlTs();
237                                   
238                                   log.info("ugroupUsrManager:updUgroupUserAlf:size1:"+oldLinkList.size());
239                                   log.info("ugroupUsrManager:updUgroupUserAlf:size2:"+(this.auditList!=null?this.auditList.size():"is null"));
240                                   
241                                   for(BaseItem user:this.auditList){
242                                          log.info("ugroupManager:updUgroupUserAlf:Login:"+((UserItem)user).getLogin());
243                                          log.info("ugroupManager:updUgroupUserAlf:UsrChecked:"+((UserItem)user).getUsrChecked());
244                                          
245                                          if(((UserItem)user).getUsrChecked().booleanValue()){ //отмечен
246                                                
247                                                 
248                                                 lguu=new LinkGroupUsersUsersKnlT(user.getBaseId(), new Long(sessionId));
249                                                 if(oldLinkList.contains(lguu)){  
250                                                 
251                                                 }else{//нет в базе
252                                                         lguu.setCreated(new Date()); 
253                                                         lguu.setCreator(au.getIdUser());
254                                                 
255                                                 oldLinkList.add(lguu);
256                                                 
257                                                  }
258                                                  
259                                          }else{
260                                                  //не отмечен
261                                                 //есть в базе
262                                                 lguu=new LinkGroupUsersUsersKnlT(user.getBaseId(), new Long(sessionId));
263                                                 if(oldLinkList.contains(lguu)){ 
264                                                        oldLinkList.remove(lguu);
265                                                        entityManager.createQuery("DELETE FROM LinkGroupUsersUsersKnlT gu " +
266                                                                                          "WHERE gu.pk.groupUser=:groupUser " +
267                                                                                          "and gu.pk.acUser=:acUser ")
268                                                            .setParameter("groupUser", new Long(sessionId))
269                                                            .setParameter("acUser", user.getBaseId())
270                                                            .executeUpdate();
271                                                  }else{
272                                                          //в базе и так нет
273                                                          log.info("ugroupManager:not_in_db");
274                                                  }
275                                          }
276                                  }
277
278                                   
279                                entityManager.flush();
280                                
281                                    entityManager.refresh(aum);
282                                  
283                                Contexts.getEventContext().set("ugroupBean", aum);
284                                 
285                                //аудит!!!
286                                UgroupManager ugroupManager = (UgroupManager)Component.getInstance("ugroupManager", ScopeType.EVENT);
287                                    ugroupManager.audit(ResourcesMap.UGROUP, ActionsMap.UPDATE_USER); 
288                                
289                                
290                                
291                             }catch (Exception e) {
292                       log.error("ugroupUsrManager:updUgroupUserAlf:ERROR:"+e);
293                     }
294                }

331 public Boolean getEvaluteForList() {
332                
333                log.info("ugroupUsrManager:evaluteForList:01");
334                if(evaluteForList==null){
335                        evaluteForList=false;
336                String remoteAuditUgroupUsr = FacesContext.getCurrentInstance().getExternalContext()
337                             .getRequestParameterMap()
338                             .get("remoteAudit");
339                   log.info("ugroupUsrManager:evaluteForList:remoteAudit:"+remoteAuditUgroupUsr);
340                
341                if(remoteAuditUgroupUsr!=null&&
342                 
343                   !"OpenCrtFact".equals(remoteAuditUgroupUsr)&&        
344                   !"OpenUpdFact".equals(remoteAuditUgroupUsr)&&
345                   !"OpenDelFact".equals(remoteAuditUgroupUsr)&&
346                       !"onSelColFact".equals(remoteAuditUgroupUsr)&&
347                       !"refreshPdFact".equals(remoteAuditUgroupUsr)){
348                        log.info("ugroupUsrManager:evaluteForList!!!");
349                            evaluteForList=true;
350                }
351                 }
352               return evaluteForList;
353           }

236

249                                        "         AND  SUB.UP_IS NOT IN (:idsArm) " +

268            }

498    public void forViewCert() {
499           try{
500             String sessionIdArmSub = FacesContext.getCurrentInstance().getExternalContext()
501                                .getRequestParameterMap()
502                                .get("sessionId");
503             log.info("forViewCert:sessionId:"+sessionIdArmSub);
504             
505             DateFormat df = new SimpleDateFormat ("dd.MM.yy");
506            
507             if(sessionIdArmSub!=null){
508                 
509                String certDataX = (String) entityManager.createNativeQuery(
510                                 "select to_char(T1.CERT_DATE) " + 
511                                 "from AC_SUBSYSTEM_CERT_BSS_T t1 " + 
512                                 "where T1.ID_SRV=? ")
513                 .setParameter(1, new Long(sessionIdArmSub))
514                 .getSingleResult();
515                
516                 log.info("forViewCert:cert_data:"+certDataX); 
517                
518             if(certDataX!=null&&!certDataX.trim().equals("")){
519                 byte[] certByteX = Base64.decode(certDataX);
520                 
521                 CertificateFactory userCf = CertificateFactory.getInstance("X.509");
522                   X509Certificate userCertX = (X509Certificate)
523                                   userCf.generateCertificate(new  ByteArrayInputStream(certByteX ));
524                  
525                   
526                   log.info("armSubManager:saveArmSubCertificate:02:"+userCertX);
527                 
528                  SystemCertItem sciArmSub = new SystemCertItem();
529                 
530                  sciArmSub.setName(userCertX.getSubjectDN().getName());
531                  sciArmSub.setIssuer(userCertX.getIssuerDN().getName());
532                  sciArmSub.setSerial(decToHex(userCertX.getSerialNumber()));
533                  sciArmSub.setDate1(df.format(userCertX.getNotBefore()));
534                  sciArmSub.setDate2(df.format(userCertX.getNotAfter()));
535                  
536                 Contexts.getEventContext().set("systemCertBean", sciArmSub);
537                 
538               }
539                 }
540           }catch(Exception e){
541                   log.error("forViewCert:Error:"+e);
542           }

640   private boolean armSubCodeExistCrt(Long idArm, String armSubCode) throws Exception {

642                if(armSubCode!=null){
643                  try{
644                          //в подсистемах
645                          List<Object> loArmSub=entityManager.createNativeQuery(
646                                          "select 1 "+
647                      "from AC_SUBSYSTEM_CERT_BSS_T ass "+
648                      "where ASS.UP_IS = ? "+
649                      "and ASS.SUBSYSTEM_CODE = ? ")
650                                        .setParameter(1, idArm)
651                                        .setParameter(2, armSubCode)
652                                        .getResultList();
653          
654                  if(!loArmSub.isEmpty()){
655                        armSubCodeExist=true;
656                        return this.armSubCodeExist;
657                  }
658                  
659                  //в системах
660                  loArmSub=entityManager.createNativeQuery(
661                                          "select 1 "+
662                      "from AC_IS_BSS_T ass "+
663                      "where ASS.SIGN_OBJECT = ? ")
664                                        .setParameter(1, armSubCode)
665                                        .getResultList();
666          
667                  if(!loArmSub.isEmpty()){
668                        armSubCodeExist=true;
669                  }
670                  
671                          log.info("ArmSubManager:ArmSubCodeExistCrt:addLoginExist!");  
672                          
673                  }catch(Exception eArmSub){
674                   log.error("ArmSubManager:ArmSubCodeExistCrt:Error:"+eArmSub);

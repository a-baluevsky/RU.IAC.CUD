062

064                          Component.getInstance("permListCached",ScopeType.SESSION);
065          if(auditList==null){
066                  log.info("getAuditList:01");

077                            Contexts.getSessionContext().set("permListCached", this.auditList);

088                                   }else{

090                                   }

092                    }
093                }

097        public void invokeLocal(String type, int firstRow, int numberOfRows,
098                   String sessionId) {
099                try{
100                         String orderQuery=null;
101                         log.info("Perm:Manager:invokeLocal");
102                         
103                         if("list".equals(type)){
104                                 log.info("Perm:invokeLocal:list:01");
105                                 
106                                 PermStateHolder permStateHolder = (PermStateHolder)
107                                                  Component.getInstance("permStateHolder",ScopeType.SESSION);
108                                 Set<Map.Entry<String, String>> set = permStateHolder.getSortOrders().entrySet();
109                 for (Map.Entry<String, String> me : set) {
110                       log.info("me.getKey+:"+me.getKey());
111                       log.info("me.getValue:"+me.getValue());
112                       
113                       if(orderQuery==null){
114                         orderQuery="order by "+me.getKey()+" "+me.getValue();
115                       }else{
116                         orderQuery=orderQuery+", "+me.getKey()+" "+me.getValue();  
117                       }
118                     }
119                 log.info("Perm:invokeLocal:list:orderQuery:"+orderQuery);
120                 
121                                         auditList = entityManager.createQuery("select o from AcPermissionsList o "+(orderQuery!=null ? orderQuery+", orderNum " : " order by orderNum "))
122                       .setFirstResult(firstRow)
123                       .setMaxResults(numberOfRows)
124                       .getResultList();
125             log.info("invokeLocal:list:02");
126  
127                         } else if("count".equals(type)){
128                                 
129                                 log.info("PermList:count:01");
130                                 auditCount = (Long)entityManager.createQuery(
131                                                 "select count(o) " +
132                                         "from AcPermissionsList o ")
133                                .getSingleResult();
134                                 
135               log.info("Perm:invokeLocal:count:02:"+auditCount);
136                 } 
137                }catch(Exception ePerm){
138                          log.error("Perm:invokeLocal:error:"+ePerm);
139                          evaluteForList=false;
140                          FacesMessages.instance().add("ќшибка!");
141                }

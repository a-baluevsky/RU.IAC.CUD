152     *

168            GeneralSecurityException, MarshalException, XMLSignatureException {

173    /**
174     * Sign a node in a document
175     *
176     * @param doc
177     * @param nodeToBeSigned
178     * @param keyPair
179     * @param publicKey
180     * @param digestMethod
181     * @param signatureMethod
182     * @param referenceURI
183     * @return
184     * @throws ParserConfigurationException
185     * @throws XMLSignatureException
186     * @throws MarshalException
187     * @throws GeneralSecurityException
188     */
189    public static Document sign(Document doc, Node nodeToBeSigned, KeyPair keyPair, String digestMethod,
190            String signatureMethod, String referenceURI) throws ParserConfigurationException, GeneralSecurityException,
191            MarshalException, XMLSignatureException {
192        if (nodeToBeSigned == null) {
193                        throw LOGGER.nullArgumentError("Node to be signed");
194                }
195        
196        if (LOGGER.isTraceEnabled()) {
197            LOGGER.trace("Document to be signed=" + DocumentUtil.asString(doc));
198        }
199
200        Node parentNodeSign = nodeToBeSigned.getParentNode();
201
202        // Let us create a new Document
203        Document newDoc = DocumentUtil.createDocument();
204        // Import the node
205        Node signingNodeSig = newDoc.importNode(nodeToBeSigned, true);
206        newDoc.appendChild(signingNodeSig);
207
208        if (!referenceURI.isEmpty()) {
209            propagateIDAttributeSetup(nodeToBeSigned, newDoc.getDocumentElement());
210        }
211        newDoc = sign(newDoc, keyPair, digestMethod, signatureMethod, referenceURI);
212
213        // if the signed element is a SAMLv2.0 assertion we need to move the signature element to the position
214        // specified in the schema (before the assertion subject element).
215        if ("Assertion".equals(nodeToBeSigned.getLocalName())
216                && WSTrustConstants.SAML2_ASSERTION_NS.equals(nodeToBeSigned.getNamespaceURI())) {
217            Node signatureNode = DocumentUtil.getElement(newDoc, new QName(WSTrustConstants.DSIG_NS, "Signature"));
218            Node subjectNode = DocumentUtil.getElement(newDoc, new QName(WSTrustConstants.SAML2_ASSERTION_NS, "Subject"));
219            if (signatureNode != null && subjectNode != null) {
220                newDoc.getDocumentElement().removeChild(signatureNode);
221                newDoc.getDocumentElement().insertBefore(signatureNode, subjectNode);
222            }
223        }
224
225        // Now let us import this signed doc into the original document we got in the method call
226        Node signedNode = doc.importNode(newDoc.getFirstChild(), true);
227
228        if (!referenceURI.isEmpty()) {
229            propagateIDAttributeSetup(newDoc.getDocumentElement(), (Element) signedNode);
230        }
231
232        parentNodeSign.replaceChild(signedNode, nodeToBeSigned);
233        
234
235        return doc;
236    }

348    /**
349     * Setup the ID attribute into <code>destElement</code> depending on the <code>isId</code> flag of an attribute of
350     * <code>sourceNode</code>.
351     *
352     * @param sourceNode
353     * @param destDocElement
354     */
355    public static void propagateIDAttributeSetup(Node sourceNode, Element destElement) {
356        NamedNodeMap nnm = sourceNode.getAttributes();
357        for (int i = 0; i < nnm.getLength(); i++) {
358            Attr attr = (Attr) nnm.item(i);
359            if (attr.isId()) {
360                destElement.setIdAttribute(attr.getName(), true);
361                break;
362            }
363        }
364    }

450    /**
451     * Validate a signed document with the given public key
452     *
453     * @param signedDoc
454     * @param publicKey
455     * @return
456     * @throws MarshalException
457     * @throws XMLSignatureException
458     */
459    @SuppressWarnings("unchecked")
460    public static boolean validate(Document signedDoc, Key publicKey) throws MarshalException, XMLSignatureException {
461        if (signedDoc == null) {
462                        throw LOGGER.nullArgumentError("Signed Document");
463                }
464
465         
466        propagateIDAttributeSetup(signedDoc.getDocumentElement(), signedDoc.getDocumentElement());
467
468        NodeList nl = signedDoc.getElementsByTagNameNS(XMLSignature.XMLNS, "Signature");
469        if (nl == null || nl.getLength() == 0) {
470            throw LOGGER.nullValueError("Cannot find Signature element");
471        }
472        if (publicKey == null) {
473                        throw LOGGER.nullValueError("Public Key");
474                }
475
476        //написан GostKeyHandler, где GOSTKeyValue из запроса помещается в "SENDER_PUBLIC_KEY"
477        //а здесь это будет родной параметр publicKey.
478        
479        DOMValidateContext valContext = new DOMValidateContext(publicKey, nl.item(0));
480        XMLSignature signature = fac.unmarshalXMLSignature(valContext);
481
482        boolean coreValidity = signature.validate(valContext);
483
484         
485        if (LOGGER.isTraceEnabled() && !coreValidity) {
486            boolean sv = signature.getSignatureValue().validate(valContext);
487            LOGGER.trace("Signature validation status: " + sv);
488
489            List<Reference> references = signature.getSignedInfo().getReferences();
490            for (Reference ref : references) {
491                LOGGER.trace("[Ref id=" + ref.getId() + ":uri=" + ref.getURI() + "]validity status:" + ref.validate(valContext));
492            }
493        }
494        return coreValidity;
495    }

523    /**
524     * Given the X509Certificate in the keyinfo element, get a {@link X509Certificate}
525     *
526     * @param certificateString
527     * @return
528     * @throws ProcessingException
529     */
530    public static X509Certificate getX509CertificateFromKeyInfoString(String certificateString) throws ProcessingException {
531        X509Certificate cert = null;
532        StringBuilder builder = new StringBuilder();
533        builder.append("-----BEGIN CERTIFICATE-----\n").append(certificateString).append("\n-----END CERTIFICATE-----");
534
535        String derFormattedString = builder.toString();
536
537        try {
538            CertificateFactory cf = CertificateFactory.getInstance("X.509");
539            ByteArrayInputStream bais = new ByteArrayInputStream(derFormattedString.getBytes());
540
541            while (bais.available() > 0) {
542                cert = (X509Certificate) cf.generateCertificate(bais);
543            }
544        } catch (java.security.cert.CertificateException e) {
545            throw LOGGER.processingError(e);
546        }
547        return cert;
548    }

551    private static void signImpl(DOMSignContext dsc, String digestMethod, String signatureMethod, String referenceURI, PublicKey publicKey,
552                                 X509Certificate x509Certificate)
553            throws GeneralSecurityException, MarshalException, XMLSignatureException {
554        
555        
556        
557        dsc.setDefaultNamespacePrefix("dsig");
558
559        DigestMethod digestMethodObj = fac.newDigestMethod(digestMethod, null);
560        Transform transform1 = fac.newTransform(Transform.ENVELOPED, (TransformParameterSpec) null);
561        Transform transform2 = fac.newTransform("http://www.w3.org/2001/10/xml-exc-c14n#", (TransformParameterSpec) null);
562
563        List<Transform> transformList = new ArrayList<Transform>();
564        transformList.add(transform1);
565        transformList.add(transform2);
566
567        Reference ref = fac.newReference(referenceURI, digestMethodObj, transformList, null, null);
568
569        CanonicalizationMethod canonicalizationMethod = fac.newCanonicalizationMethod(canonicalizationMethodType,
570            (C14NMethodParameterSpec) null);
571
572        List<Reference> referenceList = Collections.singletonList(ref);
573        SignatureMethod signatureMethodObj = fac.newSignatureMethod(signatureMethod, null);
574        SignedInfo si = fac.newSignedInfo(canonicalizationMethod, signatureMethodObj, referenceList);
575
576           
577        KeyInfo ki = null;
578        if(includeKeyInfoInSignature){
579            ki = createKeyInfo(publicKey,x509Certificate);
580        }
581        XMLSignature signature = fac.newXMLSignature(si, ki);
582
583        signature.sign(dsc);
584    }

586    private static KeyInfo createKeyInfo(PublicKey publicKey, X509Certificate x509Certificate) throws KeyException {
587        KeyInfoFactory keyInfoFactory = fac.getKeyInfoFactory();
588        KeyInfo keyInfo = null;
589        KeyValue keyValue = null;
590        //Just with public key
591        if(publicKey != null){
592            keyValue = keyInfoFactory.newKeyValue(publicKey);
593            keyInfo = keyInfoFactory.newKeyInfo(Collections.singletonList(keyValue));
594        }
595        if(x509Certificate != null){
596            List x509list = new ArrayList();
597
598            x509list.add(x509Certificate);
599            X509Data x509Data = keyInfoFactory.newX509Data(x509list);
600            List items = new ArrayList();
601
602            items.add(x509Data);
603            if(keyValue != null){
604                items.add(keyValue);
605            }
606            keyInfo = keyInfoFactory.newKeyInfo(items);
607        }
608        return keyInfo;
609    }

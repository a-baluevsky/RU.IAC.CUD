092        public void initProcBNABean(){
093                 log.info("procBindNoActManager:initProcBNABean:01");
094                 
095                 String startDateValue=null, periodValue=null, status=null;
096                 Properties properties = new Properties();
097                 String path = proc_binding_noact_exec_file;
098                 InputStream is = null;
099                 
100                 String  remoteAudit = FacesContext.getCurrentInstance().getExternalContext()
101                                .getRequestParameterMap()
102                                .get("remoteAudit");
103                 log.info("procBindNoActManager:initProcBNABean:remoteAudit:"+remoteAudit);
104                
105                 procBNABean= new ProcBNAItem();
106                 
107                 if(remoteAudit!=null && "procInfo".equals(remoteAudit)){
108                         // кнопка "обновить" задействована для обновления и Center, и Bottom панелей 
109                     // чтобы отобразить изменения, если кто другой запустил/остановил процесс
110                        
111                         log.info("confLogContrManager:initProcBindNoActBean:return");
112                 }
113                 
114                 if(remoteAudit!=null && "procCrt".equals(remoteAudit)){
115                                 procBNABean.setStatus("passive");
116                         return;
117                 }
118                 if(remoteAudit!=null && "procDel".equals(remoteAudit)){
119                         procBNABean.setStatus("active");
120                         return;
121                 }
122                 if(remoteAudit!=null && "procPause".equals(remoteAudit)){
123                         procBNABean.setStatus("active");
124                         return;
125                 }
126                 if(remoteAudit!=null && "procRun".equals(remoteAudit)){
127                         procBNABean.setStatus("pause");
128                         return;
129                 }
130                 
131                 try {
132                         DateFormat df = new SimpleDateFormat ("dd.MM.yy HH:mm");
133                    
134                   
135                     
136                     File f=new File(path); 
137                     
138                     if(f.exists()) { 
139                         
140                         properties.load(is=new FileInputStream(f));
141                         
142                         startDateValue=properties.getProperty("start_date");
143                         periodValue=properties.getProperty("period");
144                         status=properties.getProperty("status");
145                         
146                        log.info("procBindNoActManager:initProcBNABean:start_date:"+startDateValue);
147                         log.info("procBindNoActManager:initProcBNABean:period:"+periodValue);
148                         log.info("procBindNoActManager:initProcBNABean:status:"+status);
149                         
150                         if(startDateValue!=null&&periodValue!=null&&status!=null){
151                                 if("active".equals(status)||"pause".equals(status)){
152                                         procBNABean.setStartDate(df.parse(startDateValue));
153                                         procBNABean.setPeriod(new Long(periodValue));
154                                   }
155                                 procBNABean.setStatus(status);
156                                 }
157                      }else{
158                          procBNABean.setStatus("passive");
159                      }
160                 }catch (Exception e) {
161                                log.error("procBindNoActManager:initProcBNABean:error:"+e);
162                 }finally{
163                        try {
164                          if(is!=null){
165                            is.close();
166                           }
167                        } catch (Exception e) {
168                                log.error("procBindNoActManager:initProcBNABean:finally:is:error:"+e);
169                        }
170           }    
171        }

216        public synchronized void procCrt(){
217                  log.info("procBindNoActManager:procCrt:01");
218                  log.info("procBindNoActManager:procCrt:startDate:"+startDate);
219                  log.info("procBindNoActManager:procCrt:period:"+period);
220                  
221                  Properties properties = new Properties();
222                  String path = proc_binding_noact_exec_file;
223                  OutputStream os = null;
224                  
225                  if(this.period==null || this.startDate==null){
226                          log.info("procBindNoActManager:procCrt:02");
227                          return;
228                  }
229                   
230                  try {
231                         DateFormat df = new SimpleDateFormat ("dd.MM.yy HH:mm");
232                   
233                   
234                        
235                     File fBna=new File(path); 
236                     
237                       properties.setProperty("start_date", df.format(this.startDate));
238                       properties.setProperty("period", this.period.toString());
239                       properties.setProperty("status", "active");
240                       
241                       properties.store(os=new FileOutputStream(fBna), null);
242                       
243                     
244                  
245                 
246                       Context ctx = new InitialContext(); 
247                         
248                   BaseParamItem bpiBna = new BaseParamItem();
249                   bpiBna.put("gactiontype", ServiceReestrAction.PROCESS_START.name());
250                   
251                   bpiBna.put("startDate", this.startDate);
252                   bpiBna.put("period", this.period);
253                   
254                   ((IHLocal)ctx.lookup(ServiceReestr.BindingNoAct)).run(bpiBna);
255                   
256                   log.info("procBindNoActManager:procCrt:03");
257          
258                   forView("procCrt");
259                   
260                   audit(ResourcesMap.PROC_BIND_NOACT, ActionsMap.START); 
261                   
262                  }catch (Exception eBna) {
263                                log.error("procBindNoActManager:procCrt:error:"+eBna);
264                  }finally{
265                         try {
266                                if(os!=null){
267                                         os.close();
268                                }
269                         } catch (Exception eBna) {
270                                log.error("procBindNoActManager:procCrt:os:error:"+eBna);
271                         }
272                 }

333                String startDateValue=null, periodValue=null;
334                try {
335                   Context ctx = new InitialContext();
336                   
337                   BaseParamItem bpi = new BaseParamItem();
338               bpi.put("gactiontype", ServiceReestrAction.PROCESS_STOP.name());
339               ((IHLocal)ctx.lookup(ServiceReestr.BindingNoAct)).run(bpi);
340                   
341                   log.info("procBindNoActManager:procPause:02");
342                   
343                   Properties properties = new Properties();
344                   String path = proc_binding_noact_exec_file;
345                  
346                   
347              
348               
349               File f=new File(path); 
350               
351               if(f.exists()) {
352              
353               
354               
355                  properties.load(is=new FileInputStream(f));
356               
357                  periodValue=properties.getProperty("period");
358                  startDateValue=properties.getProperty("start_date");
359              
360                  if(periodValue==null || startDateValue==null){
361                            log.info("procBindNoActManager:procRun:02");

363                      }
364                
365                  this.startDate=dfBna.parse(startDateValue);
366                  this.period=new Long(periodValue);

368                  properties.setProperty("status", "pause");

371                  forView("procPause");
372               }
373               
374               audit(ResourcesMap.PROC_BIND_NOACT, ActionsMap.PAUSE);
375               
376        }catch (Exception eBna) {
377                                log.error("procBindNoActManager:procPause:error:"+eBna);

380                        if(os!=null){

460        private void forView(String typeBna){
461           try {
462                   
463                   procBNABean= new ProcBNAItem();
464                   
465                  if(typeBna.equals("procCrt")){
466                    this.procBNABean.setPeriod(this.period);
467                    this.procBNABean.setStartDate(this.startDate);
468                 
469                    procBNABean.setStatus("active");
470                  }else if (typeBna.equals("procDel")){ 
471                          procBNABean.setStatus("passive");
472                  }else if (typeBna.equals("procPause")){
473                        this.procBNABean.setPeriod(this.period);
474                        this.procBNABean.setStartDate(this.startDate);
475                        procBNABean.setStatus("pause");
476                  }else if (typeBna.equals("procRun")){
477                        this.procBNABean.setPeriod(this.period);
478                        this.procBNABean.setStartDate(this.startDate);
479                        procBNABean.setStatus("active");
480                  }
481                  
482                  Contexts.getEventContext().set("procBNABean", this.procBNABean);
483                  
484           }catch (Exception eBna) {
485                  log.error("procBindNoActManager:forView:error:"+eBna);
486           }

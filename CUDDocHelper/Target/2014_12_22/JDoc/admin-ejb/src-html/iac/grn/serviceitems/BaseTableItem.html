005 public class BaseTableItem {
006
007        private String itemLabel;
008        private String itemColumn;
009        private String itemField;
010        private String itemSortField;
011        private String itemFiltField;
012        private HeaderTableItem header;
013        private int itemSort;
014        
015        public BaseTableItem(String itemLabel, String itemColumn, String itemField){
016                this.itemLabel=itemLabel;
017                this.itemColumn=itemColumn;
018                this.itemField=itemField;
019        }
020        
021        public BaseTableItem(String itemLabel, String itemColumn, String itemField, int itemSort){
022                this.itemLabel=itemLabel;
023                this.itemColumn=itemColumn;
024                this.itemField=itemField;
025                this.itemSort=itemSort;
026        }
027        
028        public BaseTableItem(String itemLabel, String itemColumn, String itemField, String itemSortField, int itemSort){
029                this.itemLabel=itemLabel;
030                this.itemColumn=itemColumn;
031                this.itemField=itemField;
032                this.itemSortField=itemSortField;
033                this.itemSort=itemSort;
034        }
035        
036        public BaseTableItem(String itemLabel, String itemColumn, String itemField, String itemSortField, int itemSort, HeaderTableItem header){
037                this.itemLabel=itemLabel;
038                this.itemColumn=itemColumn;
039                this.itemField=itemField;
040                this.itemSortField=itemSortField;
041                this.itemSort=itemSort;
042                this.header=header;
043                
044                this.header.getItems().add(this);
045        }
046        
047        public BaseTableItem(String itemLabel, String itemColumn, String itemField, String itemSortField, String itemFiltField, int itemSort){
048                this.itemLabel=itemLabel;
049                this.itemColumn=itemColumn;
050                this.itemField=itemField;
051                this.itemSortField=itemSortField;
052                this.itemFiltField=itemFiltField;
053                this.itemSort=itemSort;
054        }
055        
056        public String getItemLabel(){
057                return this.itemLabel;
058        }
059        public void setItemLabel(String itemLabel){
060                this.itemLabel=itemLabel;
061        }
062        
063        public String getItemColumn(){
064                return this.itemColumn;
065        }
066        public void setItemColumn(String itemColumn){
067                this.itemColumn=itemColumn;
068        }
069        
070        public String getItemField(){
071                return this.itemField;
072        }
073        public void setItemField(String itemField){
074                this.itemField=itemField;
075        }
076        
077        public String getItemSortField(){
078                if(this.itemSortField==null){
079                        this.itemSortField=this.itemField;
080                }
081                return this.itemSortField;
082        }
083        public void setItemSortField(String itemSortField){
084                this.itemSortField=itemSortField;
085        }
086        
087        public String getItemFiltField(){
088                if(this.itemFiltField==null){
089                        this.itemFiltField=this.itemField;
090                }
091                return this.itemFiltField;
092        }
093        public void setItemFiltField(String itemFiltField){
094                this.itemFiltField=itemFiltField;
095        }
096        
097        public int getItemSort(){
098                return this.itemSort;
099        }
100        public void setItemSort(int itemSort){
101                this.itemSort=itemSort;
102        }
103        
104        public HeaderTableItem getHeader(){
105                return this.header;
106        }
107        public void setHeader(HeaderTableItem header){
108                this.header=header;
109        }
110        
111        @Override
112        public boolean equals(Object o) {
113                if (o == this) {
114                        return true;
115                }
116                if ( ! (o instanceof BaseTableItem)) {
117                        return false;
118                }
119                BaseTableItem other = (BaseTableItem) o;
120                return this.itemField.equals(other.itemField);
121        }
122        
123        @Override
124        public int hashCode() {
125                return this.itemField.hashCode()
126                         ^ this.itemField.hashCode()
127                        ;
128        }

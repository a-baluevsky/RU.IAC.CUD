001package iac.grn.infosweb.context.proc;
002
003import java.util.Map;
004import java.util.Set;
005import java.util.concurrent.ConcurrentHashMap;
006import java.util.concurrent.ScheduledFuture;
007
008import org.slf4j.Logger;
009import org.slf4j.LoggerFactory;
010
011 public class TaskProcessor {
012
013        private static final Logger LOGGER = LoggerFactory.getLogger(TaskProcessor.class);
014        
015        private static volatile ConcurrentHashMap<String, ScheduledFuture> controls = new ConcurrentHashMap();
016        
017        public static ConcurrentHashMap<String, ScheduledFuture> getControls(){
018                
019                Set<Map.Entry<String, ScheduledFuture>> set = controls.entrySet();
020        for (Map.Entry<String, ScheduledFuture> me : set) {
021                LOGGER.debug("TaskProcessor:key:"+me.getKey());
022                if(me.getValue()!=null) {
023                  LOGGER.debug("TaskProcessor:isCancelled:"+me.getValue().isCancelled());
024                }else{
025                  LOGGER.debug("TaskProcessor:value:is null");  
026                }
027                 }
028                
029                return controls;
030        }
031        
032        public static void setControls(ConcurrentHashMap<String, ScheduledFuture> pcontrols){
033                controls=pcontrols;
034        }
035}

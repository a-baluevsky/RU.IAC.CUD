029        public List<BaseTableItem> getAuditItemsCollection() {
030
031                List<BaseTableItem> btiList = new ArrayList<BaseTableItem>(
032                                this.auditItemsMap.values());
033                Collections.sort(btiList, new Comparator<BaseTableItem>() {
034
035                        public int compare(BaseTableItem o1, BaseTableItem o2) {
036
037                                int sort1 = o1.getItemSort();
038                                int sort2 = o2.getItemSort();
039
040                                if (sort1 > sort2) {
041                                        return 1;
042                                } else if (sort1 < sort2) {
043                                        return -1;
044                                } else {
045                                        return 0;
046                                }
047                        }
048                });
049
050                return btiList;
051        }

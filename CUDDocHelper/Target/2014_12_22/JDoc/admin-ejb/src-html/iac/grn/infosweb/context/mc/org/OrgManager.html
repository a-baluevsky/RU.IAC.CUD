057        public List<BaseItem> getAuditList(int firstRow, int numberOfRows){
058          String remoteAudit = FacesContext.getCurrentInstance().getExternalContext()
059                     .getRequestParameterMap()
060                     .get("remoteAudit");
061           log.info("auditManager:getAuditList:remoteAudit:"+remoteAudit);
062          
063          
064          log.info("getAuditList:firstRow:"+firstRow);
065          log.info("getAuditList:numberOfRows:"+numberOfRows);
066          
067          List<BaseItem> orgListCached = (List<BaseItem>)
068                          Component.getInstance("orgListCached",ScopeType.SESSION);
069          if(auditList==null){
070                  log.info("getAuditList:01");
071                        if(("rowSelectFact".equals(remoteAudit)||
072                            "selRecAllFact".equals(remoteAudit)||
073                            "clRecAllFact".equals(remoteAudit)||
074                            "clSelOneFact".equals(remoteAudit)||
075                            "onSelColSaveFact".equals(remoteAudit))&&
076                            orgListCached!=null){
077                                this.auditList=orgListCached;
078                        }else{
079                                log.info("getAuditList:03");
080                        invokeLocal("list", firstRow, numberOfRows, null);
081                            Contexts.getSessionContext().set("orgListCached", this.auditList);
082                            log.info("getAuditList:03:"+this.auditList.size());
083                        }
084                        
085                        List<String>  selRecOrg = (ArrayList<String>)
086                                          Component.getInstance("selRecOrg",ScopeType.SESSION);
087                        if(this.auditList!=null && selRecOrg!=null) {
088                                 for(BaseItem it:this.auditList){
089                                   if(selRecOrg.contains(it.getBaseId().toString())){
090                                         
091                                          it.setSelected(true);
092                                   }else{
093                                          it.setSelected(false);
094                                   }
095                                 }
096                      }
097                }
098                return this.auditList;
099        }

103        public void invokeLocal(String type, int firstRow, int numberOfRows,
104                   String sessionId) {
105                try{
106                         String orderQueryOrg=null;
107                         log.info("OrgManager:invokeLocal");
108                         
109                         if("list".equals(type)){
110                                 log.info("Org:invokeLocal:list:01");
111                                 
112                                 OrgStateHolder orgStateHolder = (OrgStateHolder)
113                                                  Component.getInstance("orgStateHolder",ScopeType.SESSION);
114                                 Set<Map.Entry<String, String>> set = orgStateHolder.getSortOrders().entrySet();
115                 for (Map.Entry<String, String> me : set) {
116                       
117                       if(orderQueryOrg==null){
118                         orderQueryOrg="order by "+me.getKey()+" "+me.getValue();
119                       }else{
120                         orderQueryOrg=orderQueryOrg+", "+me.getKey()+" "+me.getValue();  
121                       }
122                     }
123                 log.info("invokeLocal:list:orderQueryOrg:"+orderQueryOrg);
124                 
125                                 auditList = entityManager.createQuery("select o from AcOrganization o "+(orderQueryOrg!=null ? orderQueryOrg : ""))
126                       .setFirstResult(firstRow)
127                       .setMaxResults(numberOfRows)
128                       .getResultList();
129             log.info("invokeLocal:list:02");
130  
131                         } else if("count".equals(type)){
132                                 log.info("OrgList:count:01");
133                                 auditCount = (Long)entityManager.createQuery(
134                                                 "select count(o) " +
135                                         "from AcOrganization o ")
136                                .getSingleResult();
137                                 
138               log.info("Org:invokeLocal:count:02:"+auditCount);
139                 } 
140                }catch(Exception e){
141                          log.error("invokeLocal:error:"+e);
142                          evaluteForList=false;
143                          FacesMessages.instance().add("Ошибка!");
144                }
145        }

219   public void updOrg(){
220           
221           log.info("orgManager:updOrg:01");
222           
223           AcOrganization orgBean = (AcOrganization)
224                                  Component.getInstance("orgBean",ScopeType.CONVERSATION);
225           
226           String  sessionId = FacesContext.getCurrentInstance().getExternalContext()
227                        .getRequestParameterMap()
228                        .get("sessionId");
229           log.info("orgManager:updOrg:sessionId:"+sessionId);
230        
231           if(orgBean==null || sessionId==null){
232                   return;
233           }
234        
235           try {
236                   AcUser au = (AcUser) Component.getInstance("currentUser",ScopeType.SESSION);
237                   
238                 
239                  AcOrganization aom = entityManager.find(AcOrganization.class, new Long(sessionId));
240                  
241                  aom.setFullName(orgBean.getFullName());
242                  aom.setShortName(orgBean.getShortName());
243                  aom.setContactEmployeeFio(orgBean.getContactEmployeeFio());
244                  aom.setContactEmployeePosition(orgBean.getContactEmployeePosition());
245                  aom.setContactEmployeePhone(orgBean.getContactEmployeePhone());
246                  aom.setAcLegalEntityType(orgBean.getAcLegalEntityType());
247                  aom.setIsExternal(orgBean.getIsExternal());
248                  
249                  aom.setModificator(au.getIdUser());
250                  aom.setModified(new Date());
251                  
252                  entityManager.flush();
253              entityManager.refresh(aom);
254                  
255              Contexts.getEventContext().set("orgBean", aom);
256                  
257             }catch (Exception e) {
258           log.error("orgManager:updOrg:ERROR:"+e);

383            String  sessionId = FacesContext.getCurrentInstance().getExternalContext()
384                        .getRequestParameterMap()
385                        .get("sessionId");
386            log.info("selectRecord:sessionId="+sessionId);
387            
388           //  for/View(/); //!!!
389            List<String>  selRecOrg = (ArrayList<String>)
390                                  Component.getInstance("selRecOrg",ScopeType.SESSION);
391            
392            if(selRecOrg==null){
393               selRecOrg = new ArrayList<String>();
394               log.info("selectRecord:01");
395            }
396            
397          
398           AcOrganization ao = new AcOrganization();
399         
400            
401            if(ao!=null){
402             if(selRecOrg.contains(sessionId)){
403                selRecOrg.remove(sessionId);
404                ao.setSelected(false);
405                log.info("selectRecord:02");
406             }else{
407                selRecOrg.add(sessionId);
408                ao.setSelected(true);
409                log.info("selectRecord:03");
410             }
411            Contexts.getSessionContext().set("selRecOrg", selRecOrg);   
412            
413            Contexts.getEventContext().set("orgBean", ao);
414            }
415          }

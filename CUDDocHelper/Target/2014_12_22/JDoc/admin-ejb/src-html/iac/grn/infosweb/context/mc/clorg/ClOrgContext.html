021                
022                List<BaseTableItem> btiList = new ArrayList<BaseTableItem>(this.auditItemsMap.values());
023                Collections.sort(btiList, new Comparator<BaseTableItem>() {
024
025                public int compare(BaseTableItem o1, BaseTableItem o2) {
026                        
027                        int sort1=o1.getItemSort();
028                        int sort2=o2.getItemSort();
029                        
030                        if(sort1 > sort2){
031                        return 1;
032                    }else if(sort1 < sort2){
033                        return -1; 
034                    }else{
035                        return 0;
036                    }    
037                 }
038            });
039
040        return btiList;
041    }

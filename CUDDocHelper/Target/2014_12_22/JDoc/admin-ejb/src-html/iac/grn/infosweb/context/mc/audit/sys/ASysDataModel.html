024        @Logger private Log log;

026        @In(create=true)

028

032        /**
033         * This is main part of Visitor pattern. Method called by framework many times during request processing. 
034         */
035        @Override
036        public void walk(FacesContext context, DataVisitor visitor, Range range, Object argument) throws IOException {
037                
038                log.info("walk:01:start");
039                
040                
041                int firstRow = ((SequenceRange)range).getFirstRow();
042                int numberOfRows = ((SequenceRange)range).getRows();
043                
044                log.info("walk:firstRow:"+firstRow);
045                log.info("walk:numberOfRows:"+numberOfRows);
046                log.info("walk:cachedItems:01:"+(this.cachedItems==null));
047                
048                wrappedKeys = new ArrayList<Long>();
049                if(this.cachedItems==null){
050                        log.info("walk:cachedItems:02:");
051                         this.cachedItems=getDataProvider().getItemsByrange(firstRow, numberOfRows, null, true);
052                }
053                log.info("walk:cachedItems:03:");
054                if(this.cachedItems!=null){
055                  for (BaseItem itemASys:cachedItems) {
056                        wrappedKeys.add(itemASys.getBaseId());
057                        wrappedData.put(itemASys.getBaseId(), itemASys);  
058                         
059                    visitor.process(context, itemASys.getBaseId(), argument);
060                         
061                 }
062                }
063                log.info("walk:end");
064        }
065        public int getRowCount2() {
066                log.info("auditDataModel:getRowCount2!!!");
067                return getRowCount();
068        }
069        /**
070         * This method must return actual data rows count from the Data Provider. It is used by pagination control
071         * to determine total number of data items.
072         */
073         // better to buffer row count locally
074        @Override
075        public int getRowCount() {
076                
077        log.info("auditDataModel:getRowCount:01");
078                String remoteAuditASys = FacesContext.getCurrentInstance().getExternalContext()
079                         .getRequestParameterMap()
080                         .get("remoteAudit");
081                String  auditListCount = FacesContext.getCurrentInstance().getExternalContext()
082                                .getRequestParameterMap()
083                                .get("auditListCount");
084                log.info("auditDataModel:getRowCount:remoteAudit:"+remoteAuditASys);
085                log.info("auditDataModel:getRowCount:auditListCount:"+auditListCount);
086        
087                if(this.flagAction==0){
088                        if (rowCount==null) {
089                                if(auditListCount!=null){
090                                   rowCount = new Integer(auditListCount);
091                                 }else{
092                                   rowCount = 0;
093                                 }
094                                log.info("getRowCount:02:rowCount:"+rowCount);
095                        }
096                }else{
097                        if (rowCount==null) {
098                                // При selRecAllFact, clRecAllFact, clSelOneFact запросах
099                                // dataScroller не рендерится, а в параметрах
100                                // rowCount и так определяется через param['auditListCount']
101                                if(remoteAuditASys==null){
102                                        log.info("getRowCount:03_+");
103                                        return 0;
104                                }
105                                if(remoteAuditASys!=null &&
106                                        ("rowSelectFact".equals(remoteAuditASys)/*||
107                                    "selRecAllFact".equals(remoteAudit)||
108                                        "clRecAllFact".equals(remoteAudit)||
109                                        "clSelOneFact".equals(remoteAudit)*/)&&
110                                        auditListCount!=null){
111                                        rowCount = new Integer(auditListCount);
112                                }else{
113                                        rowCount = new Integer(
114                                                        getDataProvider().getRowCount());
115                                }
116                                log.info("getRowCount:03:rowCount:"+rowCount);
117                        }
118                }
119                return rowCount.intValue();
120        }

131        }

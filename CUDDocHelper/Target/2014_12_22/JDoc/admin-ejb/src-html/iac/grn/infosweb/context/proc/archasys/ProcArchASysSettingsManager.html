007import java.io.InputStream;
008import java.io.OutputStream;
009import java.util.List;
010import javax.faces.context.FacesContext;
011import javax.persistence.EntityManager;
012
013import org.jboss.seam.Component;
014import org.jboss.seam.ScopeType;
015import org.jboss.seam.annotations.In;
016import org.jboss.seam.annotations.Logger;
017import org.jboss.seam.annotations.Name;
018import org.jboss.seam.contexts.Contexts;
019import org.jboss.seam.log.Log;
020
021@Name("procArchASysSettingsManager")
022 public class ProcArchASysSettingsManager {
023
024        @Logger private Log log;
025        
026        @In
027        EntityManager entityManager;
028         
029        
030        private static String param_code="to_archive_audit_sys";
031        
032        public void init(){
033                
034                String monthInterval = null;
035                
036                try{
037                   log.info("ConfLoadDataSettingsManager:init:01");
038                   
039                    String  remoteAuditASys = FacesContext.getCurrentInstance().getExternalContext()
040                                .getRequestParameterMap()
041                                .get("remoteAudit");
042                    log.info("confLoadDataManager:init:remoteAudit:"+remoteAuditASys);
043        
044                     if(remoteAuditASys!=null ){ 
045                         //при сохранении настроек
046                         //procArchASysSettingsBean устанавливать не нужно
047                             //он автоматически продолжается в EventContext
048                             //от оправки формы до её отображения
049                           return;
050                         }
051                   
052                    
053                     ProcArchASysSettingsBean beanSettingsASys = new ProcArchASysSettingsBean();
054                        
055                     
056                                 
057                         
058                         log.info("ConfLoadDataSettingsManager:getCLDSBeanView:01");
059                         
060                         InputStream is = null;
061                         
062                         try {
063                                 
064                                                 List<String> los = entityManager.createNativeQuery(
065                                      "select ST.VALUE_PARAM "+
066                              "from SETTINGS_KNL_T st "+
067                              "where ST.SIGN_OBJECT=? ")
068                              .setParameter(1, param_code)
069                              .getResultList();
070                          
071                          if(los!=null&&!los.isEmpty()){
072                            monthInterval=los.get(0);
073                          }
074                          
075                          if(monthInterval==null){ 
076                                  monthInterval="6";
077                          }              
078                             
079                          beanSettingsASys.setParamActualData(new Long(monthInterval));
080                          
081                              Contexts.getEventContext().set("procArchASysSettingsBean", beanSettingsASys);
082                             
083                              
084                              
085                          }catch (Exception eASys) {
086                                        log.error("confLoadDataManager:initConfLDInfoBean:error:"+eASys);
087                         }finally{
088                                try {
089                                  if(is!=null){
090                                    is.close();
091                                   }
092                                } catch (Exception eASys) {
093                                        log.error("confLoadDataManager:initConfLDInfoBean:finally:is:error:"+eASys);
094                                }
095                   }    
096                  
097                   
098                }catch(Exception eASys){
099                   log.error("confLoadDataSettingsManager:init:ERROR:"+eASys);
100                } 
101        }     
102        
103        public void save(){
104                try{
105                        
106                   log.info("confLoadDataSettingsManager:save:01");
107                  
108                   OutputStream os = null;
109                         
110                   ProcArchASysSettingsBean beanSettingsASys = (ProcArchASysSettingsBean) 
111                                   Contexts.getEventContext().get("procArchASysSettingsBean");
112                   
113                   
114                   
115                          if(beanSettingsASys.getParamActualData()==null){
116                                  log.info("confLoadDataSettingsManager:save:02");
117                                  return;
118                          }
119                           
120                          try {
121                                          entityManager.createNativeQuery(
122                                      "update SETTINGS_KNL_T st " +
123                                      "set ST.VALUE_PARAM=? "+
124                              "where ST.SIGN_OBJECT=? ")
125                              .setParameter(1, beanSettingsASys.getParamActualData())
126                              .setParameter(2, param_code)
127                              .executeUpdate();
128                              
129                             log.info("confLoadDataSettingsManager:save:03");
130                  
131                             //procArchASysSettingsBean устанавливать не нужно
132                             //он автоматически продолжается в EventContext
133                             //от оправки формы до её отображения
134                             
135                             audit(ResourcesMap.PROC_ARCH_AUDIT_SYS, ActionsMap.SET_PARAM);
136                             
137                          }catch (Exception eASys) {
138                                        log.error("confLoadDataSettingsManager:save:"+eASys);
139                          }finally{
140                                 try {
141                                        if(os!=null){
142                                                 os.close();
143                                        }
144                                 } catch (Exception eASys) {
145                                        log.error("confLoadDataSettingsManager:save:os:error:"+eASys);
146                                 }
147                         }
148                           
149                }catch(Exception eASys){
150                   log.error("ConfLoadDataSettingsManager:save:ERROR:"+eASys);
151                } 
152        }     
153        
154        public void audit(ResourcesMap resourcesMap, ActionsMap actionsMap){
155                   try{
156                           AuditExportData auditExportDataASys = (AuditExportData)Component.getInstance("auditExportData",ScopeType.SESSION);
157                           auditExportDataASys.addFunc(resourcesMap.getCode()+":"+actionsMap.getCode());
158                           
159                   }catch(Exception eASys){
160                           log.error("GroupManager:audit:error:"+eASys);
161                   }
162          }

166}

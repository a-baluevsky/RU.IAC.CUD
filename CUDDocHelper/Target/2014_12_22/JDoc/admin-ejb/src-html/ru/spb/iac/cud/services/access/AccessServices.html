001package ru.spb.iac.cud.services.access;
002
003import java.util.List;
004
005import javax.jws.WebParam;
006import javax.jws.WebResult;
007import javax.jws.WebService;
008import ru.spb.iac.cud.exceptions.GeneralFailure;
009import ru.spb.iac.cud.exceptions.InvalidCredentials;
010import ru.spb.iac.cud.exceptions.TokenExpired;
011import ru.spb.iac.cud.items.Attribute;
012import ru.spb.iac.cud.items.AuditFunction;
013import ru.spb.iac.cud.items.Role;
014import ru.spb.iac.cud.items.Token;
015
016@WebService(targetNamespace = "http://access.services.cud.iac/")
017public interface AccessServices {
018        
019         public static final String NS = "http://access.services.cud.iac/";
020        
021         @WebResult(targetNamespace = NS)
022         public Token authenticate_login(
023                         @WebParam(name = "login", targetNamespace = NS) String login, 
024                         @WebParam(name = "password", targetNamespace = NS) String password) throws GeneralFailure, InvalidCredentials;
025    
026         public Token authenticate_cert() throws GeneralFailure, InvalidCredentials;
027        
028         @WebResult(targetNamespace = NS)
029         public List<Role> authorize(
030                         @WebParam(name = "idIS", targetNamespace = NS) String idIS,
031                         @WebParam(name = "subject", targetNamespace = NS) Token subject) throws GeneralFailure, TokenExpired;
032         
033         @WebResult(targetNamespace = NS)
034         public List<Attribute> attributes(
035                        /* @WebParam(name = "attributeNames", targetNamespace = NS) List<String> attributeNames,*/
036                         @WebParam(name = "subject", targetNamespace = NS) Token subject) throws GeneralFailure, TokenExpired;
037
038         @WebResult(targetNamespace = NS)
039         public void audit(
040                         @WebParam(name = "idIS", targetNamespace = NS) String idIS,
041                         @WebParam(name = "userFunctions", targetNamespace = NS) List<AuditFunction> userFunctions,
042                         @WebParam(name = "subject", targetNamespace = NS) Token subject) throws GeneralFailure;
043         
044         @WebResult(targetNamespace = NS)
045         public void logout(
046                         @WebParam(name = "subject", targetNamespace = NS) Token subject) throws GeneralFailure;
047         
048}

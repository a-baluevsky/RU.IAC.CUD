098         */
099        public RequestSecurityTokenResponse issue(RequestSecurityToken request,
100                        Principal callerPrincipal) throws WSTrustException {
101
102                 
103
104                LOGGER.trace("Issuing token for principal " + callerPrincipal);
105
106                
107
108                // first try to obtain the security token provider using the applies-to
109                // contents.
110                AppliesTo appliesTo = request.getAppliesTo();
111                X509Certificate providerCertificate = null;
112                PublicKey providerPublicKey = null;
113                if (appliesTo != null) {
114                        String serviceName = WSTrustUtil.parseAppliesTo(appliesTo);
115
116                         
117
118                        if (serviceName != null) {
119                                String tokenTypeFromServiceName = configuration
120                                                .getTokenTypeForService(serviceName);
121
122                                 
123
124                                if (request.getTokenType() == null
125                                                && tokenTypeFromServiceName != null) {
126                                        request.setTokenType(URI.create(tokenTypeFromServiceName));
127                                }
128
129                                // закомментили
130                                // providerCertificate =
131                                // this.configuration.getServiceProviderCertificate(serviceName);
132
133                                // если придёт appliesTo, то
134                                // PicketLinkSTSConfiguration.getServiceProviderCertificate ищет
135                                // соответствия по Endpoint в picketlink-sts.xml
136                                // и если не найдет выдаёт ошибку
137                                // "Error obtaining public key for service"
138
139                                // для appliesTo надо
140                                /*
141                                 * <ServiceProviders> <ServiceProvider
142                                 * Endpoint="http://cudvm/CudServicesSTS/PicketLinkSTS"
143                                 * TokenType=
144                                 * "http://docs.oasis-open.org/wss/oasis-wss-saml-token-profile-1.1#SAMLV2.0"
145                                 * TruststoreAlias="cudvm_export"/> </ServiceProviders>
146                                 */
147
148                                 
149
150                                if (providerCertificate != null) {
151                                        providerPublicKey = providerCertificate.getPublicKey();
152                                }
153
154                                // provider =
155                                // this.configuration.getProviderForService(serviceName);
156                                /*
157                                 * if (provider != null) {
158                                 * request.setTokenType(URI.create(this.configuration
159                                 * .getTokenTypeForService(serviceName))); providerPublicKey =
160                                 * this.configuration.getServiceProviderPublicKey(serviceName);
161                                 * }
162                                 */
163                        }
164                }
165
166                 
167                // if applies-to is not available or if no provider was found for the
168                // service, use the token type.
169                /*
170                 * if (provider == null && request.getTokenType() != null) { provider =
171                 * this
172                 * .configuration.getProviderForTokenType(request.getTokenType().toString
173                 * ()); } else if (appliesTo == null && request.getTokenType() == null)
174                 * throw new WSTrustException(
175                 * "Either AppliesTo or TokenType must be present in a security token request"
176                 * );
177                 * 
178                 * if (provider != null) {
179                 */
180                // create the request context and delegate token generation to the
181                // provider.
182                WSTrustRequestContext requestContext = new WSTrustRequestContext(
183                                request, callerPrincipal);
184                requestContext.setTokenIssuer(this.configuration.getSTSName());
185                if (request.getLifetime() == null
186                                && this.configuration.getIssuedTokenTimeout() != 0) {
187                        // if no lifetime has been specified, use the configured timeout
188                        // value.
189                        LOGGER.stsTokenTimeoutNotSpecified();
190                        // request.setLifetime(WSTrustUtil.createDefaultLifetime(this.configuration.getIssuedTokenTimeout()));
191                        request.setLifetime(CUDWSTrustUtil
192                                        .createDefaultLifetime(this.configuration
193                                                        .getIssuedTokenTimeout()));
194
195                }
196                requestContext.setServiceProviderPublicKey(providerPublicKey);
197
198                // process the claims if needed.
199                if (request.getClaims() != null) {
200                        ClaimsType claims = request.getClaims();
201                        ClaimsProcessor processor = this.configuration
202                                        .getClaimsProcessor(claims.getDialect());
203                        // if there is a processor, process the claims and set the resulting
204                        // attributes in the context.
205                        if (processor != null) {
206                                requestContext.setClaimedAttributes(processor.processClaims(
207                                                claims, callerPrincipal));
208                        } else if (LOGGER.isDebugEnabled()) {
209                                LOGGER.debug("Claims have been specified in the request but no processor was found for dialect "
210                                                + claims.getDialect());
211                        }
212                }
213
214                 
215
216                // get the OnBehalfOf principal, if one has been specified.
217                if (request.getOnBehalfOf() != null) {
218                        Principal onBehalfOfPrincipal = WSTrustUtil
219                                        .getOnBehalfOfPrincipal(request.getOnBehalfOf());
220                        requestContext.setOnBehalfOfPrincipal(onBehalfOfPrincipal);
221
222                         
223                         
224                }
225
226                // get the key type and size from the request, setting default values if
227                // not specified.
228                URI keyType = request.getKeyType();
229                if (keyType == null) {
230                        LOGGER.debug("No key type could be found in the request. Using the default BEARER type.");
231                        keyType = URI.create(WSTrustConstants.KEY_TYPE_BEARER);
232                        request.setKeyType(keyType);
233                }
234                long keySize = request.getKeySize();
235                if (keySize == 0) {
236                        LOGGER.debug("No key size could be found in the request. Using the default size. ("
237                                        + KEY_SIZE + ")");
238                        keySize = KEY_SIZE;
239                        request.setKeySize(keySize);
240                }
241
242                // get the key wrap algorithm.
243                URI keyWrapAlgo = request.getKeyWrapAlgorithm();
244
245                // create proof-of-possession token and server entropy (if needed).
246                RequestedProofTokenType requestedProofToken = null;
247                EntropyType serverEntropy = null;
248
249                if (WSTrustConstants.KEY_TYPE_SYMMETRIC.equalsIgnoreCase(keyType
250                                .toString())) {
251                        // symmetric key case: if client entropy is found, compute a key. If
252                        // not, generate a new key.
253                        requestedProofToken = new RequestedProofTokenType();
254
255                        byte[] serverSecret = WSTrustUtil
256                                        .createRandomSecret((int) keySize / 8);
257                        BinarySecretType serverBinarySecret = new BinarySecretType();
258                        serverBinarySecret.setType(WSTrustConstants.BS_TYPE_NONCE);
259                        serverBinarySecret.setValue(Base64.encodeBytes(serverSecret)
260                                        .getBytes());
261
262                        byte[] clientSecret = null;
263                        EntropyType clientEntropy = request.getEntropy();
264                        if (clientEntropy != null) {
265                                clientSecret = Base64.decode(new String(WSTrustUtil
266                                                .getBinarySecret(clientEntropy)));
267                                serverEntropy = new EntropyType();
268                                serverEntropy.addAny(serverBinarySecret);
269                        }
270
271                        if (clientSecret != null && clientSecret.length != 0) {
272                                // client secret has been specified - combine it with the sts
273                                // secret.
274                                requestedProofToken.add(new ComputedKeyType(
275                                                WSTrustConstants.CK_PSHA1));
276                                byte[] combinedSecret = null;
277                                try {
278                                        combinedSecret = Base64.encodeBytes(
279                                                        WSTrustUtil.P_SHA1(clientSecret, serverSecret,
280                                                                        (int) keySize / 8)).getBytes();
281                                } catch (Exception e) {
282                                        throw LOGGER.wsTrustCombinedSecretKeyError(e);
283                                }
284                                requestContext.setProofTokenInfo(WSTrustUtil.createKeyInfo(
285                                                combinedSecret, providerPublicKey, keyWrapAlgo,
286                                                providerCertificate));
287                        } else {
288                                // client secret has not been specified - use the sts secret
289                                // only.
290                                requestedProofToken.add(serverBinarySecret);
291                                requestContext.setProofTokenInfo(WSTrustUtil.createKeyInfo(
292                                                serverBinarySecret.getValue(), providerPublicKey,
293                                                keyWrapAlgo, providerCertificate));
294                        }
295                } else if (WSTrustConstants.KEY_TYPE_PUBLIC.equalsIgnoreCase(keyType
296                                .toString())) {
297                        // try to locate the client cert in the keystore using the caller
298                        // principal as the alias.
299                        Certificate certificate = this.configuration
300                                        .getCertificate(callerPrincipal.getName());
301                        if (certificate != null) {
302                                requestContext.setProofTokenInfo(WSTrustUtil
303                                                .createKeyInfo(certificate));
304                        } else if (request.getUseKey() != null) {
305                                UseKeyType useKeyType = request.getUseKey();
306                                List<Object> theList = useKeyType.getAny();
307                                for (Object value : theList) {
308                                        if (value instanceof Element) {
309                                                Element keyElement = (Element) value;
310                                                String elementName = (keyElement).getLocalName();
311                                                // if the specified key is a X509 certificate we must
312                                                // insert it into a X509Data element.
313                                                if (elementName.equals("X509Certificate")) {
314                                                        X509DataType data = new X509DataType();
315                                                        data.add(value);
316                                                        value = data;
317                                                } else if (elementName.equals("KeyValue")) {
318                                                        KeyValueType keyValue = null;
319                                                        Element child = DocumentUtil
320                                                                        .getChildElement(
321                                                                                        keyElement,
322                                                                                        new QName(
323                                                                                                        WSTrustConstants.XMLDSig.RSA_KEYVALUE));
324                                                        if (child != null) {
325                                                                try {
326                                                                        keyValue = SignatureUtil
327                                                                                        .getRSAKeyValue(child);
328                                                                } catch (ParsingException e) {
329                                                                        throw LOGGER.stsError(e);
330                                                                }
331                                                        }
332                                                        if (keyValue == null && child == null) {
333                                                                child = DocumentUtil
334                                                                                .getChildElement(
335                                                                                                keyElement,
336                                                                                                new QName(
337                                                                                                                WSTrustConstants.XMLDSig.DSA_KEYVALUE));
338                                                                if (child != null) {
339                                                                        try {
340                                                                                keyValue = SignatureUtil
341                                                                                                .getDSAKeyValue(child);
342                                                                        } catch (ParsingException e) {
343                                                                                throw LOGGER.stsError(e);
344                                                                        }
345                                                                }
346                                                                value = keyValue;
347                                                        }
348                                                }
349                                                KeyInfoType keyInfo = new KeyInfoType();
350                                                keyInfo.addContent(value);
351                                                requestContext.setProofTokenInfo(keyInfo);
352                                        } else if (value instanceof KeyInfoType) {
353                                                requestContext.setProofTokenInfo((KeyInfoType) value);
354                                        } else {
355                                                throw new WSTrustException(LOGGER.unsupportedType(value
356                                                                .toString()));
357                                        }
358                                }
359                        } else {
360                                throw LOGGER.wsTrustClientPublicKeyError();
361                        }
362                }
363
364                // issue the security token using the constructed context.
365                try {
366                        if (request.getTokenType() != null) {
367                                requestContext.setTokenType(request.getTokenType().toString());
368                        }
369                        PicketLinkCoreSTS sts = PicketLinkCoreSTS.instance();
370                        sts.initialize(configuration);
371                        sts.issueToken(requestContext);
372                        // provider.issueToken(requestContext);
373                } catch (ProcessingException e) {
374                        throw LOGGER.stsError(e);
375                }
376
377                if (requestContext.getSecurityToken() == null) {
378                        throw new WSTrustException(
379                                        LOGGER.nullValueError("Token issued by STS"));
380                }
381
382                // construct the ws-trust security token response.
383                RequestedSecurityTokenType requestedSecurityToken = new RequestedSecurityTokenType();
384
385                SecurityToken contextSecurityToken = requestContext.getSecurityToken();
386                if (contextSecurityToken == null) {
387                        throw new WSTrustException(
388                                        LOGGER.nullValueError("Security Token from context"));
389                }
390
391                requestedSecurityToken.add(contextSecurityToken.getTokenValue());
392
393                RequestSecurityTokenResponse response = new RequestSecurityTokenResponse();
394                if (request.getContext() != null) {
395                        response.setContext(request.getContext());
396                }
397
398                response.setTokenType(request.getTokenType());
399                response.setLifetime(request.getLifetime());
400                response.setAppliesTo(appliesTo);
401                response.setKeySize(keySize);
402                response.setKeyType(keyType);
403                response.setRequestedSecurityToken(requestedSecurityToken);
404
405                if (requestedProofToken != null) {
406                        response.setRequestedProofToken(requestedProofToken);
407                }
408                if (serverEntropy != null) {
409                        response.setEntropy(serverEntropy);
410                }
411
412                // set the attached and unattached references.
413                if (requestContext.getAttachedReference() != null) {
414                        response.setRequestedAttachedReference(requestContext
415                                        .getAttachedReference());
416                }
417                if (requestContext.getUnattachedReference() != null) {
418                        response.setRequestedUnattachedReference(requestContext
419                                        .getUnattachedReference());
420                }
421
422                return response;
423        }

425        /*
426         * (non-Javadoc)
427         * 
428         * @see
429         * org.picketlink.identity.federation.core.wstrust.WSTrustRequestHandler
430         * #renew(
431         * org.picketlink.identity.federation.core.wstrust.wrappers.RequestSecurityToken
432         * , java.security.Principal)
433         */
434        public RequestSecurityTokenResponse renew(RequestSecurityToken request,
435                        Principal callerPrincipal) throws WSTrustException {
436                // first validate the provided token signature to make sure it has been
437                // issued by this STS and hasn't been
438                // tempered.
439
440                 
441
442                LOGGER.trace("Validating token for renew request "
443                                + request.getContext());
444
445                if (request.getRenewTargetElement() == null) {
446                        throw new WSTrustException(LOGGER.nullValueError("renew target"));
447                }
448
449                Node securityToken = request.getRenewTargetElement().getFirstChild();
450                if (securityToken == null) {
451                        throw new WSTrustException(LOGGER.nullValueError("security token"));
452                }
453
454                 
455
456                /*
457                 * try{ Transformer transformer =
458                 * TransformerFactory.newInstance().newTransformer();
459                 * transformer.setOutputProperty(OutputKeys.OMIT_XML_DECLARATION,
460                 * "yes"); // копирование содержимого XML-документа в поток
461                 * transformer.transform(new DOMSource(oldAssertionElement), new
462                 * StreamResult(System.out)); }catch(Exception e){ }
463                 */
464                /*
465                 * SecurityTokenProvider provider =
466                 * this.configuration.getProviderForTokenElementNS
467                 * (securityToken.getLocalName(), securityToken.getNamespaceURI()); if
468                 * (provider == null) throw new
469                 * WSTrustException("No SecurityTokenProvider configured for " +
470                 * securityToken.getNamespaceURI() + ":" +
471                 * securityToken.getLocalName());
472                 */
473
474                setupIDAttribute(securityToken);
475
476                if (this.configuration.signIssuedToken()
477                                && this.configuration.getSTSKeyPair() != null) {
478                        KeyPair keyPair = this.configuration.getSTSKeyPair();
479                        try {
480                                Document tokenDocument = DocumentUtil.createDocument();
481                                Node importedNode = tokenDocument.importNode(securityToken,
482                                                true);
483                                tokenDocument.appendChild(importedNode);
484                                // XMLSignatureUtil.propagateIDAttributeSetup(securityToken,
485                                // tokenDocument.getDocumentElement());
486                                // if (!XMLSignatureUtil.validate(tokenDocument,
487                                // keyPair.getPublic()))
488
489                                GOSTXMLSignatureUtil.propagateIDAttributeSetup(securityToken,
490                                                tokenDocument.getDocumentElement());
491                                if (!GOSTXMLSignatureUtil.validate(tokenDocument,
492                                                keyPair.getPublic())) {
493                                        throw new WSTrustException(LOGGER.signatureInvalidError(
494                                                        "Validation failure during renewal", null));
495                                }
496                        } catch (Exception e) {
497                                throw new WSTrustException(LOGGER.signatureInvalidError(
498                                                "Validation failure during renewal:", e));
499                        }
500                } else {
501                        LOGGER.stsSecurityTokenSignatureNotVerified();
502                }
503
504                // set default values where needed.
505                if (request.getLifetime() == null
506                                && this.configuration.getIssuedTokenTimeout() != 0) {
507                        // if no lifetime has been specified, use the configured timeout
508                        // value.
509                        LOGGER.stsTokenTimeoutNotSpecified();
510                        // request.setLifetime(WSTrustUtil.createDefaultLifetime(this.configuration.getIssuedTokenTimeout()));
511                        request.setLifetime(CUDWSTrustUtil
512                                        .createDefaultLifetime(this.configuration
513                                                        .getIssuedTokenTimeout()));
514
515                }
516
517                // create a context and dispatch to the proper security token provider
518                // for renewal.
519                WSTrustRequestContext context = new WSTrustRequestContext(request,
520                                callerPrincipal);
521                context.setTokenIssuer(this.configuration.getSTSName());
522                // if the renew request was made on behalf of another identity, get the
523                // principal of that identity.
524                if (request.getOnBehalfOf() != null) {
525                        Principal onBehalfOfPrincipal = WSTrustUtil
526                                        .getOnBehalfOfPrincipal(request.getOnBehalfOf());
527                        context.setOnBehalfOfPrincipal(onBehalfOfPrincipal);
528                }
529                try {
530                        if (securityToken != null) {
531                                String ns = securityToken.getNamespaceURI();
532
533                                context.setQName(new QName(ns, securityToken.getLocalName()));
534                        }
535                        PicketLinkCoreSTS sts = PicketLinkCoreSTS.instance();
536                        sts.initialize(configuration);
537                        sts.renewToken(context);
538                        // provider.renewToken(context);
539                } catch (ProcessingException e) {
540                        throw new WSTrustException(e.getMessage(), e);
541                }
542
543                // create the WS-Trust response with the renewed token.
544                RequestedSecurityTokenType requestedSecurityToken = new RequestedSecurityTokenType();
545                SecurityToken contextSecurityToken = context.getSecurityToken();
546                if (contextSecurityToken == null) {
547                        throw new WSTrustException(
548                                        LOGGER.nullValueError("Security Token from context"));
549                }
550                requestedSecurityToken.add(contextSecurityToken.getTokenValue());
551
552                RequestSecurityTokenResponse response = new RequestSecurityTokenResponse();
553                if (request.getContext() != null) {
554                        response.setContext(request.getContext());
555                }
556                response.setTokenType(request.getTokenType());
557                response.setLifetime(request.getLifetime());
558                response.setRequestedSecurityToken(requestedSecurityToken);
559                if (context.getAttachedReference() != null) {
560                        response.setRequestedAttachedReference(context
561                                        .getAttachedReference());
562                }
563                if (context.getUnattachedReference() != null) {
564                        response.setRequestedUnattachedReference(context
565                                        .getUnattachedReference());
566                }
567                return response;
568        }

570        /*
571         * (non-Javadoc)
572         * 
573         * @see
574         * org.picketlink.identity.federation.core.wstrust.WSTrustRequestHandler
575         * #validate(
576         * org.picketlink.identity.federation.core.wstrust.wrappers.RequestSecurityToken
577         * , java.security.Principal)
578         */
579        public RequestSecurityTokenResponse validate(RequestSecurityToken request,
580                        Principal callerPrincipal) throws WSTrustException {
581
582                LOGGER.trace("Started validation for request " + request.getContext());
583
584                if (request.getValidateTargetElement() == null) {
585                        throw new WSTrustException(
586                                        LOGGER.nullValueError("request does not have a validate target. Unable to validate token"));
587                }
588
589                if (request.getTokenType() == null) {
590                        request.setTokenType(URI.create(WSTrustConstants.STATUS_TYPE));
591                }
592
593                Node securityToken = request.getValidateTargetElement().getFirstChild();
594                if (securityToken == null) {
595                        throw new WSTrustException(
596                                        LOGGER.nullValueError("security token:Unable to validate token"));
597                }
598
599                setupIDAttribute(securityToken);
600
601                WSTrustRequestContext context = new WSTrustRequestContext(request,
602                                callerPrincipal);
603                // if the validate request was made on behalf of another identity, get
604                // the principal of that identity.
605                if (request.getOnBehalfOf() != null) {
606                        Principal onBehalfOfPrincipal = WSTrustUtil
607                                        .getOnBehalfOfPrincipal(request.getOnBehalfOf());
608                        context.setOnBehalfOfPrincipal(onBehalfOfPrincipal);
609                }
610                StatusType status = null;
611
612                // validate the security token digital signature.
613                if (this.configuration.signIssuedToken()
614                                && this.configuration.getSTSKeyPair() != null) {
615                        KeyPair keyPair = this.configuration.getSTSKeyPair();
616                        try {
617                                if (LOGGER.isTraceEnabled()) {
618                                        try {
619                                                LOGGER.trace("Going to validate signature for: "
620                                                                + DocumentUtil.getNodeAsString(securityToken));
621                                        } catch (Exception e) {
622                                        }
623                                }
624                                Document tokenDocument = DocumentUtil.createDocument();
625                                Node importedNode = tokenDocument.importNode(securityToken,
626                                                true);
627                                tokenDocument.appendChild(importedNode);
628                                // XMLSignatureUtil.propagateIDAttributeSetup(securityToken,
629                                // tokenDocument.getDocumentElement());
630                                // if (!XMLSignatureUtil.validate(tokenDocument,
631                                // keyPair.getPublic())) {
632
633                                GOSTXMLSignatureUtil.propagateIDAttributeSetup(securityToken,
634                                                tokenDocument.getDocumentElement());
635                                if (!GOSTXMLSignatureUtil.validate(tokenDocument,
636                                                keyPair.getPublic())) {
637
638                                        status = new StatusType();
639                                        status.setCode(WSTrustConstants.STATUS_CODE_INVALID);
640                                        status.setReason("Validation failure: digital signature is invalid");
641                                }
642                        } catch (Exception e) {
643                                status = new StatusType();
644                                status.setCode(WSTrustConstants.STATUS_CODE_INVALID);
645                                status.setReason("Validation failure: unable to verify digital signature: "
646                                                + e.getMessage());
647                        }
648                } else {
649                        LOGGER.stsSecurityTokenSignatureNotVerified();
650                }
651
652                // if the signature is valid, then let the provider perform any
653                // additional validation checks.
654                if (status == null) {
655                        LOGGER.trace("Delegating token validation to token provider. Token NS: "
656                                        + securityToken.getNamespaceURI()
657                                        + " ::LocalName: "
658                                        + securityToken.getLocalName());
659                        try {
660                                if (securityToken != null) {
661                                        context.setQName(new QName(securityToken.getNamespaceURI(),
662                                                        securityToken.getLocalName()));
663                                }
664                                PicketLinkCoreSTS sts = PicketLinkCoreSTS.instance();
665                                sts.initialize(configuration);
666                                sts.validateToken(context);
667                                // provider.validateToken(context);
668                        } catch (ProcessingException e) {
669                                throw LOGGER.stsError(e);
670                        }
671                        status = context.getStatus();
672                }
673
674                // construct and return the response.
675                RequestSecurityTokenResponse response = new RequestSecurityTokenResponse();
676                if (request.getContext() != null) {
677                        response.setContext(request.getContext());
678                }
679                response.setTokenType(request.getTokenType());
680                response.setStatus(status);
681
682                return response;
683        }

685        /*
686         * (non-Javadoc)
687         * 
688         * @see
689         * org.picketlink.identity.federation.core.wstrust.WSTrustRequestHandler
690         * #cancel(
691         * org.picketlink.identity.federation.core.wstrust.wrappers.RequestSecurityToken
692         * , java.security.Principal)
693         */
694        public RequestSecurityTokenResponse cancel(RequestSecurityToken request,
695                        Principal callerPrincipal) throws WSTrustException {
696                // check if request contains all required elements.
697                if (request.getCancelTargetElement() == null) {
698                        throw new WSTrustException(
699                                        LOGGER.nullValueError("request does not have a cancel target. Unable to cancel token"));
700                }
701
702                // obtain the token provider that will handle the request.
703                Node securityToken = request.getCancelTargetElement().getFirstChild();
704                if (securityToken == null) {
705                        throw new WSTrustException(
706                                        LOGGER.nullValueError("security token. Unable to cancel token"));
707                }
708
709                /*
710                 * SecurityTokenProvider provider =
711                 * this.configuration.getProviderForTokenElementNS
712                 * (securityToken.getLocalName(), securityToken.getNamespaceURI()); if
713                 * (provider == null) throw new
714                 * WSTrustException("No SecurityTokenProvider configured for " +
715                 * securityToken.getNamespaceURI() + ":" +
716                 * securityToken.getLocalName());
717                 */
718
719                // create a request context and dispatch to the provider.
720                WSTrustRequestContext context = new WSTrustRequestContext(request,
721                                callerPrincipal);
722                // if the cancel request was made on behalf of another identity, get the
723                // principal of that identity.
724                if (request.getOnBehalfOf() != null) {
725                        Principal onBehalfOfPrincipal = WSTrustUtil
726                                        .getOnBehalfOfPrincipal(request.getOnBehalfOf());
727                        context.setOnBehalfOfPrincipal(onBehalfOfPrincipal);
728                }
729                try {
730                        if (securityToken != null) {
731                                context.setQName(new QName(securityToken.getNamespaceURI(),
732                                                securityToken.getLocalName()));
733                        }
734                        PicketLinkCoreSTS sts = PicketLinkCoreSTS.instance();
735                        sts.initialize(configuration);
736                        sts.cancelToken(context);
737                        // provider.cancelToken(context);
738                } catch (ProcessingException e) {
739                        throw LOGGER.stsError(e);
740                }
741
742                // if no exception has been raised, the token has been successfully
743                // canceled.
744                RequestSecurityTokenResponse response = new RequestSecurityTokenResponse();
745                if (request.getContext() != null) {
746                        response.setContext(request.getContext());
747                }
748                response.setRequestedTokenCancelled(new RequestedTokenCancelledType());
749                return response;
750        }

923        /**

925         * element.
926         * 
927         * @param node
928         *            The node representing the SAML Assertion
929         * @return A reference to the correct ID

931        private static String setupIDAttribute(Node node) {
932                if (node instanceof Element) {
933                        Element assertion = (Element) node;
934                        if (assertion.getLocalName().equals("Assertion")) {
935                                if (assertion.getNamespaceURI().equals(
936                                                WSTrustConstants.SAML2_ASSERTION_NS)
937                                                && assertion.hasAttribute("ID")) {
938                                        assertion.setIdAttribute("ID", true);
939                                        return "#" + assertion.getAttribute("ID");
940                                } else if (assertion.getNamespaceURI().equals(
941                                                SAML11Constants.ASSERTION_11_NSURI)
942                                                && assertion.hasAttribute(SAML11Constants.ASSERTIONID)) {
943                                        assertion.setIdAttribute(SAML11Constants.ASSERTIONID, true);
944                                        return "#"
945                                                        + assertion
946                                                                        .getAttribute(SAML11Constants.ASSERTIONID);
947                                }
948                        }
949                }
950                return "";
951        }

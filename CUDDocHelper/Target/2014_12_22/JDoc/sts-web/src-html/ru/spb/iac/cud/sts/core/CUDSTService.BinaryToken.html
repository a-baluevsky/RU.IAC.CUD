095        public SOAPMessage invoke(SOAPMessage request) {
096                String valueType = null;
097                Node binaryToken = null;
098                boolean soap12 = false;
099
100                MessageContext mess_context = context.getMessageContext();
101
102                HttpServletRequest http_request = (HttpServletRequest) mess_context
103                                .get(MessageContext.SERVLET_REQUEST);
104
105                HttpSession http_session = http_request.getSession();
106
107                String user_principal_name = (String) http_session
108                                .getAttribute("user_principal");
109                String system_principal_name = (String) http_session
110                                .getAttribute("system_principal");
111        String auth_type = (String)  http_session
112                                .getAttribute("cud_auth_type");
113                        
114                LOGGERSLF4J.debug("invoke:01:" + user_principal_name);
115                LOGGERSLF4J.debug("invoke:02:" + system_principal_name);
116
117                cud_principal = new CudPrincipal();
118                cud_principal.setSystemName(system_principal_name);
119                
120
121                if (user_principal_name != null) {
122                        cud_principal.setUserName(user_principal_name);
123                        cud_principal.setAuthType(auth_type);
124                }
125
126                // Check headers
127                try {
128
129                        soap12 = SOAPUtil.isSOAP12(request);
130                        SOAPHeader soapHeader = request.getSOAPHeader();
131                        binaryToken = getBinaryToken(soapHeader);
132                        if (binaryToken != null) {
133                                NamedNodeMap namedNodeMap = binaryToken.getAttributes();
134                                int length = namedNodeMap != null ? namedNodeMap.getLength()
135                                                : 0;
136                                for (int i = 0; i < length; i++) {
137                                        Node nodeValueType = namedNodeMap
138                                                        .getNamedItem(WSSE.VALUE_TYPE);
139                                        if (nodeValueType != null) {
140                                                valueType = nodeValueType.getNodeValue();
141                                                break;
142                                        }
143                                }
144                        }
145                } catch (SOAPException e) {
146                        LOGGERSLF4J.error("invoke:04_error:", e);
147                        throw LOGGER.stsWSError(e);
148                }
149
150                Node payLoad;
151                BaseRequestSecurityToken baseRequest;
152                try {
153                        payLoad = SOAPUtil.getSOAPData(request);
154
155                        WSTrustParser parser = new WSTrustParser();
156
157                        baseRequest = (BaseRequestSecurityToken) parser.parse(DocumentUtil
158                                        .getNodeAsStream(payLoad));
159                } catch (Exception e) {
160                        LOGGERSLF4J.error("invoke:05_error:", e);
161                        throw LOGGER.stsWSError(e);
162                }
163
164                if (baseRequest instanceof RequestSecurityToken) {
165
166                        RequestSecurityToken req = (RequestSecurityToken) baseRequest;
167                        try {
168                                req.setRSTDocument((Document) payLoad);
169                        } catch (Exception e) {
170                                throw new RuntimeException(e);
171                        }
172
173                        if (binaryToken != null) {
174                                req.setBinaryToken(binaryToken);
175                        }
176
177                        if (valueType != null) {
178                                req.setBinaryValueType(URI.create(valueType));
179                        }
180                        Source theResponse = this.handleTokenRequest(req);
181
182                        return convert(theResponse, soap12);
183                } else if (baseRequest instanceof RequestSecurityTokenCollection) {
184
185                        return convert(
186                                        this.handleTokenRequestCollection((RequestSecurityTokenCollection) baseRequest),
187                                        soap12);
188                } else {
189                        throw LOGGER.stsWSInvalidTokenRequestError();
190                }
191        }
192
193        private SOAPMessage convert(Source theResponse, boolean wantSOAP12) {
194                try {
195                        SOAPMessage response = null;
196
197                        if (wantSOAP12) {
198                                response = SOAPUtil.createSOAP12();
199                        } else {
200                                response = SOAPUtil.create();
201                        }
202                        Document theResponseDoc = (Document) DocumentUtil
203                                        .getNodeFromSource(theResponse);
204                        response.getSOAPBody().addDocument(theResponseDoc);
205
206                
207                        return response;
208                } catch (Exception e) {
209                        throw LOGGER.stsWSError(e);
210                }
211        }
212
213        private Node getBinaryToken(SOAPHeader soapHeader) {
214                if (soapHeader != null) {
215                        NodeList children = soapHeader.getChildNodes();
216                        int length = children != null ? children.getLength() : 0;
217                        for (int i = 0; i < length; i++) {
218                                Node child = children.item(i);
219                                if (child.getNodeName().contains(WSSE.BINARY_SECURITY_TOKEN)) {
220                                        return child;
221                                }
222                        }
223                }
224                return null;
225        }

227        /**
228         * <p>
229         * Process a security token request.
230         * </p>
231         * 
232         * @param request
233         *            a {@code RequestSecurityToken} instance that contains the
234         *            request information.
235         * @return a {@code Source} instance representing the marshalled response.
236         * @throws WebServiceException
237         *             Any exception encountered in handling token
238         */
239        protected Source handleTokenRequest(RequestSecurityToken request) {
240                if (context == null) {
241                        throw new IllegalStateException(ErrorCodes.NULL_VALUE
242                                        + "WebServiceContext");
243                }
244                if (this.config == null) {
245                        try {
246                                LOGGER.debug("Loading STS configuration");
247                                this.config = this.getConfiguration();
248                        } catch (ConfigurationException e) {
249                                throw LOGGER.stsWSConfigurationError(e);
250                        }
251                }
252
253                WSTrustRequestHandler handler = this.config.getRequestHandler();
254                if (handler == null) {
255                        throw LOGGER.nullValueError("WSTrustRequestHandler");
256                }
257
258                String requestType = request.getRequestType().toString();
259
260                LOGGER.trace("STS received request of type " + requestType);
261
262                try {
263                        if (requestType.equals(WSTrustConstants.ISSUE_REQUEST)) {
264
265                                Source source = this.marshallResponse(handler.issue(request,
266                                                cud_principal));
267
268                                Document doc = handler.postProcess(
269                                                (Document) ((DOMSource) source).getNode(), request);
270
271                                 
272
273                                return new DOMSource(doc);
274                        } else if (requestType.equals(WSTrustConstants.RENEW_REQUEST)) {
275                                Source source = this.marshallResponse(handler.renew(request,
276                                                cud_principal));
277
278                                Document document = handler.postProcess(
279                                                (Document) ((DOMSource) source).getNode(), request);
280                                return new DOMSource(document);
281                        } else if (requestType.equals(WSTrustConstants.CANCEL_REQUEST)) {
282                                return this.marshallResponse(handler.cancel(request,
283                                                this.context.getUserPrincipal()));
284                        } else if (requestType.equals(WSTrustConstants.VALIDATE_REQUEST)) {
285                                return this.marshallResponse(handler.validate(request,
286                                                this.context.getUserPrincipal()));
287                        } else {
288                                throw LOGGER.stsWSInvalidRequestTypeError(requestType);
289                        }
290                } catch (WSTrustException we) {
291                        throw LOGGER.stsWSHandlingTokenRequestError(we);
292                }
293        }

310        /**
311         * <p>
312         * Marshalls the specified {@code RequestSecurityTokenResponse} into a
313         * {@code Source} instance.
314         * </p>
315         * 
316         * @param response
317         *            the {@code RequestSecurityTokenResponse} to be marshalled.
318         * @return the resulting {@code Source} instance.
319         */
320        protected Source marshallResponse(RequestSecurityTokenResponse response) {
321                // add the single response to a RequestSecurityTokenResponse collection,
322                // as per the specification.
323                RequestSecurityTokenResponseCollection responseCollection = new RequestSecurityTokenResponseCollection();
324                responseCollection.addRequestSecurityTokenResponse(response);
325
326                try {
327                        DOMResult result = new DOMResult(DocumentUtil.createDocument());
328                        WSTrustResponseWriter writer = new WSTrustResponseWriter(result);
329                        writer.write(responseCollection);
330                        return new DOMSource(result.getNode());
331                } catch (Exception e) {
332                        throw LOGGER.stsWSResponseWritingError(e);
333                }
334        }

336        /**
337         * <p>
338         * Obtains the STS configuration options.
339         * </p>
340         * 
341         * @return an instance of {@code STSConfiguration} containing the STS
342         *         configuration properties.
343         */
344        protected STSConfiguration getConfiguration() throws ConfigurationException {
345                URL configurationFileURL = null;
346
347                try {
348                        // check the user home for a configuration file generated by the
349                        // picketlink console.
350                        String configurationFilePath = System.getProperty("user.home")
351                                        + SEPARATOR + STS_CONFIG_DIR + STS_CONFIG_FILE;
352                        File configurationFile = new File(configurationFilePath);
353                        if (configurationFile.exists()) {
354                                configurationFileURL = configurationFile.toURI().toURL();
355                        } else {
356                                configurationFileURL = SecurityActions.loadResource(getClass(),
357                                                PICKETLINK_CONFIG_FILE);
358
359                                // fallback to the old configuration
360                                if (configurationFileURL == null) {
361                                        configurationFileURL = SecurityActions.loadResource(
362                                                        getClass(), STS_CONFIG_FILE);
363                                }
364                        }
365
366                        // if no configuration file was found, log a warn message and use
367                        // default configuration values.
368                        if (configurationFileURL == null) {
369                                LOGGER.stsUsingDefaultConfiguration("");
370                                return new PicketLinkSTSConfiguration();
371                        }
372
373                        InputStream stream = configurationFileURL.openStream();
374                        STSType stsConfig = (STSType) new STSConfigParser().parse(stream);
375                        STSConfiguration configuration = new PicketLinkSTSConfiguration(
376                                        stsConfig);
377                        if (LOGGER.isInfoEnabled()) {
378                                LOGGER.stsConfigurationFileLoaded(STS_CONFIG_FILE);
379                        }
380                        return configuration;
381                } catch (Exception e) {
382                        throw LOGGER.stsConfigurationFileParsingError(e);
383                }
384        }

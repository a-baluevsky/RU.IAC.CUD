001package ru.spb.iac.cud.infosweb.web.init;
002
003import javax.servlet.ServletConfig;
004import javax.servlet.ServletContext;
005import javax.servlet.ServletException;
006import javax.servlet.http.HttpServlet;
007
008import mypackage.Configuration;
009
010import org.slf4j.Logger;
011import org.slf4j.LoggerFactory;
012
013
014
015/**
016 * Servlet implementation class StartupServlet
017 */
018 public class StartupServlet extends HttpServlet {
019
020        private static final Logger LOGGER = LoggerFactory.getLogger(StartupServlet.class);
021
022        private static final long serialVersionUID = 1L;
023
024        /**
025         * @see HttpServlet#HttpServlet()
026         */
027        public StartupServlet() {
028                super();
029                // TODO Auto-generated constructor stub
030        }
031
032        /**
033         * @see Servlet#init(ServletConfig)
034         */
035        public void init(ServletConfig config) throws ServletException {
036                
037                
038                try {
039                        ServletContext context = config.getServletContext();
040                
041                        String signRequired = context.getInitParameter("SIGN_REQUIRED"); 
042                        
043                        String samlRequestLogin = context.getInitParameter("SAML_REQUEST_LOGIN"); 
044                        
045                        String samlRequestLogout = context.getInitParameter("SAML_REQUEST_LOGOUT"); 
046                        
047            String samlAssertion = context.getInitParameter("SAML_ASSERTION"); 
048                        
049                        String storePath = context.getInitParameter("STORE_PATH"); 
050                        
051                        String archiveAuditFunc = context.getInitParameter("ARHIVE_AUDIT_FUNC"); 
052                        
053                        String archiveAuditSys = context.getInitParameter("ARHIVE_AUDIT_SYS"); 
054                        
055                        String archiveToken = context.getInitParameter("ARHIVE_TOKEN");
056                        
057                        String uccert = context.getInitParameter("UCCERT");
058                        
059                        String auditService = context.getInitParameter("AUDIT_SERVICE");
060                        
061                        String stsOboService = context.getInitParameter("STS_OBO_SERVICE");
062                        
063                        String stsService = context.getInitParameter("STS_SERVICE");
064                        
065                        String ucCertReestr = context.getInitParameter("UC_CERT_REESTR");
066                        
067                        String classifService = context.getInitParameter("CLASSIF_SERVICE");
068                        
069                        String classifLoadPatch = context.getInitParameter("CLASSIF_LOAD_PATH");
070                        
071                        String classifLoadTmp = context.getInitParameter("CLASSIF_LOAD_TMP");
072                        
073                        String jasperServer = context.getInitParameter("JASPER_SERVER");
074                        
075                        String jasperLogin = context.getInitParameter("JASPER_LOGIN");
076                        
077                        String jasperPassword = context.getInitParameter("JASPER_PASSWORD");
078                        
079                        Configuration.setSignRequired(
080                                                !"FALSE".equals(signRequired)&&!"false".equals(signRequired));
081                
082                
083                        Configuration.setSamlRequestLogin(samlRequestLogin);
084                        Configuration.setSamlRequestLogout(samlRequestLogout);
085                        Configuration.setSamlAssertion(samlAssertion);
086                        Configuration.setStorePath(storePath);
087                        Configuration.setArchiveAuditFunc(archiveAuditFunc);
088                        Configuration.setArchiveAuditSys(archiveAuditSys);
089                        Configuration.setArchiveToken(archiveToken);
090                        Configuration.setUccert(uccert);
091                        Configuration.setAuditService(auditService);
092                        Configuration.setStsOboService(stsOboService);
093                        Configuration.setStsService(stsService);
094                        Configuration.setUcCertReestr(ucCertReestr);
095                        Configuration.setClassifService(classifService);
096                        Configuration.setClassifLoadPatch(classifLoadPatch);
097                        Configuration.setClassifLoadTmp(classifLoadTmp);
098                        Configuration.setJasperServer(jasperServer);
099                        Configuration.setJasperLogin(jasperLogin);
100                        Configuration.setJasperPassword(jasperPassword);
101                        
102                } catch (Exception e) {
103                        LOGGER.error("init:error:", e);
104                }
105        }
106
107}

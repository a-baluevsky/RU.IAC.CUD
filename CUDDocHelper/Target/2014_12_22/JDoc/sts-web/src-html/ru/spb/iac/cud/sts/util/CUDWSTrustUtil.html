001package ru.spb.iac.cud.sts.util;
002
003import org.picketlink.identity.federation.core.wstrust.WSTrustUtil;
004import org.picketlink.identity.federation.core.wstrust.wrappers.Lifetime;
005import java.util.GregorianCalendar;
006
007 public class CUDWSTrustUtil extends WSTrustUtil {
008
009        /**
010         * <p>
011         * Creates a {@code Lifetime} instance that specifies a range of time that
012         * starts at the current GMT time and has the specified duration in
013         * milliseconds.
014         * </p>
015         * 
016         * @param tokenTimeout
017         *            the token timeout value (in milliseconds).
018         * @return the constructed {@code Lifetime} instance.
019         */
020        public static Lifetime createDefaultLifetime(long tokenTimeout) {
021                GregorianCalendar created = new GregorianCalendar();
022                GregorianCalendar expires = new GregorianCalendar();
023                expires.setTimeInMillis(created.getTimeInMillis() + tokenTimeout);
024
025                
026                return new CUDLifetime(created, expires);
027        }
028
029}

101        /**
102         * Sign an RequestType at the root
103         * 
104         * @param request
105         * @param keypair
106         *            Key Pair
107         * @param digestMethod
108         *            (Example: DigestMethod.SHA1)
109         * @param signatureMethod
110         *            (Example: SignatureMethod.DSA_SHA1)
111         * @return
112         * @throws ParserConfigurationException
113         * @throws IOException
114         * @throws SAXException
115         * @throws XMLSignatureException
116         * @throws MarshalException
117         * @throws GeneralSecurityException
118         */
119        public Document sign(RequestAbstractType request, KeyPair keypair)
120                        throws SAXException, IOException, ParserConfigurationException,
121                        GeneralSecurityException, MarshalException, XMLSignatureException {
122                SAML2Request saml2Request = new SAML2Request();
123                Document doc = saml2Request.convert(request);
124                doc.normalize();
125
126                Node theSibling = getNextSiblingOfIssuer(doc);
127                if (theSibling != null) {
128                        this.sibling = theSibling;
129                }
130
131                return sign(doc, request.getID(), keypair);
132        }

165        /**
166         * Sign an Document at the root
167         * 
168         * @param response
169         * @param keyPair
170         *            Key Pair
171         * @param digestMethod
172         *            (Example: DigestMethod.SHA1)
173         * @param signatureMethod
174         *            (Example: SignatureMethod.DSA_SHA1)
175         * @return
176         * @throws ParserConfigurationException
177         * @throws XMLSignatureException
178         * @throws MarshalException
179         * @throws GeneralSecurityException
180         */
181        public Document sign(Document doc, String referenceID, KeyPair keyPair)
182                        throws ParserConfigurationException, GeneralSecurityException,
183                        MarshalException, XMLSignatureException {
184                String referenceURI = "#" + referenceID;
185
186                configureIdAttribute(doc);
187
188                if (sibling != null) {
189                        SignatureUtilTransferObject dto = new SignatureUtilTransferObject();
190                        dto.setDocumentToBeSigned(doc);
191                        dto.setKeyPair(keyPair);
192                        dto.setDigestMethod(digestMethod);
193                        dto.setSignatureMethod(signatureMethod);
194                        dto.setReferenceURI(referenceURI);
195                        dto.setNextSibling(sibling);
196
197                        if (x509Certificate != null) {
198                                dto.setX509Certificate(x509Certificate);
199                        }
200
201                        
202                        return GOSTXMLSignatureUtil.sign(dto);
203                }
204                return GOSTXMLSignatureUtil.sign(doc, keyPair, digestMethod,
205                                signatureMethod, referenceURI);
206
207        }

266        /**
267         * Sign a SAML Document
268         * 
269         * @param samlDocument
270         * @param keypair
271         * @throws ProcessingException
272         */
273        public void signSAMLDocument(Document samlDocument, KeyPair keypair)
274                        throws ProcessingException {
275                // Get the ID from the root
276                String id = samlDocument.getDocumentElement().getAttribute(
277                                ID_ATTRIBUTE_NAME);
278                try {
279                        sign(samlDocument, id, keypair);
280                } catch (Exception e) {
281                        throw new ProcessingException(LOGGER.signatureError(e));
282                }
283        }

285        /**
286         * Validate the SAML2 Document
287         * 
288         * @param signedDocument
289         * @param publicKey
290         * @return
291         * @throws ProcessingException
292         */
293        public boolean validate(Document signedDocument, PublicKey publicKey)
294                        throws ProcessingException {
295                try {
296                        configureIdAttribute(signedDocument);
297                        return GOSTXMLSignatureUtil.validate(signedDocument, publicKey);
298
299                } catch (MarshalException me) {
300                        throw new ProcessingException(LOGGER.signatureError(me));
301                } catch (XMLSignatureException xse) {
302                        throw new ProcessingException(LOGGER.signatureError(xse));
303                }
304        }

306        /**
307         * Given a {@link Document}, find the {@link Node} which is the sibling of
308         * the Issuer element
309         * 
310         * @param doc
311         * @return
312         */
313        public Node getNextSiblingOfIssuer(Document doc) {
314                // Find the sibling of Issuer
315                NodeList nl = doc.getElementsByTagNameNS(
316                                JBossSAMLURIConstants.ASSERTION_NSURI.get(),
317                                JBossSAMLConstants.ISSUER.get());
318                if (nl.getLength() > 0) {
319                        Node issuer = nl.item(0);
320
321                        return issuer.getNextSibling();
322                }
323                return null;
324        }

326        /**
327         * <p>
328         * Sets the IDness of the ID attribute. Santuario 1.5.1 does not assumes
329         * IDness based on attribute names anymore. This method should be called
330         * before signing/validating a saml document.
331         * </p>
332         * 
333         * @param document
334         *            SAML document to have its ID attribute configured.
335         */
336        private void configureIdAttribute(Document document) {
337                // Estabilish the IDness of the ID attribute.
338                document.getDocumentElement().setIdAttribute(ID_ATTRIBUTE_NAME, true);
339
340                NodeList nodes = document.getElementsByTagNameNS(
341                                JBossSAMLURIConstants.ASSERTION_NSURI.get(),
342                                JBossSAMLConstants.ASSERTION.get());
343
344                for (int i = 0; i < nodes.getLength(); i++) {
345                        Node n = nodes.item(i);
346                        if (n instanceof Element) {
347                                ((Element) n).setIdAttribute(ID_ATTRIBUTE_NAME, true);
348                        }
349                }
350        }

192        /**
193         * Sign a node in a document
194         * 
195         * @param doc
196         * @param nodeToBeSigned
197         * @param keyPair
198         * @param publicKey
199         * @param digestMethod
200         * @param signatureMethod
201         * @param referenceURI
202         * @return
203         * @throws ParserConfigurationException
204         * @throws XMLSignatureException
205         * @throws MarshalException
206         * @throws GeneralSecurityException
207         */
208        public static Document sign(Document doc, Node nodeToBeSigned,
209                        KeyPair keyPair, String digestMethod, String signatureMethod,
210                        String referenceURI) throws ParserConfigurationException,
211                        GeneralSecurityException, MarshalException, XMLSignatureException {
212                if (nodeToBeSigned == null) {
213                        throw LOGGER.nullArgumentError("Node to be signed");
214                }
215
216                if (LOGGER.isTraceEnabled()) {
217                        LOGGER.trace("Document to be signed=" + DocumentUtil.asString(doc));
218                }
219
220                Node parentNodeSign = nodeToBeSigned.getParentNode();
221
222                // Let us create a new Document
223                Document newDoc = DocumentUtil.createDocument();
224                // Import the node
225                Node signingNodeSign = newDoc.importNode(nodeToBeSigned, true);
226                newDoc.appendChild(signingNodeSign);
227
228                if (!referenceURI.isEmpty()) {
229                        propagateIDAttributeSetup(nodeToBeSigned,
230                                        newDoc.getDocumentElement());
231                }
232                newDoc = sign(newDoc, keyPair, digestMethod, signatureMethod,
233                                referenceURI);
234
235                // if the signed element is a SAMLv2.0 assertion we need to move the
236                // signature element to the position
237                // specified in the schema (before the assertion subject element).
238                if ("Assertion".equals(nodeToBeSigned.getLocalName())
239                                && WSTrustConstants.SAML2_ASSERTION_NS.equals(nodeToBeSigned
240                                                .getNamespaceURI())) {
241                        Node signatureNode = DocumentUtil.getElement(newDoc, new QName(
242                                        WSTrustConstants.DSIG_NS, "Signature"));
243                        Node subjectNode = DocumentUtil.getElement(newDoc, new QName(
244                                        WSTrustConstants.SAML2_ASSERTION_NS, "Subject"));
245                        if (signatureNode != null && subjectNode != null) {
246                                newDoc.getDocumentElement().removeChild(signatureNode);
247                                newDoc.getDocumentElement().insertBefore(signatureNode,
248                                                subjectNode);
249                        }
250                }
251
252                // Now let us import this signed doc into the original document we got
253                // in the method call
254                Node signedNode = doc.importNode(newDoc.getFirstChild(), true);
255
256                if (!referenceURI.isEmpty()) {
257                        propagateIDAttributeSetup(newDoc.getDocumentElement(),
258                                        (Element) signedNode);
259                }
260
261                parentNodeSign.replaceChild(signedNode, nodeToBeSigned);
262        
263
264                return doc;
265        }

403        /**
404         * Setup the ID attribute into <code>destElement</code> depending on the
405         * <code>isId</code> flag of an attribute of <code>sourceNode</code>.
406         * 
407         * @param sourceNode
408         * @param destDocElement
409         */
410        public static void propagateIDAttributeSetup(Node sourceNode,
411                        Element destElement) {
412                NamedNodeMap nnm = sourceNode.getAttributes();
413                for (int i = 0; i < nnm.getLength(); i++) {
414                        Attr attr = (Attr) nnm.item(i);
415                        if (attr.isId()) {
416                                destElement.setIdAttribute(attr.getName(), true);
417                                break;
418                        }
419                }
420        }

516        /**
517         * Validate a signed document with the given public key
518         * 
519         * @param signedDoc
520         * @param publicKey
521         * @return
522         * @throws MarshalException
523         * @throws XMLSignatureException
524         */
525        @SuppressWarnings("unchecked")
526        public static boolean validate(Document signedDoc, Key publicKey)
527                        throws MarshalException, XMLSignatureException {
528                if (signedDoc == null) {
529                        throw LOGGER.nullArgumentError("Signed Document");
530                }
531
532                 
533
534                propagateIDAttributeSetup(signedDoc.getDocumentElement(),
535                                signedDoc.getDocumentElement());
536
537                NodeList nl = signedDoc.getElementsByTagNameNS(XMLSignature.XMLNS,
538                                "Signature");
539                if (nl == null || nl.getLength() == 0) {
540                        throw LOGGER.nullValueError("Cannot find Signature element");
541                }
542                if (publicKey == null) {
543                        throw LOGGER.nullValueError("Public Key");
544                }
545
546                //  GostKeyHandler,  GOSTKeyValue    
547                // "SENDER_PUBLIC_KEY"
548                //       publicKey.
549
550                DOMValidateContext valContext = new DOMValidateContext(publicKey,
551                                nl.item(0));
552                XMLSignature signature = fac.unmarshalXMLSignature(valContext);
553
554                boolean coreValidity = signature.validate(valContext);
555
556                 
557
558                if (LOGGER.isTraceEnabled() && !coreValidity) {
559                        boolean sv = signature.getSignatureValue().validate(valContext);
560                        LOGGER.trace("Signature validation status: " + sv);
561
562                        List<Reference> references = signature.getSignedInfo()
563                                        .getReferences();
564                        for (Reference ref : references) {
565                                LOGGER.trace("[Ref id=" + ref.getId() + ":uri=" + ref.getURI()
566                                                + "]validity status:" + ref.validate(valContext));
567                        }
568                }
569                return coreValidity;
570        }

606        /**
607         * Given the X509Certificate in the keyinfo element, get a
608         * {@link X509Certificate}
609         * 
610         * @param certificateString
611         * @return
612         * @throws ProcessingException
613         */
614        public static X509Certificate getX509CertificateFromKeyInfoString(
615                        String certificateString) throws ProcessingException {
616                X509Certificate cert = null;
617                StringBuilder builder = new StringBuilder();
618                builder.append("-----BEGIN CERTIFICATE-----\n")
619                                .append(certificateString)
620                                .append("\n-----END CERTIFICATE-----");
621
622                String derFormattedString = builder.toString();
623
624                try {
625                        CertificateFactory cf = CertificateFactory.getInstance("X.509");
626                        ByteArrayInputStream bais = new ByteArrayInputStream(
627                                        derFormattedString.getBytes());
628
629                        while (bais.available() > 0) {
630                                cert = (X509Certificate) cf.generateCertificate(bais);
631                        }
632                } catch (java.security.cert.CertificateException e) {
633                        throw LOGGER.processingError(e);
634                }
635                return cert;
636        }

639        private static void signImpl(DOMSignContext dsc, String digestMethod,
640                        String signatureMethod, String referenceURI, PublicKey publicKey,
641                        X509Certificate x509Certificate) throws GeneralSecurityException,
642                        MarshalException, XMLSignatureException {
643
644                 
645
646                dsc.setDefaultNamespacePrefix("dsig");
647
648                DigestMethod digestMethodObj = fac.newDigestMethod(digestMethod, null);
649                Transform transform1 = fac.newTransform(Transform.ENVELOPED,
650                                (TransformParameterSpec) null);
651                Transform transform2 = fac.newTransform(
652                                "http://www.w3.org/2001/10/xml-exc-c14n#",
653                                (TransformParameterSpec) null);
654
655                List<Transform> transformList = new ArrayList<Transform>();
656                transformList.add(transform1);
657                transformList.add(transform2);
658
659                Reference ref = fac.newReference(referenceURI, digestMethodObj,
660                                transformList, null, null);
661
662                CanonicalizationMethod canonicalizationMethod = fac
663                                .newCanonicalizationMethod(canonicalizationMethodType,
664                                                (C14NMethodParameterSpec) null);
665
666                List<Reference> referenceList = Collections.singletonList(ref);
667                SignatureMethod signatureMethodObj = fac.newSignatureMethod(
668                                signatureMethod, null);
669                SignedInfo si = fac.newSignedInfo(canonicalizationMethod,
670                                signatureMethodObj, referenceList);
671
672                KeyInfo ki = null;
673                if (includeKeyInfoInSignature) {
674                        ki = createKeyInfo(publicKey, x509Certificate);
675                }
676                XMLSignature signature = fac.newXMLSignature(si, ki);
677
678                signature.sign(dsc);
679        }

681        private static KeyInfo createKeyInfo(PublicKey publicKey,
682                        X509Certificate x509Certificate) throws KeyException {
683                KeyInfoFactory keyInfoFactory = fac.getKeyInfoFactory();
684                KeyInfo keyInfo = null;
685                KeyValue keyValue = null;
686                // Just with public key
687                if (publicKey != null) {
688                        keyValue = keyInfoFactory.newKeyValue(publicKey);
689                        keyInfo = keyInfoFactory.newKeyInfo(Collections
690                                        .singletonList(keyValue));
691                }
692                if (x509Certificate != null) {
693                        List x509list = new ArrayList();
694
695                        x509list.add(x509Certificate);
696                        X509Data x509Data = keyInfoFactory.newX509Data(x509list);
697                        List items = new ArrayList();
698
699                        items.add(x509Data);
700                        if (keyValue != null) {
701                                items.add(keyValue);
702                        }
703                        keyInfo = keyInfoFactory.newKeyInfo(items);
704                }
705                return keyInfo;
706        }

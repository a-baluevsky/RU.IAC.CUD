001package ru.spb.iac.cud.idp.core;
002
003import java.security.Principal;
004import java.util.ArrayList;
005import java.util.List;
006
007import org.slf4j.Logger;
008import org.slf4j.LoggerFactory;
009
010import ru.spb.iac.cud.context.ContextIDPAccessManager;
011import ru.spb.iac.cud.idp.items.CUDRoleGenerator;
012
013 public class RolesManager implements CUDRoleGenerator {
014
015        private String codeSystem;
016
017        private static final Logger LOGGER = LoggerFactory.getLogger(RolesManager.class);
018
019        public List<String> generateRoles(Principal principal) {
020
021                List<String> roles = new ArrayList<String>();
022                try {
023
024                         
025                         
026
027                        String domain = this.codeSystem;
028
029                        roles = (new ContextIDPAccessManager()).rolesCodes(
030                                        principal.getName(), domain);
031
032                } catch (Exception e) {
033                        LOGGER.error("generateRoles:error:", e);
034                }
035                return roles;
036        }
037
038        public List<String> generateResources(Principal principal) {
039
040                List<String> resources = new ArrayList<String>();
041                try {
042
043                        LOGGER.debug("generateResources:01:" + principal);
044                        LOGGER.debug("generateResources:02:" + this.codeSystem);
045
046                        String domain = this.codeSystem;
047
048                        resources = (new ContextIDPAccessManager()).resources(
049                                        principal.getName(), domain);
050
051                } catch (Exception e) {
052                        LOGGER.debug("generateResources:error:", e);
053                }
054                return resources;
055        }
056
057        public void setSystemCode(String codeSystem) {
058                this.codeSystem = codeSystem;
059
060        }
061}

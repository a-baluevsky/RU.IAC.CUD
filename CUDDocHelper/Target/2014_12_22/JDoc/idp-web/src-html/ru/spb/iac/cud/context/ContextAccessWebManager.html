001package ru.spb.iac.cud.context;
002
003import javax.naming.Context;
004import javax.naming.InitialContext;
005import org.slf4j.Logger;
006import org.slf4j.LoggerFactory;
007
008import ru.spb.iac.cud.core.AccessManagerLocal;
009import ru.spb.iac.cud.exceptions.GeneralFailure;
010import ru.spb.iac.cud.exceptions.InvalidCredentials;
011import ru.spb.iac.cud.exceptions.RevokedCertificate;
012import ru.spb.iac.cud.items.AuthMode;
013
014 public class ContextAccessWebManager {
015
016        static Context ctx;
017        AccessManagerLocal aml = null;
018
019        private static final Logger LOGGER = LoggerFactory
020                        .getLogger(ContextAccessWebManager.class);
021
022        public static void initContext(String jboss_jndi_port) {
023                try {
024                        LOGGER.debug("initContext:jboss_jndi_port:" + jboss_jndi_port);
025
026                        ctx = new InitialContext();
027
028                } catch (Exception e) {
029                        LOGGER.error("initContext:error:", e);
030                }
031        }
032
033        public ContextAccessWebManager() {
034                try {
035
036                        this.aml = (AccessManagerLocal) ctx
037                                        .lookup("java:global/AuthServices/AccessManager!ru.spb.iac.cud.core.AccessManagerLocal");
038
039                } catch (Exception e) {
040                        LOGGER.error("error:", e);
041                }
042        }
043
044        public String authenticate_cert_sn(String sn, String IPAddress,
045                        String codeSys) throws GeneralFailure, InvalidCredentials,
046                        RevokedCertificate {
047                LOGGER.debug("authenticate_cert_sn");
048                return aml.authenticate_cert_sn(sn, AuthMode.HTTP_REDIRECT, IPAddress,
049                                codeSys);
050        }
051
052        public String authenticate_login(String login, String password,
053                        AuthMode authMode, String IPAddress, String codeSys)
054                        throws GeneralFailure, InvalidCredentials {
055                LOGGER.debug("authenticate_login");
056                return aml.authenticate_login(login, password, authMode, IPAddress,
057                                codeSys);
058        }
059
060        public void sys_audit_public(Long idServ, String inp_param, String result,
061                        String ip_adr, Long idUser, String loginUser, String codeSys) {
062                LOGGER.debug("sys_audit_public");
063                aml.sys_audit_public(idServ, inp_param, result, ip_adr, idUser,
064                                loginUser, codeSys);
065        }
066
067        public String authenticate_uid_obo(String uid, AuthMode authMode,
068                        String IPAddress, String codeSys) throws GeneralFailure {
069                LOGGER.debug("authenticate_uid_obo");
070                return aml.authenticate_uid_obo(uid, authMode, IPAddress, codeSys);
071        }
072}

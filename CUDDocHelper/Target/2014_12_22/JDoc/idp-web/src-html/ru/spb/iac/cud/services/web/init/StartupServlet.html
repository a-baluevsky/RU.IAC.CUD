001package ru.spb.iac.cud.services.web.init;
002
003import javax.servlet.ServletConfig;
004import javax.servlet.ServletContext;
005import javax.servlet.ServletException;
006import javax.servlet.http.HttpServlet;
007
008import org.slf4j.Logger;
009import org.slf4j.LoggerFactory;
010
011import ru.spb.iac.cud.context.ContextAccessWebManager;
012import ru.spb.iac.cud.context.ContextIDPAccessManager;
013import ru.spb.iac.cud.context.ContextIDPUtilManager;
014
015/**
016 * Servlet implementation class StartupServlet
017 */
018 public class StartupServlet extends HttpServlet {
019
020        private static final Logger LOGGER = LoggerFactory.getLogger(StartupServlet.class);
021
022        private static final long serialVersionUID = 1L;
023
024        /**
025         * @see HttpServlet#HttpServlet()
026         */
027        public StartupServlet() {
028                super();
029                // TODO Auto-generated constructor stub
030        }
031
032        /**
033         * @see Servlet#init(ServletConfig)
034         */
035        public void init(ServletConfig config) throws ServletException {
036                try {
037                        ServletContext context = config.getServletContext();
038                        String jboss_jndi_port = context
039                                        .getInitParameter("JBOSS_JNDI_PORT");
040                        ContextAccessWebManager.initContext(jboss_jndi_port);
041                        ContextIDPUtilManager.initContext(jboss_jndi_port);
042                        ContextIDPAccessManager.initContext(jboss_jndi_port);
043
044                        
045                        String signRequired = context.getInitParameter("SIGN_REQUIRED"); 
046                        //String /encryptRequired = /context./getInitParameter/("ENCRYPT_REQUIRED") 
047                        
048                        Configuration.setSignRequired(
049                                                !"FALSE".equals(signRequired)&&!"false".equals(signRequired));
050                
051                        //Configuration/.setEncryptRequired/(
052                        //              !"FALSE"/.equals(encryptRequired)/&&!"false"./equals/(encryptRequired))
053        
054                } catch (Exception e) {
055                        LOGGER.error("init:error:", e);
056                }
057        }
058
059}

094    public void handleRequestType(SAML2HandlerRequest request, SAML2HandlerResponse response) throws ProcessingException {
095        if (request.getSAML2Object() instanceof AuthnRequestType == false){
096            return;}
097
098        if (getType() == HANDLER_TYPE.IDP) {
099            idp.handleRequestType(request, response);
100        } else {
101            sp.handleRequestType(request, response);
102        }
103    }

107        if (request.getSAML2Object() instanceof ResponseType == false) {
108                        return;
109                }
110
111        if (getType() == HANDLER_TYPE.IDP) {
112            idp.handleStatusResponseType(request, response);
113        } else {
114            sp.handleStatusResponseType(request, response);
115        }

120        if (GENERATE_REQUEST_TYPE.AUTH != request.getTypeOfRequestToBeGenerated()) {
121                        return;

126            response.setSendRequest(true);

129            response.setSendRequest(true);
130        }

143        public void handleRequestType(SAML2HandlerRequest request, SAML2HandlerResponse response) throws ProcessingException {
144            HTTPContext httpContext = (HTTPContext) request.getContext();
145            ServletContext servletContext = httpContext.getServletContext();
146
147            AuthnRequestType art = (AuthnRequestType) request.getSAML2Object();
148            if (art == null){
149                throw logger.samlHandlerAuthnRequestIsNull();
150                }
151
152            String destination = art.getAssertionConsumerServiceURL().toASCIIString();
153
154            //urn:eis:sys
155            String issuer = art.getIssuer().getValue();
156            //[urn:eis:sys]http://...
157            String complex_id = "["+issuer+"]"+destination;
158            
159            //art/.isForceAuthn/();
160            //art/.isIsPassive/()
161            
162            logger.trace("Destination = " + destination);
163
164            response.setDestination(destination);
165
166            HttpSession session = BaseSAML2Handler.getHttpSession(request);
167            Principal userPrincipal = (Principal) session.getAttribute(GeneralConstants.PRINCIPAL_ID);
168            if (userPrincipal == null){
169                userPrincipal = httpContext.getRequest().getUserPrincipal();
170                }
171            /*
172             * List/<String> r/oles = (List<String>)/ session/.getAttribute(GeneralConstants.ROLES_ID);
173             */
174            try {
175                
176                //!!!
177                //session/.setAttribute(GeneralConstants.PRINCIPAL_ID, /userPrincipal);
178                
179                /*
180                 * Map/<String,Object> attribs = /(Map<String, Object>) /request.getOptions().get(GeneralConstants.ATTRIBUTES);
181                 * long /assertionValidity = (Long) request.getOptions()/.get(GeneralConstants.ASSERTIONS_VALIDITY); String
182                 * destination = /art.getAssertionConsumerServiceURL()/.toASCIIString(); Document samlResponse =
183                 * this/.getResponse(destination, /userPrincipal, roles, request.getIssuer().getValue(), attribs,
184                 * assertionValidity, art.getID());
185                 */
186
187                Document samlResponse = this.getResponse(request);
188
189               
190                // Update the Identity Server
191                boolean isPost = httpContext.getRequest().getMethod().equalsIgnoreCase("POST");
192                IdentityServer identityServer = (IdentityServer) servletContext.getAttribute(GeneralConstants.IDENTITY_SERVER);
193                // We will try to find URL for global logout from SP metadata (if they are provided) and use SP logout URL
194                // for registration to IdentityServer
195                String participantLogoutURL = getParticipantURL(destination, request);
196                 
197                logger.trace("Participant " + destination + " will be registered to IdentityServer with logout URL "
198                        + participantLogoutURL);
199
200                // If URL is null, participant doesn't support global logout
201                if (participantLogoutURL != null) {
202                   // identityServer/.stack()/.register(session.getId()/, participantLogoutURL, isPost);
203                    identityServer.stack().register(session.getId(), complex_id, isPost);
204                }
205
206                // Check whether we use POST binding for response
207                boolean strictPostBinding = request.getOptions().get(GeneralConstants.SAML_IDP_STRICT_POST_BINDING) != null
208                        && (Boolean) request.getOptions().get(GeneralConstants.SAML_IDP_STRICT_POST_BINDING);
209                boolean postBindingForResponse = isPost || strictPostBinding;
210
211               
212               
213                
214                
215                response.setResultingDocument(samlResponse);
216                response.setRelayState(request.getRelayState());
217                response.setPostBindingForResponse(postBindingForResponse);
218            } catch (Exception e) {
219                logger.samlHandlerAuthenticationError(e);
220                throw logger.processingError(e);
221            }
222        }

224        @SuppressWarnings("unchecked")
225        public Document getResponse(SAML2HandlerRequest request) throws ConfigurationException, ProcessingException {
226          
227                
228                HTTPContext httpContext = (HTTPContext) request.getContext();
229            AuthnRequestType art = (AuthnRequestType) request.getSAML2Object();
230            HttpSession session = BaseSAML2Handler.getHttpSession(request);
231            Principal userPrincipal = (Principal) session.getAttribute(GeneralConstants.PRINCIPAL_ID);
232            if (userPrincipal == null) {
233                                userPrincipal = httpContext.getRequest().getUserPrincipal();
234                        }
235
236            String assertionConsumerURL = art.getAssertionConsumerServiceURL().toASCIIString();
237            List<String> roles = (List<String>) session.getAttribute(GeneralConstants.ROLES_ID);
238            String identityURL = request.getIssuer().getValue();
239            Map<String, Object> attribs = (Map<String, Object>) request.getOptions().get(GeneralConstants.ATTRIBUTES);
240            String requestID = art.getID();
241
242            Document samlResponseDocument = null;
243
244            String authMethod = (String) request.getOptions().get(GeneralConstants.LOGIN_TYPE);
245
246           
247            
248            logger.trace("AssertionConsumerURL=" + assertionConsumerURL);
249
250            ResponseType responseType = null;
251
252            SAML2Response saml2Response = new SAML2Response();
253
254            // Create a response type
255            //<samlp:Response  ID="..."
256            String id = IDGenerator.create("ID_");
257            
258            IssuerInfoHolder issuerHolder = new IssuerInfoHolder(identityURL);
259            issuerHolder.setStatusCode(JBossSAMLURIConstants.STATUS_SUCCESS.get());
260
261            IDPInfoHolder idp = new IDPInfoHolder();
262            idp.setNameIDFormatValue(userPrincipal.getName());
263            idp.setNameIDFormat(JBossSAMLURIConstants.NAMEID_FORMAT_PERSISTENT.get());
264
265            String assertionID = (String) session.getAttribute(GeneralConstants.ASSERTION_ID);
266
267            if (assertionID != null) {
268                // Just renew the assertion
269                AssertionType latestAssertion = (AssertionType) session.getAttribute(GeneralConstants.ASSERTION);
270                if (latestAssertion != null){
271                    idp.setAssertion(latestAssertion);}
272            }
273
274            SPInfoHolder sp = new SPInfoHolder();
275            sp.setResponseDestinationURI(assertionConsumerURL);
276            sp.setRequestID(requestID);
277            sp.setIssuer(art.getIssuer().getValue());
278            responseType = saml2Response.createResponseType(id, sp, idp, issuerHolder);
279
280           
281            
282            // Add information on the roles
283            AssertionType assertion = responseType.getAssertions().get(0).getAssertion();
284
285            
286            // Create an AuthnStatementType
287            if (handlerConfig.getParameter(DISABLE_AUTHN_STATEMENT) == null) {
288                String authContextRef = JBossSAMLURIConstants.AC_PASSWORD.get();
289                if (isNotNull(authMethod)){
290                    authContextRef = authMethod;}
291
292                AuthnStatementType authnStatement = StatementUtil.createAuthnStatement(XMLTimeUtil.getIssueInstant(),
293                        authContextRef);
294
295                  authnStatement.setSessionIndex(assertion.getID());
296
297                assertion.addStatement(authnStatement);
298            }
299
300            AttributeStatementType attrStatement = null;
301            if (handlerConfig.getParameter(DISABLE_SENDING_ROLES) == null && (roles != null && !roles.isEmpty())) {
302                
303                // AttributeStatementType /attrStatement = /null;
304                if (handlerConfig.getParameter(USE_MULTI_VALUED_ROLES) != null) {
305                    // attrStatement /= S/tatementUtil/.createAttributeStatementForRoles(roles, true);
306                     attrStatement = CUDStatementUtil.createAttributeStatementForRoles(roles, true);
307                     
308                } else {
309                    attrStatement = StatementUtil.createAttributeStatement(roles);
310                }
311                if (attrStatement != null) {
312                    assertion.addStatement(attrStatement);
313                }
314             }
315            
316            List<String> resources = (List<String>) session.getAttribute("CUD_RESOURCES");
317            
318             if ((resources != null && !resources.isEmpty())) {
319                
320                 attrStatement = CUDStatementUtil.createAttributeStatementForResources(resources, true);
321               
322                if (attrStatement != null) {
323                    assertion.addStatement(attrStatement);
324                }
325             }
326            
327            if (attribs != null){
328                
329                try{
330                        
331                
332                 KeyPair keypair = (KeyPair) handlerChainConfig.getParameter(GeneralConstants.KEYPAIR);
333                
334                 if (keypair == null) {
335                     logger.samlHandlerKeyPairNotFound();
336                     throw logger.samlHandlerKeyPairNotFoundError();
337                 }
338                 
339                 ConditionsType conditions = assertion.getConditions();
340                 String lifetime = Long.toString(
341                                 conditions.getNotOnOrAfter().toGregorianCalendar().getTimeInMillis());
342                 String userUID = (String) attribs.get("USER_UID");
343                 
344                 //имеющийся тип аутентификации
345                 String main_auth_type = (String) session.getAttribute("cud_auth_type");
346                                
347                 LOGGERSLF4J.debug("getResponse:01:"+main_auth_type);
348                 
349              
350                 StringBuilder sb = new StringBuilder();
351                 
352                 sb.append(userUID).append("_").append(lifetime).append("_").append(main_auth_type!=null?main_auth_type:auth_type_password);
353                 
354                 byte[] sigValue = GOSTSignatureUtil.sign(sb.toString(), keypair.getPrivate());
355                 
356                 String base64SigValue = Base64.encodeBytes(sigValue, Base64.DONT_BREAK_LINES);
357                 
358                 String tokenID = sb.toString()+"_"+base64SigValue;
359                 
360                 String base64tokenID = Base64.encodeBytes(tokenID.getBytes("utf-8"), Base64.DONT_BREAK_LINES);
361                 
362                 attribs.put("TOKEN_ID", base64tokenID);
363
364                 LOGGERSLF4J.debug("getResponse:02:"+tokenID);
365                                 LOGGERSLF4J.debug("getResponse:03:"+base64tokenID);
366                 
367                                 
368                                 (new ContextIDPAccessManager())
369                                   .saveTokenID(base64tokenID, userUID);
370                                 
371                }catch(Exception e){
372                        throw new ProcessingException("ERROR Generate TOKEN_ID:"+e);
373                }
374             }
375        
376            
377            
378            // Add in the attributes information
379            if (attribs != null && attribs.size() > 0) {
380                // AttributeStatementType /attStatement /= StatementUtil.createAttributeStatement(attribs);
381                // assertion/.addStatement(attStatement)/;
382                 
383                AttributeStatementType attStatement = StatementUtil.createAttributeStatement(attribs);
384                if(attrStatement==null){
385                   assertion.addStatement(attStatement);
386                }else{
387                  List <ASTChoiceType> list_attr = attStatement.getAttributes();
388                  
389                  for(ASTChoiceType attribute :list_attr){
390                          attrStatement.addAttribute(attribute);
391                  }
392                  
393                }
394              }
395
396            // Add assertion to the session
397            session.setAttribute(GeneralConstants.ASSERTION, assertion);
398
399            Map<String, Object> requestOptions = request.getOptions();
400            PicketLinkAuditHelper auditHelper = (PicketLinkAuditHelper) requestOptions.get(GeneralConstants.AUDIT_HELPER);
401            if (auditHelper != null) {
402                PicketLinkAuditEvent auditEvent = new PicketLinkAuditEvent(AuditLevel.INFO);
403                auditEvent.setWhoIsAuditing((String) requestOptions.get(GeneralConstants.CONTEXT_PATH));
404                auditEvent.setType(PicketLinkAuditEventType.CREATED_ASSERTION);
405                auditEvent.setAssertionID(id);
406                auditHelper.audit(auditEvent);
407            }
408
409           
410            try {
411                samlResponseDocument = saml2Response.convert(responseType);
412
413                if (logger.isTraceEnabled()) {
414                    logger.trace("SAML Response Document: " + DocumentUtil.asString(samlResponseDocument));
415                }
416            } catch (Exception e) {
417                throw logger.samlAssertionMarshallError(e);
418            }
419
420            return samlResponseDocument;
421        }

423        private String getParticipantURL(String destination, SAML2HandlerRequest request) {
424            SPSSODescriptorType spMetadata = (SPSSODescriptorType) request.getOptions().get(
425                    GeneralConstants.SP_SSO_METADATA_DESCRIPTOR);
426
427            // Metadata not found. We will use destination for registration to IdentityServer
428            if (spMetadata == null) {
429                return destination;
430            }
431
432            List<EndpointType> logoutEndpoints = spMetadata.getSingleLogoutService();
433
434            // If endpoint not found, we assume that SP doesn't support logout profile
435            if (logoutEndpoints == null || logoutEndpoints.size() == 0) {
436                return null;
437            }
438
439            // Use first endpoint for now (Maybe later we can find logoutType according to bindingType from SAMLRequest)
440            EndpointType logoutEndpoint = logoutEndpoints.get(0);
441            return logoutEndpoint.getLocation().toASCIIString();
442        }

445    private class SPAuthenticationHandler {
446        public void generateSAMLRequest(SAML2HandlerRequest request, SAML2HandlerResponse response) throws ProcessingException {
447            String issuerValue = request.getIssuer().getValue();
448
449            SAML2Request samlRequest = new SAML2Request();
450            String id = IDGenerator.create("ID_");
451
452            String assertionConsumerURL = (String) handlerConfig.getParameter(SAML2Handler.ASSERTION_CONSUMER_URL);
453            if (StringUtil.isNullOrEmpty(assertionConsumerURL)) {
454                assertionConsumerURL = issuerValue;
455            }
456
457            // Check if there is a nameid policy
458            String nameIDFormat = (String) handlerConfig.getParameter(GeneralConstants.NAMEID_FORMAT);
459            if (isNotNull(nameIDFormat)) {
460                samlRequest.setNameIDFormat(nameIDFormat);
461            }
462            try {
463                AuthnRequestType authn = samlRequest.createAuthnRequestType(id, assertionConsumerURL,
464                        response.getDestination(), issuerValue);
465
466                createRequestedAuthnContext(authn);
467
468                String bindingType = getSPConfiguration().getBindingType();
469                boolean isIdpUsesPostBinding = getSPConfiguration().isIdpUsesPostBinding();
470
471                if (bindingType != null) {
472                    if (bindingType.equals("POST") || isIdpUsesPostBinding) {
473                        authn.setProtocolBinding(URI.create(JBossSAMLURIConstants.SAML_HTTP_POST_BINDING.get()));
474                    } else if (bindingType.equals("REDIRECT")) {
475                        authn.setProtocolBinding(URI.create(JBossSAMLURIConstants.SAML_HTTP_REDIRECT_BINDING.get()));
476                    } else {
477                        throw logger.samlInvalidProtocolBinding();
478                    }
479                }
480
481                response.setResultingDocument(samlRequest.convert(authn));
482                response.setSendRequest(true);
483
484                Map<String, Object> requestOptions = request.getOptions();
485                PicketLinkAuditHelper auditHelper = (PicketLinkAuditHelper) requestOptions.get(GeneralConstants.AUDIT_HELPER);
486                if (auditHelper != null) {
487                    PicketLinkAuditEvent auditEvent = new PicketLinkAuditEvent(AuditLevel.INFO);
488                    auditEvent.setWhoIsAuditing((String) requestOptions.get(GeneralConstants.CONTEXT_PATH));
489                    auditEvent.setType(PicketLinkAuditEventType.CREATED_ASSERTION);
490                    auditEvent.setAssertionID(id);
491                    auditHelper.audit(auditEvent);
492                }
493
494                // Save AuthnRequest ID into sharedState, so that we can later process it by another handler
495                request.addOption(GeneralConstants.AUTH_REQUEST_ID, id);
496            } catch (Exception e) {
497                throw logger.processingError(e);
498            }
499        }

501        public void handleStatusResponseType(SAML2HandlerRequest request, SAML2HandlerResponse response)
502                throws ProcessingException {
503            HTTPContext httpContext = (HTTPContext) request.getContext();
504            ResponseType responseType = (ResponseType) request.getSAML2Object();
505            List<RTChoiceType> assertions = responseType.getAssertions();
506            if (assertions.size() == 0) {
507                                throw logger.samlHandlerNoAssertionFromIDP();
508                        }
509
510            PrivateKey privateKey = (PrivateKey) request.getOptions().get(GeneralConstants.DECRYPTING_KEY);
511
512            Object assertion = assertions.get(0).getEncryptedAssertion();
513            if (assertion instanceof EncryptedAssertionType) {
514                responseType = this.decryptAssertion(responseType, privateKey);
515                assertion = responseType.getAssertions().get(0).getAssertion();
516            }
517            if (assertion == null) {
518                assertion = assertions.get(0).getAssertion();
519            }
520
521            request.addOption(GeneralConstants.ASSERTION, assertion);
522
523            Principal userPrincipal = handleSAMLResponse(responseType, response);
524            if (userPrincipal == null) {
525                response.setError(403, "User Principal not determined: Forbidden");
526            } else {
527                HttpSession session = httpContext.getRequest().getSession(false);
528
529                // add the principal to the session
530                session.setAttribute(GeneralConstants.PRINCIPAL_ID, userPrincipal);
531
532                Document assertionDocument = AssertionUtil.asDocument((AssertionType) assertion);
533
534                String assertionAttributeName = (String) handlerConfig
535                        .getParameter(GeneralConstants.ASSERTION_SESSION_ATTRIBUTE_NAME);
536
537                if (assertionAttributeName != null) {
538                    session.setAttribute(assertionAttributeName, assertionDocument);
539                }
540
541                session.setAttribute(GeneralConstants.ASSERTION_SESSION_ATTRIBUTE_NAME, assertionDocument);
542            }
543        }

548        private ResponseType decryptAssertion(ResponseType responseType, PrivateKey privateKey) throws ProcessingException {
549            if (privateKey == null) {
550                                throw logger.nullArgumentError("privateKey");
551                        }
552            
553         
554            
555            SAML2Response saml2Response = new SAML2Response();
556            try {
557                Document doc = saml2Response.convert(responseType);
558
559                Element enc = DocumentUtil.getElement(doc, new QName(JBossSAMLConstants.ENCRYPTED_ASSERTION.get()));
560                if (enc == null){
561                    throw logger.samlHandlerNullEncryptedAssertion();}
562                String oldID = enc.getAttribute(JBossSAMLConstants.ID.get());
563                Document newDoc = DocumentUtil.createDocument();
564                Node importedNode = newDoc.importNode(enc, true);
565                newDoc.appendChild(importedNode);
566
567                Element decryptedDocumentElement = XMLEncryptionUtil.decryptElementInDocument(newDoc, privateKey);
568                SAMLParser parser = new SAMLParser();
569
570                JAXPValidationUtil.checkSchemaValidation(decryptedDocumentElement);
571                AssertionType assertion = (AssertionType) parser.parse(StaxParserUtil.getXMLEventReader(DocumentUtil
572                        .getNodeAsStream(decryptedDocumentElement)));
573
574                responseType.replaceAssertion(oldID, new RTChoiceType(assertion));
575                return responseType;
576            } catch (Exception e) {
577                throw logger.processingError(e);
578            }

581        private Principal handleSAMLResponse(ResponseType responseType, SAML2HandlerResponse response)
582                throws ProcessingException {
583            if (responseType == null){
584                throw logger.nullArgumentError("response type");}
585
586            StatusType statusType = responseType.getStatus();
587            if (statusType == null){
588                throw logger.nullArgumentError("Status Type from the IDP");}
589
590            String statusValue = statusType.getStatusCode().getValue().toASCIIString();
591            if (JBossSAMLURIConstants.STATUS_SUCCESS.get().equals(statusValue) == false)
592            { throw logger.samlHandlerIDPAuthenticationFailedError();}
593
594            List<RTChoiceType> assertions = responseType.getAssertions();
595            if (assertions.size() == 0)
596            { throw logger.samlHandlerNoAssertionFromIDP();}
597
598            AssertionType assertion = assertions.get(0).getAssertion();
599            // Check for validity of assertion
600            boolean expiredAssertion;
601            try {
602                String skew = (String) handlerConfig.getParameter(SAML2Handler.CLOCK_SKEW_MILIS);
603                if (isNotNull(skew)) {
604                    long skewMilis = Long.parseLong(skew);
605                    expiredAssertion = AssertionUtil.hasExpired(assertion, skewMilis);
606                } else {
607                                        expiredAssertion = AssertionUtil.hasExpired(assertion);
608                                }
609            } catch (ConfigurationException e) {
610                throw new ProcessingException(e);
611            }
612            if (expiredAssertion) {
613                AssertionExpiredException aee = new AssertionExpiredException();
614                aee.setId(assertion.getID());
615                throw logger.assertionExpiredError(aee);
616            }
617
618            SubjectType subject = assertion.getSubject();
619            /*
620             * JAXBElement<NameIDType> jnameID = (JAXBElement<NameIDType>) subject.getContent().get(0); NameIDType nameID =
621             * jnameID.getValue();
622             */
623            if (subject == null) {
624                                throw logger.nullValueError("Subject in the assertion");
625                        }
626
627            STSubType subType = subject.getSubType();
628            if (subType == null){
629                throw logger.nullValueError("Unable to find subtype via subject");}
630            NameIDType nameID = (NameIDType) subType.getBaseID();
631
632            if (nameID == null){
633                throw logger.nullValueError("Unable to find username via subject");}
634
635            final String userName = nameID.getValue();
636            List<String> roles = new ArrayList<String>();
637
638            // Let us get the roles
639            Set<StatementAbstractType> statements = assertion.getStatements();
640            for (StatementAbstractType statement : statements) {
641                if (statement instanceof AttributeStatementType) {
642                    AttributeStatementType attributeStatement = (AttributeStatementType) statement;
643                    roles.addAll(getRoles(attributeStatement));
644                }
645            }
646
647            response.setRoles(roles);
648
649            Principal principal = new SerializablePrincipal(userName);
650
651            if (handlerChainConfig.getParameter(GeneralConstants.ROLE_VALIDATOR_IGNORE) == null) {
652                // Validate the roles
653                IRoleValidator roleValidator = (IRoleValidator) handlerChainConfig
654                        .getParameter(GeneralConstants.ROLE_VALIDATOR);
655                if (roleValidator == null) {
656                                        throw logger.nullValueError("Role Validator");
657                                }
658
659                boolean validRole = roleValidator.userInRole(principal, roles);
660
661                if (!validRole) {
662                    logger.trace("Invalid role: " + roles);
663                    principal = null;
664                }
665            }
666            return principal;
667        }

669        /**
670         * Get the roles from the attribute statement
671         *
672         * @param attributeStatement
673         * @return
674         */
675        private List<String> getRoles(AttributeStatementType attributeStatement) {
676            List<String> roles = new ArrayList<String>();
677
678            // PLFED-141: Disable role picking from IDP response
679            if (handlerConfig.containsKey(DISABLE_ROLE_PICKING)) {
680                String val = (String) handlerConfig.getParameter(DISABLE_ROLE_PICKING);
681                if (isNotNull(val) && "true".equalsIgnoreCase(val)){
682                    return roles;}
683            }
684
685            // PLFED-140: which of the attribute statements represent roles?
686            List<String> roleKeys = new ArrayList<String>();
687
688            if (handlerConfig.containsKey(ROLE_KEY)) {
689                String roleKey = (String) handlerConfig.getParameter(ROLE_KEY);
690                if (isNotNull(roleKey)) {
691                    roleKeys.addAll(StringUtil.tokenize(roleKey));
692                }
693            }
694
695            List<ASTChoiceType> attList = attributeStatement.getAttributes();
696            for (ASTChoiceType obj : attList) {
697                AttributeType attr = obj.getAttribute();
698                if (roleKeys.size() > 0) {
699                    if (!roleKeys.contains(attr.getName())){
700                        continue;}
701                }
702                List<Object> attributeValues = attr.getAttributeValue();
703                if (attributeValues != null) {
704                    for (Object attrValue : attributeValues) {
705                        if (attrValue instanceof String) {
706                            roles.add((String) attrValue);
707                        } else if (attrValue instanceof Node) {
708                            Node roleNode = (Node) attrValue;
709                            roles.add(roleNode.getFirstChild().getNodeValue());
710                        } else{
711                            throw logger.unsupportedRoleType(attrValue);}
712                    }
713                }
714            }
715            return roles;
716        }

719            SPType spConfiguration = (SPType) handlerChainConfig.getParameter(GeneralConstants.CONFIGURATION);

729    private void createRequestedAuthnContext(final AuthnRequestType authn) {
730        String authnContextClasses = (String) handlerConfig.getParameter(GeneralConstants.AUTHN_CONTEXT_CLASSES);
731
732        if (isNotNull(authnContextClasses)) {
733            RequestedAuthnContextType requestAuthnContext = new RequestedAuthnContextType();
734
735            for (String classFqn: authnContextClasses.split(",")) {
736                SAMLAuthenticationContextClass standardClass = SAMLAuthenticationContextClass.forAlias(classFqn);
737
738                if (standardClass != null) {
739                    classFqn = standardClass.getFqn();
740                }
741
742                requestAuthnContext.addAuthnContextClassRef(classFqn);
743            }
744
745            if (!requestAuthnContext.getAuthnContextClassRef().isEmpty()) {
746                String comparison = (String) handlerConfig.getParameter(GeneralConstants.REQUESTED_AUTHN_CONTEXT_COMPARISON);
747
748                if (isNotNull(comparison)) {
749                    requestAuthnContext.setComparison(AuthnContextComparisonType.fromValue(comparison));
750                }
751
752                authn.setRequestedAuthnContext(requestAuthnContext);
753            } else {
754                logger.debug("RequestedAuthnContext not set for AuthnRequest. No class was provided.");
755            }
756        }

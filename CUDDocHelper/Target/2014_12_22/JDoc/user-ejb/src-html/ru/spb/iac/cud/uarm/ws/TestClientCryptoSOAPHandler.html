042 public class TestClientCryptoSOAPHandler implements SOAPHandler<SOAPMessageContext>{
043
044        private static final Logger LOGGER = LoggerFactory.getLogger(TestClientCryptoSOAPHandler.class);
045        
046          private static PublicKey publicKey = null;
047            
048            private static PrivateKey privateKey = null;
049            
050
051
052 public Set<QName> getHeaders() {
053    return null;
054 }
055
056

058         LOGGER.debug("close");
059 }

063    return true;

066
067 public boolean handleMessage(SOAPMessageContext mc) {
068         
069         LOGGER.debug("TestClientCryptoSOAPHandler:handleMessage:01:"+mc.get(MessageContext.MESSAGE_OUTBOUND_PROPERTY));
070        
071         
072         
073         try{
074                        SOAPMessage sm = mc.getMessage();
075                        
076            
077                    SOAPHeader header = sm.getSOAPHeader();
078                    SOAPBody body = sm.getSOAPBody();
079                 
080                    char[] signingKeyPass="Access_Control".toCharArray();
081                        String signingAlias="cudvm_export";
082                
083                        if(publicKey==null) {   
084                        KeyStore ks  = KeyStore.getInstance("HDImageStore", "JCP");
085                        ks.load(null, null);
086                        
087                        privateKey = (PrivateKey)ks.getKey(signingAlias, signingKeyPass);
088                        
089                        Certificate cert = ks.getCertificate(signingAlias);
090                        publicKey = cert.getPublicKey() ;
091                
092                        }
093                //запрос 
094          if (Boolean.TRUE.equals(mc.get(MessageContext.MESSAGE_OUTBOUND_PROPERTY))) {
095        
096        
097                
098                
099
100                if(!org.apache.xml.security.Init.isInitialized()){
101                  org.apache.xml.security.Init.init();
102                }
103                  
104            Provider xmlDSigProvider = new ru.CryptoPro.JCPxml.dsig.internal.dom.XMLDSigRI();
105          
106            XMLSignatureFactory fac = XMLSignatureFactory.getInstance("DOM", xmlDSigProvider);
107
108             
109            Node SecuritySOAP = header.getFirstChild();
110            
111            
112            
113            
114            body.addNamespaceDeclaration("wsu", "http://docs.oasis-open.org/wss/2004/01/oasis-200401-wss-wssecurity-utility-1.0.xsd");
115            body.setAttributeNS("http://docs.oasis-open.org/wss/2004/01/oasis-200401-wss-wssecurity-utility-1.0.xsd", "wsu:Id","Body");
116          
117            
118            header.addNamespaceDeclaration("wsu", "http://docs.oasis-open.org/wss/2004/01/oasis-200401-wss-wssecurity-utility-1.0.xsd");
119            header.setAttributeNS("http://docs.oasis-open.org/wss/2004/01/oasis-200401-wss-wssecurity-utility-1.0.xsd", "wsu:Id","Header");
120        
121            
122                
123            List<Transform> transformList = new ArrayList<Transform>();
124                Transform transform = fac.newTransform(Transform.ENVELOPED, (XMLStructure) null);
125                Transform transformC14N = fac.newTransform(Transforms.TRANSFORM_C14N_EXCL_OMIT_COMMENTS, (XMLStructure) null);
126                transformList.add(transform);
127                transformList.add(transformC14N);
128                
129                
130                 Reference ref1 = fac.newReference("#Header", fac.newDigestMethod("http://www.w3.org/2001/04/xmldsig-more#gostr3411", null) 
131                                ,transformList, null, null);
132                Reference ref2 = fac.newReference("#Body", fac.newDigestMethod("http://www.w3.org/2001/04/xmldsig-more#gostr3411", null) 
133                                ,transformList, null, null);
134                List<Reference> referenceList = new ArrayList<Reference>();
135                
136                referenceList.add(ref1);
137                referenceList.add(ref2);
138                
139                SignedInfo si = fac.newSignedInfo( fac.newCanonicalizationMethod(CanonicalizationMethod.EXCLUSIVE,
140                                                                                                        (C14NMethodParameterSpec) null),
141                                                                                           fac.newSignatureMethod("http://www.w3.org/2001/04/xmldsig-more#gostr34102001-gostr3411", null),
142                                                                                          referenceList);
143
144                        KeyInfoFactory kif = fac.getKeyInfoFactory();
145                        
146                        KeyValue kv = kif.newKeyValue(publicKey);
147                        KeyInfo ki = kif.newKeyInfo(Collections.singletonList(kv));
148                        
149                    javax.xml.crypto.dsig.XMLSignature sig = fac.newXMLSignature(si, ki);
150
151                        //куда вставлять подпись
152                        DOMSignContext signContext = new DOMSignContext(privateKey, SecuritySOAP); 
153                        
154                        signContext.putNamespacePrefix(XMLSignature.XMLNS, "dsig");
155                        
156                        //фиксация аттрибута id в подписываемом элементе
157                        //место ответственное за факт появления Pre-digested input в логе
158                
159                        signContext.setIdAttributeNS(body, "http://docs.oasis-open.org/wss/2004/01/oasis-200401-wss-wssecurity-utility-1.0.xsd", "Id");
160                        signContext.setIdAttributeNS(header, "http://docs.oasis-open.org/wss/2004/01/oasis-200401-wss-wssecurity-utility-1.0.xsd", "Id");
161                
162                
163                        
164                         
165                            
166            sig.sign(signContext);
167                    
168          
169            
170            
171            
172    }else{
173    //ответ
174        
175         
176        
177         Provider xmlDSigProvider = new ru.CryptoPro.JCPxml.dsig.internal.dom.XMLDSigRI();
178          
179         XMLSignatureFactory fac = XMLSignatureFactory.getInstance("DOM", xmlDSigProvider);
180                
181         Node securityNode = header.getFirstChild(); 
182         
183         if(securityNode==null){
184                 LOGGER.debug("TestClientCryptoSOAPHandler:handleMessage:06");
185                throw new Exception("This service requires <wsse:Security>, which is missing!!!");
186         }
187         
188         NodeList securityNodeChilds = securityNode.getChildNodes();
189
190         Node signatureNode1 = null;
191         
192         for (int i = 0; i<securityNodeChilds.getLength(); i++) {
193                 
194                 
195                 
196                 if(securityNodeChilds.item(i).getLocalName()!=null&&securityNodeChilds.item(i).getLocalName().equals("Signature")){
197                         signatureNode1 = securityNodeChilds.item(i);
198                 }
199         }
200         
201         
202         
203         //signature
204         
205         if(signatureNode1==null){
206                 LOGGER.debug("TestClientCryptoSOAPHandler:handleMessage:09");
207                throw new Exception("This service requires <dsig:Signature>, which is missing!!!");
208         }
209
210        
211         
212        
213     
214         
215        DOMValidateContext valContext1 = new DOMValidateContext(publicKey, signatureNode1);
216        
217        valContext1.putNamespacePrefix(XMLSignature.XMLNS, "dsig");
218         
219        valContext1.setIdAttributeNS(header, "http://docs.oasis-open.org/wss/2004/01/oasis-200401-wss-wssecurity-utility-1.0.xsd", "Id");
220        valContext1.setIdAttributeNS(body, "http://docs.oasis-open.org/wss/2004/01/oasis-200401-wss-wssecurity-utility-1.0.xsd", "Id");
221      
222                    
223               javax.xml.crypto.dsig.XMLSignature signature1 = fac.unmarshalXMLSignature(valContext1);
224         
225                boolean result1 = signature1.validate(valContext1);
226                         
227                 LOGGER.debug("TestClientCryptoSOAPHandler:handleMessage:010+:"+result1);
228
229        
230    }
231
232         } catch (Exception e) {
233                 LOGGER.error("TestClientCryptoSOAPHandler:handleMessage:error:"+e);
234         throw new ProtocolException(e);
235     }
236    return true;
237 }
